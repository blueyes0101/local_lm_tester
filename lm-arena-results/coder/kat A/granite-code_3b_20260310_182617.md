        recipe,
        score,
        comment
      });
      recipe.ratings.push(rating);
      await recipe.save();
      return rating;
    },
    updateRating: async (parent, { recipeId, score, comment }, { loaders }) => {
      const rating = await RecipeRating.findOneAndUpdate({
        recipeId,
        score
      }, {
        $set: {
          score,
          comment
        }
      });
      return rating;
    }
  },

        return ingredients.map(ingredient => ingredient.name);
      }),
    averageRating: (parent, args, { loaders }) =>
      loaders.recipeRatings.load(parent.id).then(ratings => {
        return ratings.reduce((acc, rating) => acc + rating.score, 0) / ratings.length;
      })
