```typescript
    const { workspaceId, taskId } = req.params;

    if (!Number.isInteger(Number(workspaceId)) || !Number.isInteger(Number(taskId))) {
      return res.status(400).json({ error: 'workspaceId and taskId must be integers' });

      // Validate assignee_id
      if (result.assignee_id && !await Task.findWhere({
        where: { id: result.assignee_id },
        select: 'workspace_id'
      })) {
        return res.status(400).json({ error: 'Assignee does not belong to the workspace' });
      }

      // Update task
      const updatedTask = await Task.update({
        title: result.title,
        status: result.status
      }, { where: { id: Number(taskId) } });

      if (updatedTask.rows.length === 0) {
        return res.status(404).json({ error: 'Task not found' });
      }

      // Check for cancelled status with open subtasks
      const hasOpenSubtasks = await Task.count({
        where: { id: Number(taskId) },
        subquery: { where: { status: 'cancelled', completed_tasks_count_gt: 0 } }}
      );

      if (result.status === 'cancelled' && hasOpenSubtasks > 0) {
        return res.status(409).json({ error: 'Cannot cancel task with open subtasks' });

      return res.json(updatedTask.rows[0]);
```

```python
# Data structure: {
#   "token": "<token>",
#   "issued_at": timestamp,
#   "expires_at": timestamp,
#   "blacklisted": boolean (false by default),
#   "user_id": <unique user ID>
# }

class TokenStore:
    def __init__(self, redis_client):
        self.redis = redis_client

    async def store_token(self, token: str, user_id: str) -> None:
        """Store a new refresh token with its expiry and blacklisting status"""
        expire_redis_key(f"rt:{token}", 3600)
        await self.redis.set(f"rt:{token}", json.dumps({
            "issued_at": time.time(),
            "expires_at": time.time() + 3600,
            "user_id": user_id
        }))

    async def rotate_refresh_token(self, old_token: str) -> None:
        """Rotate an existing refresh token by generating a new one"""
        expire_redis_key(f"rt:{old_token}", 3600)
        
        # Generate new refresh token
        new_token = generate_new_refresh_token()
        await self.redis.set(f"rt:{new_token}", json.dumps({
            "issued_at": time.time(),
            "expires_at": time.time() + 3600,
            "user_id": user_id  # Get user ID from session or elsewhere
        }))

    async def blacklist_token(self, token: str) -> None:
        """Mark a token as blacklisted"""
        await self.redis.set(f"rt:{token}", json.dumps({
            "blacklisted": True,
            "expires_at": time.time() + 3600
```

```python
        """
        Check if the same refresh token has been used within a window
```

6. **Token Rotation vs Blacklisting**:

- **Rotation**: 
  - Only valid for a specific period (TTL)
  - Tokens can be rotated multiple times within the TTL
  - Used when you want to periodically refresh or change tokens without requiring explicit blacklisting
  
- **Blacklisting**:
  - Permanently marks a token as invalid and unusable
  - Requires manual intervention to add/remove tokens from blacklist
  - Used when you need to permanently revoke access due to security breaches, compromised credentials, etc.

When to use each:

- Use rotation for periodic refreshes or changes where tokens can be rotated multiple times within the TTL.
- Use blacklisting for permanent revocation of token usage, especially in cases involving security compromises or credential resets.

```javascript
// Server Implementation
const WebSocket = require('ws');
const uuid = require('uuid');

class DocumentEditorServer {
  constructor(port) {
    this.clients = new Map();
    this.documentState = {};
    
    // Create WSS server
    const wss = new WebSocket.Server({ port });
    
    // Connection handling
    wss.on('connection', (ws, req) => {
      const userId = uuid.v4(); // Generate unique user ID
      
      // Associate connection with user and document
      this.clients.set(userId, [ws]);
      
      ws.on('close', () => {
        this.clients.get(userId).pop().remove();
        if (this.clients.get(userId).length === 0) {
          this.clients.delete(userId);
        }
      });
    });

    // Message handling
    wss.on('message', (msg) => {
      try {
        const data = JSON.parse(msg);
        
        switch(data.type) {
          case 'JOIN':
            return this.handleJoin(userId, data.documentId, ws);
            
          case 'TEXT_CHANGE':
            return this.handleTextChange(userId, data.position, data.text, ws);
            
          case 'CURSOR_MOVE':
            return this.handleCursorMove(userId, data.cursorPosition, ws);
            
          case 'USER_LEFT':
            return this.handleUserLeft(userId, ws);
        }
      } catch (e) {
        console.error('Message parsing error:', e);
      }
    });
  }

  handleJoin(userId, documentId, ws) {
    // Create new client array for user
    const clients = this.clients.get(userId) || [];
    clients.push(ws);
    
    // Update document state with new cursor position
    this.documentState[documentId] = this.documentState[documentId] || {};
    this.documentState[documentId][userId] = { cursorPosition: ws._socket._socket.remoteAddress };
  }

  handleTextChange(userId, position, text, ws) {
    // Update document state with new text change
    this.documentState[position] = text;
    
    // Broadcast text change to all clients in the same document
    const documentId = Object.keys(this.documentState).find(doc => doc === position);
    if (documentId) {
      this.clients.get(userId).forEach(client => {
        if (client !== ws) {
          client.send({
            type: 'TEXT_CHANGE',
            userId,
            position,
            text
          });
        }
      });
    }
  }

  handleCursorMove(userId, cursorPosition, ws) {
    // Update document state with new cursor position
    this.documentState[userId][cursorPosition] = cursorPosition;
    
    // Broadcast cursor move to all clients in the same document except user A
    const userIdsInDocument = Object.keys(this.documentState).reduce(
      (ids, docId) => [...ids, ...this.documentState[docId].map(u => u.userId)],
      []
    );
    
    if (userIdsInDocument.length > 1) {
      this.clients.get(userId).forEach(client => {
        if (client !== ws && !userIdsInDocument.includes(client.userId)) {
          client.send({
            type: 'CURSOR_MOVE',
            userId,
            cursorPosition
          });
        }
      });
    }
  }

  handleUserLeft(userId, ws) {
    // Remove user from document state and client array
    this.clients.get(userId).pop().remove();
    if (this.clients.get(userId).length === 0) {
      this.clients.delete(userId);
    }
    
    // Broadcast user left event to all clients in the same document
    const documentId = Object.keys(this.documentState).find(doc => doc === userId);
    if (documentId) {
      this.clients.get(userId).forEach(client => {
        client.send({
          type: 'USER_LEFT',
          userId
        });
      });
    }
  }

  handleReconnection(userId, ws) {
    // Get latest document state for the user
    const clients = this.clients.get(userId);
    if (clients && clients.length > 0) {
      clients[clients.length - 1].send({
        type: 'JOIN',
        userId,
        documentId: Object.keys(this.documentState).find(doc => doc === userId),
        data: this.documentState[userId]
      });
    }
  }

  broadcastChange(userId, message) {
    // Broadcast change to all clients in the same document
    const documentId = Object.keys(this.documentState).find(doc => doc === userId);
    if (documentId && this.clients.get(userId)) {
      this.clients.get(userId).forEach(client => {
        client.send(message);
      });
    }
  }

  broadcastDocumentState() {
    // Broadcast updated document state to all clients
    const documents = Object.keys(this.documentState);
    for (const doc of documents) {
      this.broadcastChange(doc, {
        type: 'DOCUMENT_STATE',
        data: this.documentState[doc]
      });
    }
  }
}
```

5. GraphQL vs REST:
Choose GraphQL when:
- The client needs to fetch multiple related resources at once
- You want to implement dynamic APIs where the schema changes based on user input
- You need to optimize bandwidth by reducing unnecessary queries
- You want to implement real-time updates or subscriptions

Don't choose GraphQL when:
- Your API is simple and mostly CRUD operations
- The client can be well-served with REST endpoints
- Your data model doesn't fit well into a schema-based system like GraphQL

When NOT choosing GraphQL:
```plaintext
REST is better for scenarios where you have simple, predictable APIs with clear endpoints. GraphQL might add complexity without providing significant benefits in these cases.
```

The decision should be based on the specific use case and requirements of your application. If you need flexibility in how clients access your data, then GraphQL might be a good choice. Otherwise, REST is usually sufficient and simpler to implement.
