function DataTable<T extends object>({
  columns,
  data,
  onRowClick,
  onSelectionChange,
  selectedRows,
  pageSize = 10,
  pageSizeOptions = [10, 25, 50],
  currentPage = 1,
  onPageChange,
  sortable = false,
  onSort,
  renderCell,
}: DataTableProps<T>): JSX.Element {
  const [sortedData, setSortedData] = useState<T[]>(data);
  const [selectedRowKeys, setSelectedRowKeys] = useState<string[]>([]);
  const [pageSizeState, setPageSizeState] = useState<number>(pageSize);
  const [currentPageState, setCurrentPageState] = useState<number>(currentPage);

  useEffect(() => {
    setSortedData(data);
  }, [data]);

  useEffect(() => {
    setSelectedRowKeys(selectedRows.map((row) => row.key));
  }, [selectedRows]);

  const handleRowClick = (row: T) => {
    onRowClick?.(row);
  };

  const handleSelectionChange = (row: T, checked: boolean) => {
    setSelectedRowKeys((prevKeys) => {
      if (checked) {
        return [...prevKeys, row.key];
      } else {
        return prevKeys.filter((key) => key !== row.key);
      }
    });
  };

  const handlePageChange = (page: number) => {
    setCurrentPageState(page);
    onPageChange?.(page);
  };

  const handlePageSizeChange = (size: number) => {
    setPageSizeState(size);
    onPageSizeChange?.(size);
  };

  const handleSort = (column: DataTableColumn<T>, direction: 'asc' | 'desc') => {
    setSortedData((prevData) => {
      return [...prevData].sort((a, b) => {
        const sortValueA = a[column.key];
        const sortValueB = b[column.key];
        if (sortValueA < sortValueB) {
          return -1;
        } else if (sortValueA > sortValueB) {
          return 1;
        } else {
          return 0;
        }
      });
    });
    onSort?.(column, direction);
  };

  const headerCells = columns.map((column) => {
    return (
      <th key={column.key}>
        {column.header}
        {sortable && (
          <button onClick={() => handleSort(column, 'asc')}>
            {column.sortable && <span>↑</span>}
          </button>
          <button onClick={() => handleSort(column, 'desc')}>
            {column.sortable && <span>↓</span>}
          </button>

          <option key={size} value={size}>
            {size}
          </option>

function useSearch(query) {
  const [loading, setLoading] = useState(false);

  const [error, setError] = useState(null);
  const cache = useRef({});
  const debounce = useRef(null);

  useEffect(() => {
    // Cancel any in-flight requests
    if (debounce.current) {
      clearTimeout(debounce.current);
      debounce.current = null;

          setLoading(false);

          // Set the error and loading state
          setError(error);
          setLoading(false);
