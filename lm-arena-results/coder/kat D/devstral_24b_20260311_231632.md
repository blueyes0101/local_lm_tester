```typescript
import { Request, Response, NextFunction } from 'express';
import { UpdateTaskInput, updateTaskSchema } from '../schemas/taskSchema';
import { getTaskById, getTasksByWorkspaceId, updateTask } from '../services/taskService';

export const updateTaskController = async (req: Request, res: Response, next: NextFunction) => {
  const workspaceId = parseInt(req.params.workspaceId, 10);
  const taskId = parseInt(req.params.taskId, 10);

  if (isNaN(workspaceId) || isNaN(taskId)) {
    return res.status(400).json({ error: 'Invalid workspaceId or taskId' });
  }

  const task = getTaskById(taskId);
  if (!task || task.workspaceId !== workspaceId) {
    return res.status(404).json({ error: 'Task not found' });
  }

  try {
    const validationResult = updateTaskSchema.safeParse(req.body);
    if (!validationResult.success) {
      return res.status(400).json({ errors: validationResult.error.errors });
    }

    const updateData: UpdateTaskInput = validationResult.data;

    const updatedTask = updateTask(taskId, updateData);
    if (!updatedTask) {
      return res.status(404).json({ error: 'Task not found' });
    }

    return res.status(200).json(updatedTask);
  } catch (error) {
    next(error);
  }
};
```

### 2. Component Implementation

Now, let's implement the DataTable component.

```tsx
import React, { useMemo, useState, useEffect, useCallback } from 'react';
import { Column, DataTableProps, SortConfig, SORT_DIRECTIONS } from './DataTable.types';

function DataTable<T>({
  columns,
  data,
  onSortChange,
  onSelectRows,
  onPageChange,
  onPageSizeChange,
  sortConfig: initialSortConfig,
  selectedRows: initialSelectedRows,
  page: initialPage = 1,
  pageSize: initialPageSize = 10
}: DataTableProps<T>) {
  const [sortConfig, setSortConfig] = useState<SortConfig<T> | undefined>(initialSortConfig);
  const [selectedRows, setSelectedRows] = useState<T[]>(initialSelectedRows || []);
  const [page, setPage] = useState<number>(initialPage);
  const [pageSize, setPageSize] = useState<number>(initialPageSize);

  const sortedData = useMemo(() => {
    if (!sortConfig || !sortConfig.key || !sortConfig.direction) {
      return data;
    }
    return [...data].sort((a, b) => {
      const aValue = a[sortConfig.key];
      const bValue = b[sortConfig.key];
      if (aValue < bValue) return sortConfig.direction === 'asc' ? -1 : 1;
      if (aValue > bValue) return sortConfig.direction === 'asc' ? 1 : -1;
      return 0;
    });
  }, [data, sortConfig]);

  const paginatedData = useMemo(() => {
    const startIndex = (page - 1) * pageSize;
    return sortedData.slice(startIndex, startIndex + pageSize);
  }, [sortedData, page, pageSize]);

  const handleSort = useCallback(
    (key: keyof T) => {
      if (sortConfig && sortConfig.key === key) {
        const nextDirectionIndex =
          (SORT_DIRECTIONS.indexOf(sortConfig.direction!) + 1) % SORT_DIRECTIONS.length;
        setSortConfig({
          key,
          direction: SORT_DIRECTIONS[nextDirectionIndex]
        });
      } else {
        setSortConfig({ key, direction: 'asc' });
      }
    },
    [sortConfig]
  );

  useEffect(() => {
    if (onSortChange && sortConfig) {
      onSortChange(sortedData, sortConfig);
    }
  }, [onSortChange, sortedData, sortConfig]);

  useEffect(() => {
    if (onSelectRows) {
      onSelectRows(selectedRows);
    }
  }, [onSelectRows, selectedRows]);

  const toggleSelectAll = useCallback(() => {
    if (selectedRows.length === paginatedData.length) {
      setSelectedRows([]);
    } else {
      setSelectedRows(paginatedData);
    }
  }, [selectedRows, paginatedData]);

  const toggleSelectRow = useCallback(
    (row: T) => {
      setSelectedRows(prevSelectedRows =>
        prevSelectedRows.includes(row)
          ? prevSelectedRows.filter(r => r !== row)
          : [...prevSelectedRows, row]
      );
    },
    [setSelectedRows]
  );

  const handlePageChange = useCallback(
    (newPage: number) => {
      setPage(newPage);
      if (onPageChange) {
        onPageChange(newPage);
      }
    },
    [onPageChange]
  );

  const handlePageSizeChange = useCallback(
    (newPageSize: number) => {
      setPageSize(newPageSize);
      setPage(1);
      if (onPageSizeChange) {
        onPageSizeChange(newPageSize);
      }
    },
    [onPageSizeChange]
  );

  return (
    <div>
      <table>
        <thead>
          <tr>
            <th>
              <input
                type="checkbox"
                checked={selectedRows.length === paginatedData.length}
                onChange={toggleSelectAll}
              />
            </th>
            {columns.map(col => (
              <th
                key={String(col.key)}
                style={{ width: col.width }}
                onClick={() => col.sortable && handleSort(col.key)}
              >
                {col.header}
                {sortConfig && sortConfig.key === col.key && (
                  <span>{sortConfig.direction === 'asc' ? ' ↑' : ' ↓'}</span>
                )}
              </th>
            ))}
          </tr>
```

Cancelling in-flight requests is important to prevent displaying outdated or incorrect data. Here's the bug it prevents:

1. User types "a" in the search box → Request 1 is sent
2. Before Request 1 completes, user types "ab" → Request 2 is sent
3. Request 1 completes and updates the UI with results for "a"
4. Request 2 completes and updates the UI with results for "ab"

This can lead to a confusing user experience where the search results don't match the current query. By cancelling in-flight requests when a new search starts, we ensure that only the most recent request's results are displayed, providing a more consistent and accurate user experience.

In the implementation above, we use a debounce mechanism with a timeout to effectively "cancel" previous requests by clearing the timeout when a new search is initiated.

#### Problem Example

Suppose we have 50 recipes and we fetch each recipe's author individually:

```javascript
const resolvers = {
  Recipe: {
    author: async (recipe, _, { dataSources }) => {
      return dataSources.recipeAPI.getAuthorById(recipe.authorId);
    },
  },
};
```

This would result in 50 separate database queries for authors, causing the N+1 problem.

#### Solution with DataLoader

```javascript
const DataLoader = require('dataloader');

const authorLoader = new DataLoader(async (authorIds) => {
  const authors = await dataSources.recipeAPI.getAuthorsByIds(authorIds);
  return authorIds.map(id => authors.find(author => author.id === id));
});

  Recipe: {
    author: (recipe) => authorLoader.load(recipe.authorId),
  },
};
```

#### When Not to Choose GraphQL:
- For simple CRUD operations where the structure of data is well-defined.
- When performance is critical and you need to optimize for speed over flexibility.
- When security is a major concern, as GraphQL can be more complex to secure.
- When the team is more familiar with RESTful APIs and prefer to stick with what they know.
