def validate_refresh_token(refresh_token):
    refresh_token_key = f'refresh_token:{refresh_token}'
    if redis.exists(refresh_token_key):
        # Check if the refresh token is already in use
        if redis.hget(refresh_token_key, 'in_use'):
            return False
        else:
            # Mark the refresh token as in use
            redis.hset(refresh_token_key, 'in_use', True)
            return True
    else:
        return False

def invalidate_refresh_token(refresh_token):

def invalidate_all_sessions(user_id):
    # Get all refresh tokens associated with the user

6. Token blacklisting and token rotation:
- Token rotation: Invalidating the previous refresh token and generating a new one. This ensures that even if a refresh token is compromised, it can only be used once.
- Token blacklisting: Keeping a list of blacklisted refresh tokens in the database. When a refresh token is used, it is checked against the blacklist. If it is on the list, the token is considered invalid. This approach may require more database lookups, but it provides more flexibility in handling token reuse detection.

export interface Column<T> {
  key: string;
  header: string;
  width: number;
  sortable: boolean;
  renderCell?(row: T): ReactNode;
}

export interface DataTableProps<T> {
  columns: Column<T>[];
  data: T[];
  pageSize?: number;
  selectedRows?: T[];
  onSelectionChange?(selectedRows: T[]): void;
}

export default function DataTable<T>(props: DataTableProps<T>) {
  const { columns, data, pageSize = 10, selectedRows = [], onSelectionChange } = props;
  const [sortedData, setSortedData] = useState(data);
  const [currentPage, setCurrentPage] = useState(0);
  const [selectedRowsState, setSelectedRows] = useState(selectedRows);

  useEffect(() => {
    setSortedData(data);
  }, [data]);

  const handleSort = (columnKey: string) => {
    const sorted = [...sortedData].sort((a, b) => {
      const valA = a[columnKey];
      const valB = b[columnKey];
      if (valA < valB) return -1;
      if (valA > valB) return 1;
      return 0;
    });
    setSortedData(sorted);
  };

  const handleSelectionChange = (row: T) => {
    const updatedSelection = selectedRowsState.includes(row)
      ? selectedRowsState.filter((r) => r !== row)
      : [...selectedRowsState, row];
    setSelectedRows(updatedSelection);
    if (onSelectionChange) {
      onSelectionChange(updatedSelection);
    }
  };

  const renderColumnHeader = (column: Column<T>) => {
    if (column.sortable) {
      return (
        <button type="button" onClick={() => handleSort(column.key)}>
          {column.header}
        </button>
      );
    }
    return <span>{column.header}</span>;
  };

  const renderRow = (row: T) => {
    return (
      <tr>
        {columns.map((column) => (
          <td key={column.key}>
            {column.renderCell ? column.renderCell(row) : row[column.key]}
          </td>
        ))}
      </tr>
    );
  };

  const renderRows = () => {
    const start = currentPage * pageSize;
    const end = start + pageSize;
    return sortedData.slice(start, end).map((row) => renderRow(row));
  };

  const handlePageChange = (newPage: number) => {
    setCurrentPage(newPage);
  };

  return (
    <div>
      <table>
        <thead>
          <tr>
            {columns.map((column) => (
              <th key={column.key} style={{ width: column.width }}>
                {renderColumnHeader(column)}
              </th>
            ))}
          </tr>
        </thead>
        <tbody>{renderRows()}</tbody>
      </table>
      <div>
        <button type="button" onClick={() => handlePageChange(currentPage - 1)}>
          Previous
        </button>
        <span>Page {currentPage + 1}</span>
        <button type="button" onClick={() => handlePageChange(currentPage + 1)}>
          Next
        </button>
      </div>
      <div>
        <label>
          <input
            type="checkbox"
            checked={selectedRowsState.length === data.length}
            onChange={() => {
              const updatedSelection = selectedRowsState.length === data.length
                ? []
                : data;
              setSelectedRows(updatedSelection);
              if (onSelectionChange) {
                onSelectionChange(updatedSelection);
              }
            }}
          />
          Select All
        </label>
      </div>
      <div>
        {selectedRowsState.map((row) => (
          <p key={row.id}>{JSON.stringify(row)}</p>
        ))}
      </div>
    </div>
  );
}

import React, { useState, useEffect, useCallback, useRef } from 'react';

function useSearch(query) {
  const [results, setResults] = useState(null);
  const [error, setError] = useState(null);
  const [isFetching, setIsFetching] = useState(false);
  const [lastQuery, setLastQuery] = useState(query);
  const [cache, setCache] = useState({});

  const fetchData = useCallback(async () => {
    if (query === lastQuery) {
      try {
        const response = await fetch('/api/search?q=' + query);
        const data = await response.json();

        if (data && Array.isArray(data)) {
          setResults(data);
          setCache((cache) => ({
            ...cache,
            [query]: data,
          }));
        } else {
          setResults([]);
        }
      } catch (error) {
        setError(error.message);
      } finally {
        setIsFetching(false);
      }
    }
  }, [query, lastQuery]);

  useEffect(() => {
    if (query !== lastQuery) {
      setLastQuery(query);
      setResults(null);
      setIsFetching(true);
      fetchData();
    }
  }, [query, lastQuery, fetchData]);

  return {
    results,
    error,
    isFetching,
  };
}

function SearchPage() {
  const [query, setQuery] = useState('');
  const { results, error, isFetching } = useSearch(query);

  const debouncedQuery = useRef(null);
  useEffect(() => {
    clearTimeout(debouncedQuery.current);
    debouncedQuery.current = setTimeout(() => {
      setQuery(debouncedQuery.current);
    }, 300);
    return () => clearTimeout(debouncedQuery.current);
  }, [query]);

  return (
    <>
      <input
        type="text"
        value={query}
        onChange={(e) => setQuery(e.target.value)}
      />
      {isFetching && <div>Loading...</div>}
      {error && (
        <div>
          Error: {error} <button onClick={() => setQuery(query)}>Retry</button>
        </div>
      )}
      {results &&
        (results.length > 0 ? (
          results.map((result) => <div key={result.id}>{result.title}</div>)
        ) : (
          <div>No results</div>
        ))}
    </>
  );
}

export default SearchPage;

2. Message protocol — define the message types (JSON) for: join document, text change, cursor move, user left
3. Broadcasting — when user A makes a change, how do you send it to all other users in the same document but NOT back to user A?
4. Reconnection — how do you handle a user rejoining after disconnect?
5. What happens if two users edit the exact same position simultaneously? You don't need to implement OT/CRDT — just explain the problem and name the solution approaches.

**A:**
```javascript
const WebSocket = require("ws");
const wss = new WebSocket.Server({ port: 8080 });

const connections = new Map();

wss.on("connection", (ws) => {
  ws.on("message", (message) => {
    const data = JSON.parse(message);
    switch (data.type) {
      case "join":
        const { documentId, userId } = data;
        connections.set(userId, { ws, documentId });
        break;
      case "text-change":
        const { userId, position, text } = data;
        const user = connections.get(userId);
        if (user) {
          const otherUsers = Array.from(connections.values())
            .filter(
              (u) => u.documentId === user.documentId && u.userId !== user.userId
            )
            .map((u) => u.ws);
          otherUsers.forEach((ws) => {
            ws.send(
              JSON.stringify({
                type: "text-change",
                position,
                text,
              })
            );
          });
        }
        break;
      case "cursor-move":
        const { userId, position } = data;
        const user = connections.get(userId);
        if (user) {
          const otherUsers = Array.from(connections.values())
            .filter(
              (u) => u.documentId === user.documentId && u.userId !== user.userId
            )
            .map((u) => u.ws);
          otherUsers.forEach((ws) => {
            ws.send(
              JSON.stringify({
                type: "cursor-move",
                position,
              })
            );
          });
        }
        break;
      case "user-left":
        const { userId } = data;
        connections.delete(userId);
        break;
      default:
        break;
    }
  });

  ws.on("close", () => {
    for (const [key, value] of connections) {
      if (value.ws === ws) {
        connections.delete(key);
        break;
      }
    }
  });
});
