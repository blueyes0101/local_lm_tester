1. To implement the refresh token rotation, you could use a Redis Hash data structure where the refresh token is the key and the user ID is the value. You can also set a TTL (time to live) for each refresh token in the Redis store, so that tokens that are not used within a certain time frame are automatically invalidated.

2. To invalidate a refresh token, you could simply delete it from the Redis store. When the refresh token is deleted, it will no longer be found in the Redis store, so it will be recognized as an invalid token.

5. Token blacklisting and token rotation are two different strategies for handling cases where refresh tokens are compromised or used illegally. In general, token rotation is a simpler approach that doesn't require any additional storage or processing, but it can lead to a larger number of refresh tokens if they are used frequently. Token blacklisting requires additional storage and processing, but it only maintains a single valid refresh token per user, thus reducing the potential for misuse. import React, { useState, useMemo, ReactNode } from 'react';

function DataTable<T>(props: DataTableProps<T>): ReactNode {
    const { columns, data, onSortChange, onSelectionChange, pageSizeOptions = [10, 25, 50] } = props;
    const [sort, setSort] = useState<{ key: keyof T; direction: 'asc' | 'desc' | 'none' }>({ key: columns[0].key, direction: 'none' });
    const [selected, setSelected] = useState<T[]>([]);

    const handleSortChange = (key: keyof T) => {
        let direction: 'asc' | 'desc' | 'none' = 'asc';

        if (sort.key === key && sort.direction === 'asc') {
            direction = 'desc';
        } else if (sort.key === key && sort.direction === 'desc') {
            direction = 'none';
        }

        setSort({ key, direction });
        onSortChange && onSortChange(key, direction);
    };

    const handleSelectAll = (selected: boolean) => {
        if (selected) {
            setSelected(data);
        } else {
            setSelected([]);
        }
    };

    const handleRowSelection = (row: T) => {
        let newSelected;
        if (selected.includes(row)) {
            newSelected = selected.filter(item => item !== row);
        } else {
            newSelected = [...selected, row];
        }

        setSelected(newSelected);
        onSelectionChange && onSelectionChange(newSelected);
    };

    const sortedData = useMemo(() => {
        if (sort.direction === 'none') {
            return data;
        }

        return data.sort((a, b) => {
            const valA = a[sort.key];
            const valB = b[sort.key];

            if (typeof valA === 'string' && typeof valB === 'string') {
                return sort.direction === 'asc' ? valA.localeCompare(valB) : valB.localeCompare(valA);
            }

            return sort.direction === 'asc' ? (valA > valB ? 1 : -1) : (valA > valB ? -1 : 1);
        });
    }, [data, sort]);

    return (
        <div>
            <table>
                <thead>
                    <tr>
                        <th>
                            <input type="checkbox" checked={selected.length === data.length} onChange={e => handleSelectAll(e.target.checked)} />
                        </th>
                        {columns.map(column => (
                            <th key={column.key.toString()} onClick={() => handleSortChange(column.key)}>
                                {column.header}
                            </th>
                        ))}
                    </tr>
                </thead>
                <tbody>
                    {sortedData.map((row, index) => (
                        <tr key={index}>
                            <td>
                                <input type="checkbox" checked={selected.includes(row)} onChange={() => handleRowSelection(row)} />
                            </td>
                            {columns.map(column => (
                                <td key={column.key.toString()}>
                                    {column.renderCell ? column.renderCell(row) : row[column.key]}
                                </td>
                            ))}
                        </tr>
                    ))}
                </tbody>
            </table>
        </div>
    );
}

export default DataTable;
