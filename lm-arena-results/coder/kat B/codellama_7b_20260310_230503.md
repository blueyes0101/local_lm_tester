When the user logs out of the application, the server invalidates the refresh token and clears the cookie containing the refresh token. This ensures that the user is logged out of the application and that any further attempts to use the refresh token will fail.

Token rotation, on the other hand, involves issuing a new token and invalidating the old token. The main difference between the two is that token blacklisting prevents the use of any token that is in the list of blacklisted tokens, while token rotation prevents the use of a specific token that has been used before.

Token blacklisting is useful when a user's account is compromised or when a user's access token is leaked. By blacklisting the compromised access token, the user is prevented from using it to access the application, which helps to protect the user's account and data.

Token rotation, on the other hand, is useful when a user's access token has been compromised or when a user's access token has been leaked. By rotating the access token, the user is prevented from using the compromised or leaked access token, which helps to protect the user's account and data.

In general, token blacklisting is used when a user's account is compromised or when a user's access token is leaked, while token rotation is used when a user's access token has been compromised or when a user's access token has been leaked.

import React, { useState } from 'react';
import { useTable, useSortBy, useSelect } from 'react-table';
import { memoize } from 'lodash';

function DataTable<T>(props: DataTableProps<T>) {
  const { columns, data, isControlled, pageSize, onSort, onSelect } = props;
  const { getTableProps, headerGroups, rows, prepareRow } = useTable(
    { columns, data },
    useSortBy,
    useSelect
  );

  const [sortState, setSortState] = useState<{ column: string; direction: 'asc' | 'desc' }>({
    column: '',
    direction: 'none',
  });

  const handleSort = (column: string, direction: 'asc' | 'desc') => {
    setSortState({ column, direction });
    if (onSort) {
      onSort(column, direction);
    }
  };

  const handleSelect = (rows: T[]) => {
    if (onSelect) {
      onSelect(rows);
    }
  };

  const memoizedHeaderGroups = memoize(headerGroups, (headerGroup) => {
    return headerGroup.headers;
  });

  return (
    <table {...getTableProps()}>
      <thead>
        {memoizedHeaderGroups.map((headerGroup) => (
          <tr {...headerGroup.getHeaderGroupProps()}>
            {headerGroup.headers.map((column) => (
              <th {...column.getHeaderProps()}>
                {column.render('Header')}
                {column.canSort && (
                  <button
                    type="button"
                    aria-label={`Sort ${column.id}`}
                    onClick={() => handleSort(column.id, sortState.direction === 'asc' ? 'desc' : 'asc')}
                  >
                    {sortState.column === column.id && (
                      <span>
                        {sortState.direction === 'asc' ? ' 🔼' : ' 🔽'}
                      </span>
                    )}
                  </button>
                )}
              </th>
            ))}
          </tr>
        ))}
      </thead>
      <tbody>
        {rows.map((row) => {
          prepareRow(row);
          return (
            <tr {...row.getRowProps()}>
              {row.cells.map((cell) => {
                return (
                  <td {...cell.getCellProps()}>
                    {cell.render('Cell')}
                  </td>
                );
              })}
            </tr>
          );

  );
}

**A:**

import { useState, useEffect, useRef } from 'react';

const useSearch = (query) => {
  const [results, setResults] = useState([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);
  const cancelRef = useRef(null);

  const fetchResults = async () => {
    setLoading(true);
    try {
      const response = await fetch(`/api/search?q=${query}`);
      const data = await response.json();
      setResults(data);
      setLoading(false);
    } catch (error) {
      setError(error);
      setLoading(false);
    }
  };

  const debouncedFetch = useCallback(
    debounce(fetchResults, 300),
    [query]
  );

  const cancel = useCallback(() => {
    if (cancelRef.current) {
      cancelRef.current();
    }
  }, []);

  useEffect(() => {
    if (query) {
      cancel();
      debouncedFetch();
    }
  }, [query, debouncedFetch, cancel]);

  useEffect(() => {
    if (error) {
      setLoading(false);
    }
  }, [error]);

  useEffect(() => {
    if (results.length === 0) {
      setLoading(false);
    }
  }, [results]);

  return { results, loading, error, cancel };
};
