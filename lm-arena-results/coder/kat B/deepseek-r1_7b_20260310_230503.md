const updateTask = async (request: Request, response: Response) => {
  const { title, status, assigneeId, dueDate } = request.json();

  // Validate request body
  const validatedRequest = taskSchema.validate({
    title,
    status,
    assigneeId,
    dueDate,
  });

  // Check if assigneeId is an integer or null
  if (assignedeeId && !Number.isInteger(assignedeeId)) {
    throw new Error('AssigneeId must be an integer or null');
  }

  // Get workspace and task from database
  const workspace = await getWorkspaceById(request, workspaceId);
  const task = await workspace.tasks.find((task) => task.id === taskId);

  // Check if task exists
  if (!task) {
    return response.status(404).json({ error: 'Task not found' });
  }

  // Check if assigneeId is valid for the workspace
  if (assignedeeId && !workspace.members.contains(assignedeeId)) {
    return response.status(400).json({ error: 'Assignee not allowed for this workspace' });
  }

  // Partial update logic
  let updatedTask = task as Task;
  for (const field of [title, status, assigneeId, dueDate]) {
    if (field) {
      updatedTask[field as keyof Task] = field;
    }
  }

  // Update the task in the database
  const savedTask = await workspace.tasks.save(updatedTask);
  if (savedTask.id !== task.id) {
    return response.status(200).json(savedTask);
  }

  // Check if status is 'cancelled' and subtasks exist
  if (status === 'cancelled') {
    const openSubtasks = await workspace.subtasks.find(
      (subtask) => subtask.taskId === task.id && subtask.active
    );
    if (openSubtasks.length > 0) {
      return response.status(409).json({ error: 'Cannot set status to cancelled if subtasks are open' });
    }
  }

  // Return updated task
  response.status(200).json(savedTask);
};

// Middleware for validation
async function validateRequest(request: Request, response: Response) {
  try {
    // Validate workspaceId and taskId are integers
    const workspaceId = parseInt(request.json().workspaceId, 10);
    const taskId = parseInt(request.json().taskId, 10);

    if (isNaN(workspaceId) || isNaN(taskId)) {
      throw new Error('workspaceId and taskId must be integers');

      throw new Error('workspaceId is invalid');
    }

    if (taskId < 1 || taskId > workspace.tasks.count) {
      throw new Error('taskId must be between 1 and ' + workspace.tasks.count);
    }
