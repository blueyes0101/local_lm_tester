  const { assignee_id } = req.body;

  try {
    // Validate workspaceId and taskId are integers
    const idRegex = /^\d+$/;
    if (!idRegex.test(workspaceId) || !idRegex.test(taskId)) {
      return res.status(400).json({ error: 'Invalid workspace or task ID' });

    // Validate task belongs to workspace
    if (!(await req.workspaceService.hasTask(taskId, workspaceId))) {

      return res.status(400).json({ error: 'Assignee is not a member of the workspace' });
    }

    next();
  } catch (error) {
    next(error);
  }
};

export class TaskController {
  constructor(private workspaceService: WorkspaceService) {}

  async update(req: Request, res: Response) {
    try {
      const { workspaceId, taskId } = req.params;
      const updateData = TaskUpdateSchema.parse(req.body);

      // Check for cancelled status with open subtasks
      if (updateData.status === 'cancelled') {
        const subtasks = await req.workspaceService.getSubtasks(taskId);
        if (subtasks.some(st => st.status !== 'done')) {
          return res.status(409).json({ error: 'Cannot cancel task with open subtasks' });
        }
      }

      // Update task
      await req.workspaceService.updateTask(taskId, {
        ...updateData,
        workspaceId,
      });

      res.status(200).send();
    } catch (error) {
      res.status(400).json({ error: 'Validation failed' });
    }
  }
}

6. **Token Blacklisting vs Token Rotation**

**Token Blacklisting:**
- Stores a list of "blacklisted" tokens that should never be used again
- Good for immediate invalidation of tokens
- Uses a set to track blacklisted tokens
- Typically used for:
  - Immediate logout
  - Password change
  - Two-factor authentication tokens
  - Suspicious login attempts

**Token Rotation:**
- Regularly issues new tokens and invalidates old ones
- Good for long-term security and preventing token reuse
- Uses a time-based strategy (TTL)
- Typically used for:
  - Access tokens in OAuth systems
  - Refresh tokens with limited use
  - Short-lived tokens that need regular refresh

import React, { useState, useCallback, useMemo, forwardRef } from 'react';
import { DataTable as MuiDataTable } from '@mui/material';

const DataTable = forwardRef<T, DataTableProps<T>>((props, ref) => {
  const {
    columns,
    data,
    currentPage,
    pageSize,
    onPageChange,
    onPageSizeChange,
    onSelectChange,
    isSelected,
    selectedRows,
    defaultSortColumn,
    defaultSortDirection,
    isControlled,
    ...rest
  } = props;

  const [state, setState] = useState<DataTableState>(() => ({
    sortColumn: defaultSortColumn,
    sortDirection: defaultSortDirection,
    selectedRows: selectedRows,
  }));

  const handleSort = useCallback((column: string) => {
    if (state.sortColumn === column) {
      setState(prev => ({
        ...prev,
        sortDirection: prev.sortDirection === 'asc' ? 'desc' : 'asc',
      }));
    } else {
      setState(prev => ({
        ...prev,
        sortColumn: column,
        sortDirection: 'asc',
      }));
    }
  }, [state]);

  const handleSelectChange = useCallback(
    (rows: T[]) => {
      if (onSelectChange) {
        onSelectChange(rows);
      }
      setState(prev => ({ ...prev, selectedRows: rows }));
    },
    [onSelectChange]
  );

  const handleSelectAll = useCallback(() => {
    if (isSelected) {
      const allRows = data.filter((row, index) => 
        columns.some(col => col.key === 'checkbox')
      );
      handleSelectChange(allRows);
    }
  }, [isSelected, handleSelectChange, data, columns]);

  const handlePageSizeChange = useCallback(
    (size: number) => {
      if (onPageSizeChange) {
        onPageSizeChange(size);
      }
      setState(prev => ({ ...prev, pageSize: size }));
    },
    [onPageSizeChange]
  );

  const handlePageChange = useCallback(
    (page: number) => {
      if (onPageChange) {
        onPageChange(page);
      }
      setState(prev => ({ ...prev, currentPage: page }));
    },
    [onPageChange]
  );

  const memoizedColumns = useMemo(
    () => columns.map(col => ({
      ...col,
      renderCell: col.renderCell ? col.renderCell : (value: T) => value,
      sortFunction: (a: T, b: T, sortColumn: string) => {
        if (col.sortable) {
          const aValue = a[sortColumn];
          const bValue = b[sortColumn];
          if (sortColumn === 'checkbox') {
            return aValue === bValue ? 0 : aValue > bValue ? 1 : -1;
          }
          if (aValue < bValue) return -1;
          if (aValue > bValue) return 1;
          return 0;
        }
        return 0;
      },
    })),
    [columns]
  );

  const memoizedState = useMemo(
    () => ({
      ...state,
      sortColumn: state.sortColumn || defaultSortColumn,
      sortDirection: state.sortDirection || defaultSortDirection,
    }),
    [state, defaultSortColumn, defaultSortDirection]
  );

  return (
    <MuiDataTable
      ref={ref}
      rowsPerPageOptions={[10, 25, 50]}
      columns={memoizedColumns}
      data={data}
      page={currentPage}
      rowsPerPage={pageSize}
      onPageChange={handlePageChange}
      onPageSizeChange={handlePageSizeChange}
      sortBy={handleSort}
      sortDescFirst={false}
      onRowSelect={(rows) => handleSelectChange(rows)}
      selectAllRows={handleSelectAll}
      selectRowsSelected={isSelected}
      {...rest}
    />
  );
});

const WebSocket = require('ws');
const { v4: uuidv4 } = require('uuid');

class DocumentServer {
    constructor(port = 8080) {
        this.wss = new WebSocket.Server({ port });
        this.documents = new Map(); // Map of document ID to user connections
        this.documentMap = new Map(); // Map of user to document connections
        this.setupMessageHandlers();
        this.setupConnectionHandlers();
    }

    // Message types
    static MESSAGE_TYPES = {
        JOIN: 'join',
        TEXT_CHANGE: 'textChange',
        CURSOR_MOVE: 'cursorMove',
        USER_LEFT: 'userLeft',
        GET_STATE: 'getState'
    };

    setupMessageHandlers() {
        this.wss.on('connection', (ws) => {
            ws.on('message', (message) => {
                const { type, documentId, userId, text, position, cursor } = JSON.parse(message);
                this.handleMessage(ws, type, documentId, userId, text, position, cursor);
            });

            ws.on('close', () => {
                this.handleUserDisconnect(ws);
            });
        });
    }

    handleMessage(ws, type, documentId, userId, text, position, cursor) {
        switch (type) {
            case DocumentServer.MESSAGE_TYPES.JOIN:
                this.joinDocument(ws, documentId, userId);
                break;
            case DocumentServer.MESSAGE_TYPES.TEXT_CHANGE:
                this.broadcastTextChange(documentId, userId, text, position);
                break;
            case DocumentServer.MESSAGE_TYPES.CURSOR_MOVE:
                this.broadcastCursorMove(documentId, userId, cursor);
                break;
            case DocumentServer.MESSAGE_TYPES.USER_LEFT:
                this.handleUserLeft(documentId, userId);
                break;
            case DocumentServer.MESSAGE_TYPES.GET_STATE:
                this.broadcastGetState(documentId, userId);
                break;
        }
    }

    // Handle user joining a document
    joinDocument(ws, documentId, userId) {
        const document = this.documents.get(documentId) || {
            connections: new Set(),
            text: '',
            cursorPositions: {}
        };
        document.connections.add(ws);
        this.documentMap.set(userId, documentId);
        document.text = document.text || '';
        document.cursorPositions[userId] = { x: 0, y: 0 };
        this.documents.set(documentId, document);
        ws.documentId = documentId;
        ws.userId = userId;

        // Send initial document state
        this.broadcastGetState(documentId, userId);
    }

    // Broadcast text change to all users in the document
    broadcastTextChange(documentId, userId, text, position) {
        const document = this.documents.get(documentId);
        if (document) {
            document.text = document.text.substring(0, position) + text + document.text.substring(position);
            // Don't broadcast back to the user who made the change
            document.connections.forEach((otherWs) => {
                if (otherWs !== document.connections.get(userId)) {
                    otherWs.send(JSON.stringify({
                        type: DocumentServer.MESSAGE_TYPES.TEXT_CHANGE,
                        userId,
                        text,
                        position
                    }));
                }
            });
        }
    }

    // Broadcast cursor move to all users in the document
    broadcastCursorMove(documentId, userId, { x, y }) {
        const document = this.documents.get(documentId);
        if (document) {
            document.cursorPositions[userId] = { x, y };
            document.connections.forEach((otherWs) => {
                if (otherWs !== document.connections.get(userId)) {
                    otherWs.send(JSON.stringify({
                        type: DocumentServer.MESSAGE_TYPES.CURSOR_MOVE,
                        userId,
                        cursor: { x, y }
                    }));
                }
            });
        }
    }

    // Handle user disconnecting
    handleUserDisconnect(ws) {
        const userId = ws.userId;
        const documentId = ws.documentId;
        const document = this.documents.get(documentId);

        if (document) {
            document.connections.delete(ws);
            if (document.connections.size === 0) {
                document.text = '';
                document.cursorPositions = {};
                this.documents.delete(documentId);
                this.documentMap.delete(userId);
            } else {
                // Send final document state to disconnected user
                this.broadcastGetState(documentId, userId);
            }
        }
    }

    // Start the server
    start() {
        this.wss.listen(8080, () => {
            console.log('Document server started on port 8080');
        });
    }
}

## Question 12
**Q:** You are asked to add real-time notifications to the existing project management tool. Users should be notified when: a task is assigned to them, a comment is added to their task, a deadline is approaching (within 24h).

Design the notification system:
1. What transport mechanism would you use and why? (WebSocket, SSE, polling, push notifications — compare and justify)
2. How do you store notifications? Design the database schema.
3. How do you handle a user who is offline — do they get missed notifications when they come back online?
4. The deadline check needs to run every hour for all tasks. How do you implement this? (cron job, queue, event-driven?) — show the implementation approach.
5. How do you scale this if there are 100,000 concurrent users?

**A:**
I'll address each point:

1. Transport Mechanism:
   - WebSocket is the best choice here because:
     * Real-time, bidirectional communication
     * Low latency for immediate notifications
     * Persistent connection (not blocked by firewalls)
     * Suitable for web applications
   - SSE is good but only unidirectional
   - Polling is inefficient for real-time needs
   - Push notifications (like Firebase) are good for mobile, but for web apps, WebSocket is more suitable

2. Database Schema:
```sql
CREATE TABLE notifications (
    id BIGINT PRIMARY KEY,
    user_id BIGINT NOT NULL,
    type VARCHAR(50) NOT NULL,  -- 'assignment', 'comment', 'deadline'
    task_id BIGINT,
    read_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE task_deadlines (
    id BIGINT PRIMARY KEY,
    task_id BIGINT NOT NULL,
    deadline_at TIMESTAMP NOT NULL,
    notified_at TIMESTAMP,
    last_check_at TIMESTAMP
);
```

3. Offline Handling:
   - Store notifications in database when a user goes offline
   - Use WebSocket's reconnect feature to automatically reconnect
   - When user comes back online, check for unprocessed notifications
   - Mark read notifications as read_at is set

4. Deadline Check Implementation:
```python
# Using Celery (task queue) for periodic tasks
from celery import Celery
from datetime import datetime, timedelta

app = Celery('notifications', broker='redis://localhost:6379/0')

@app.task
def check_deadlines():
    # Query all tasks with deadlines within 24 hours
    deadline_threshold = datetime.now() + timedelta(days=1)
    tasks = Task.objects.filter(
        deadline_at__lte=deadline_threshold,
        deadline_at__gte=datetime.now()
    )
    
    for task in tasks:
        # Check if deadline notification already sent
        if not task.task_deadlines.filter(
            notified_at__isnull=True
        ).exists():
            # Send notification
            NotificationService.send_deadline_notification(task)
            
            # Mark deadline as notified
            task.task_deadlines.filter(
                notified_at__isnull=True
            ).update(
                notified_at=datetime.now()
            )

    # Schedule next check
    check_deadlines.apply_async(
        countdown=3600,  # 1 hour
        queue='notifications'
    )
```

5. Scaling for 100,000 Concurrent Users:
   - Use Redis for WebSocket connections (maintains connection pool)
   - Use load balancer to distribute WebSocket connections
   - Implement WebSocket clustering using frameworks like Socket.io
   - Use message queue (RabbitMQ/Kafka) for notification distribution
   - Cache frequently accessed data (user notifications, task deadlines)
   - Use database sharding for notifications table
   - Implement rate limiting for WebSocket connections
   - Use connection pooling for database queries
