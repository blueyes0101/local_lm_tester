```typescript
import { getTaskById, getWorkspaceById, getSubtasksByTaskId, updateTask } from './services';

const router = express.Router();

router.patch('/workspaces/:workspaceId/tasks/:taskId', async (req, res, next) => {
  try {
    // Validate workspaceId and taskId are integers
    const workspaceId = parseInt(req.params.workspaceId);
    const taskId = parseInt(req.params.taskId);

    if (isNaN(workspaceId) || isNaN(taskId)) {
      return res.status(400).json({
        error: 'Validation Error',
        details: [
          { field: 'workspaceId', message: 'Must be an integer' },
          { field: 'taskId', message: 'Must be an integer' },
        ],
      });
    }

    // Validate input
    const validatedInput = taskUpdateSchema.parse(req.body);

    // Get task and workspace
    const task = await getTaskById(taskId);
    const workspace = await getWorkspaceById(workspaceId);

    if (!task || task.workspace_id !== workspaceId) {
      return res.status(404).json({ error: 'Task not found or does not belong to workspace' });
    }

    if (!workspace) {
      return res.status(404).json({ error: 'Workspace not found' });
    }

    // Validate assignee_id if provided
    if (validatedInput.assignee_id !== undefined && !workspace.members.includes(validatedInput.assignee_id)) {
      return res.status(400).json({
        error: 'Validation Error',
        details: [
          {
            field: 'assignee_id',
            message: 'Assignee must be a member of the workspace',
          },
        ],
      });
    }

    // Check for open subtasks if status is being set to cancelled
    if (validatedInput.status === 'cancelled') {
      const subtasks = await getSubtasksByTaskId(taskId);
      const hasOpenSubtasks = subtasks.some(
        subtask => subtask.status !== 'done' && subtask.status !== 'cancelled'
      );

      if (hasOpenSubtasks) {
        return res.status(409).json({
          error: 'Conflict',
          message: 'Cannot cancel task with open subtasks',
        });
      }
    }

    // Update task
    const updates: Partial<Omit<Task, 'id'>> = {
      ...validatedInput,
      workspace_id: workspaceId, // Ensure workspace_id is preserved
    };

    await updateTask(taskId, updates);

    res.status(200).json({ success: true });
  } catch (err) {
    next(err);
  }
});

export default router;
```

**A:**

### **1. POST `/auth/refresh`**
**Flow:**
1. Extract the `refresh_token` from the `HttpOnly` cookie.
2. Validate the token (signature, expiration, and existence in Redis).
3. Issue a new **access token** (short-lived, e.g., 15-30 mins) and a **new refresh token** (long-lived, e.g., 7-30 days).
4. **Rotate the refresh token**:
   - Invalidate the old refresh token in Redis.
   - Store the new refresh token with a new TTL.
5. Return the new access token in the response (not in a cookie for security).

**Example (Pseudocode):**
```python
@app.post("/auth/refresh")
def refresh():
    refresh_token = request.cookies.get("refresh_token")
    if not refresh_token:
        return {"error": "No refresh token"}, 401

    # Validate token in Redis
    stored_token = redis.get(f"refresh_token:{refresh_token}")
    if not stored_token:
        return {"error": "Invalid refresh token"}, 401

    # Generate new tokens
    new_access_token = generate_access_token(user_id=stored_token)
    new_refresh_token = generate_refresh_token(user_id=stored_token)

    # Rotate: Invalidate old, store new
    redis.delete(f"refresh_token:{refresh_token}")
    redis.setex(f"refresh_token:{new_refresh_token}", 7 * 24 * 60 * 60, stored_token)

    # Return new access token (no cookie for refresh)
    return {"access_token": new_access_token}
```

### **3. Redis Data Structure & TTL Strategy**
**Redis Key Structure:**
```
refresh_token:<token_value> → user_id
```
- **TTL (Time-to-Live):**
  - Refresh tokens: **7 days** (adjust based on security needs).
  - Access tokens: Handled by JWT expiration (no Redis storage needed).
- **Why Redis?**
  - Fast lookups for validation.
  - Automatic expiration (no manual cleanup).
  - Supports blacklisting (see below).

**Example Redis Commands:**
```bash
# Store a refresh token (7 days TTL)
SETEX refresh_token:abc123 604800 "user_42"

# Delete (invalidate)
DEL refresh_token:abc123
```

### **4. Token Reuse Detection**
**Problem:** An attacker steals a refresh token and tries to reuse it after the legitimate user logs out.

**Solution:**
- **Add a `used` flag** to Redis (or a separate key for tracking).
- **At refresh time**, check if the token was already used:
  - If yes → Reject.
  - If no → Mark as used and proceed.

**Redis Structure:**
```
refresh_token:<token_value> → {
  "user_id": "user_42",
  "used": false  # or a separate key: used_tokens:<token_value>
}
```

**Implementation:**
```python
@app.post("/auth/refresh")
def refresh():
    refresh_token = request.cookies.get("refresh_token")
    token_data = redis.hgetall(f"refresh_token:{refresh_token}")

    if not token_data or token_data.get(b"used") == b"true":
        return {"error": "Token already used"}, 401

    # Mark as used (atomic operation)
    redis.hset(f"refresh_token:{refresh_token}", "used", "true")

    # Proceed with rotation...
```

**Alternative (Simpler):**
Use a **set** to track used tokens:
```bash
# At refresh time:
SISMEMBER used_tokens abc123  # Returns 1 if already used
SADD used_tokens abc123      # Mark as used
```

```tsx
  const [internalSortDirection, setInternalSortDirection] = useState<SortDirection>(defaultSortDirection);

    if (!currentSortColumn || currentSortDirection === 'none') return data;

    return [...data].sort((a, b) => {
      const aValue = a[currentSortColumn];
      const bValue = b[currentSortColumn];

      if (aValue === bValue) return 0;
      if (aValue === null || aValue === undefined) return 1;
      if (bValue === null || bValue === undefined) return -1;

      if (typeof aValue === 'number' && typeof bValue === 'number') {
        return currentSortDirection === 'asc' ? aValue - bValue : bValue - aValue;
      }

      return String(aValue).localeCompare(String(bValue)) * (currentSortDirection === 'asc' ? 1 : -1);
    });
  }, [data, currentSortColumn, currentSortDirection]);

  const totalPages = useMemo(() => {
    return Math.ceil(sortedData.length / currentPageSize);
  }, [sortedData.length, currentPageSize]);

    const startIndex = (currentPage - 1) * currentPageSize;

      }

              </th>
            )}

              <th
                key={String(column.key)}
                style={{ width: column.width }}
                className="sortable-header"
                onClick={() => column.sortable && handleSort(column.key)}
              >
                {column.header}
                {column.sortable && (
                  <span className="sort-indicator">
                    {currentSortColumn === column.key && (
                      currentSortDirection === 'asc' ? ' ↑' : ' ↓'
                    )}
                  </span>
                )}
              </th>
            ))}
          </tr>
        </thead>
        <tbody className={bodyClassName}>
          {paginatedData.length > 0 ? (
            paginatedData.map((row, index) => (
              <tr key={`${String(row)}-${index}`}>

            <tr>
              <td colSpan={rowSelection ? columns.length + 1 : columns.length}>
                No data available

};
```

### 1. Custom Hook: `useSearch`

```tsx
import { useState, useEffect, useRef, useCallback } from 'react';

interface SearchResult {
  id: string;
  title: string;
  // ... other result properties
}

interface UseSearchResult {
  data: SearchResult[] | null;
  loading: boolean;
  error: string | null;
  refetch: () => void;
}

const useSearch = (query: string): UseSearchResult => {
  const [data, setData] = useState<SearchResult[] | null>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const abortControllerRef = useRef<AbortController | null>(null);
  const timeoutRef = useRef<NodeJS.Timeout | null>(null);
  const cache = useRef<Map<string, { data: SearchResult[]; timestamp: number }>>(new Map());

  const fetchData = useCallback(async (searchQuery: string) => {
    if (!searchQuery.trim()) {
      setData([]);
      return;
    }

    // Check cache first
    const cached = cache.current.get(searchQuery);
    if (cached && Date.now() - cached.timestamp < 5 * 60 * 1000) {
      setData(cached.data);
      return;
    }

    setLoading(true);
    setError(null);

    // Abort previous request if exists
    if (abortControllerRef.current) {
      abortControllerRef.current.abort();
    }

    abortControllerRef.current = new AbortController();
    const signal = abortControllerRef.current.signal;

    try {
      const response = await fetch(`/api/search?q=${encodeURIComponent(searchQuery)}`, { signal });

      if (!response.ok) {
        throw new Error(`HTTP error! status: ${response.status}`);
      }

      const result: SearchResult[] = await response.json();

      // Update cache
      cache.current.set(searchQuery, {
        data: result,
        timestamp: Date.now()
      });

      setData(result);
    } catch (err) {
      if (err.name !== 'AbortError') {
        setError('Failed to fetch search results. Please try again.');
      }
    } finally {
      setLoading(false);
      abortControllerRef.current = null;
    }
  }, []);

  useEffect(() => {
    // Clear timeout from previous search
    if (timeoutRef.current) {
      clearTimeout(timeoutRef.current);
    }

    timeoutRef.current = setTimeout(() => {
      fetchData(query);
    }, 300);

    return () => {
      if (timeoutRef.current) {
        clearTimeout(timeoutRef.current);
      }
      if (abortControllerRef.current) {
        abortControllerRef.current.abort();
      }
    };
  }, [query, fetchData]);

  const refetch = useCallback(() => {
    fetchData(query);
  }, [query, fetchData]);

  return { data, loading, error, refetch };
};

export default useSearch;
```

### 4. Why Cancelling In-Flight Requests is Important

**The Problem It Prevents:**
Without cancelling in-flight requests, you would encounter the **"stale data" bug** where:
1. User types "apple" → request A starts
2. User quickly types "appl" → request B starts
3. User quickly types "app" → request C starts
4. Request A completes with "apple" results while user sees "app" in the input
5. The UI shows "apple" results for the "app" search

**Why It's Important:**
- Prevents race conditions where older responses overwrite newer ones
- Improves performance by not processing unnecessary requests
- Prevents memory leaks from abandoned fetch operations
- Provides a better user experience with immediate feedback

**Implementation Details:**
- We use `AbortController` to cancel pending requests
- Each new search creates a new controller and aborts the previous one
- We clean up controllers in the effect cleanup function
- We specifically check for `AbortError` to distinguish between actual errors and cancelled requests

The cancellation is implemented in the `useSearch` hook:
```tsx
// Abort previous request if exists
if (abortControllerRef.current) {
  abortControllerRef.current.abort();
}

abortControllerRef.current = new AbortController();
const signal = abortControllerRef.current.signal;

// In fetch options
fetch(`/api/search?q=${encodeURIComponent(searchQuery)}`, { signal });
```

This ensures that only the most recent request will complete and update the UI.

### **2. Migration: Adding a `slug` Field to `Post` (10M Rows)**

1. **Add the Column (Non-Nullable, Default Value):**

2. **Create a Unique Index (Concurrently):**
   ```sql
   CREATE UNIQUE INDEX CONCURRENTLY idx_post_slug_unique ON "Post" ("slug");
   ```

   - `CONCURRENTLY` avoids locking the table.

3. **Update Slugs for Existing Rows (Batch Process):**
   - Run a background job (e.g., cron) to update slugs in batches:
     ```sql
     UPDATE "Post" SET "slug" = generate_slug("title") WHERE "slug" IS NULL;
     ```
   - Use a **transaction batch** (e.g., 10K rows at a time) to avoid long-running queries.

4. **Drop the Default (Optional):**
   ```sql
   ALTER TABLE "Post" ALTER COLUMN "slug" DROP DEFAULT;
   ```

**Tools/Libraries:**
- Use a library like [`slugify`](https://www.npmjs.com/package/slugify) to generate slugs deterministically.
- For large-scale updates, consider a **zero-downtime migration tool** like [Sailpoint](https://github.com/sailpoint/sailpoint) or [Ghost](https://github.com/ghostferry/ghost).

### **3. Diagnosing and Fixing a Slow `findMany` Query (800ms)**
#### **Problem:**
```typescript
const posts = await prisma.post.findMany({
  where: { tags: { some: { name: "backend" } } },
});
```

#### **Diagnosis:**
1. **Check Generated SQL:**
   Prisma likely generates:
   ```sql
   SELECT * FROM "Post"
   WHERE EXISTS (
     SELECT 1 FROM "Tag"
     WHERE "Tag"."postId" = "Post"."id"
     AND "Tag"."name" = 'backend'
   );
   ```
   - **Issue:** No index on `Tag(postId, name)` → Full table scan on `Tag`.

2. **Add an Index:**
   ```sql
   CREATE INDEX idx_tag_postid_name ON "Tag" ("postId", "name");
   ```
   - This speeds up the `EXISTS` subquery.

3. **Optimize the Prisma Query:**
   ```typescript
   const posts = await prisma.post.findMany({
     where: { tags: { some: { name: "backend" } } },
     include: { tags: true }, // Only include if needed
     select: { id: true, title: true }, // Prune fields
   });
   ```

4. **Alternative: Use `has` Instead of `some` (Prisma v5+):**
   ```typescript
   where: { tags: { has: "backend" } } // May generate a JOIN instead of EXISTS
   ```

5. **Verify with `EXPLAIN ANALYZE`:**
   ```sql
   EXPLAIN ANALYZE
   SELECT * FROM "Post"
   WHERE EXISTS (...);
   ```
   - Look for `Seq Scan` (bad) vs. `Index Scan` (good).

---

### **4. When to Drop Down to Raw SQL**
**Example:** Fetching **paginated posts with tags and comment counts** efficiently.

#### **Prisma Limitation:**
Prisma’s `findMany` with `include` and `_count` generates **multiple queries** or complex SQL that’s hard to optimize.

#### **Raw SQL Solution:**
```sql
WITH post_tags AS (
  SELECT p.id, jsonb_agg(t.name) AS tag_names
  FROM "Post" p
  JOIN "Tag" t ON p.id = t."postId"
  WHERE p.published = true
  GROUP BY p.id
)
SELECT
  p.*,
  pt.tag_names,
  (SELECT COUNT(*) FROM "Comment" c WHERE c."postId" = p.id) AS comment_count
FROM "Post" p
LEFT JOIN post_tags pt ON p.id = pt.id
WHERE p.published = true
ORDER BY p."createdAt" DESC
LIMIT 10 OFFSET 0;
```

**Why Raw SQL?**
- **Performance:** Single query with JOINs and CTEs (no N+1).
- **Flexibility:** Use PostgreSQL-specific features like `jsonb_agg` or window functions.
- **Complex Aggregations:** Easier to handle nested counts/filters.

**When to Avoid Raw SQL:**
- For simple CRUD (use Prisma).
- When schema changes frequently (Prisma migrations help).

---
### **Summary of Key Takeaways:**
1. **Optimized Queries:** Use `_count`, `select`, and `take` to avoid N+1.
2. **Migrations:** Use `CONCURRENTLY` and batch updates for large tables.
3. **Slow Queries:** Add indexes on join/where columns and analyze with `EXPLAIN`.
4. **Raw SQL:** Use for complex aggregations or when Prisma’s ORM overhead is prohibitive.

**Meaning:**

### **4. Zero-Downtime Deployment with Schema Change**
**Assumptions:**
- Additive schema change (new nullable column).
- Using **PostgreSQL** with **migrations** (e.g., `prisma`, `knex`, or raw SQL).

**Deployment Sequence:**

#### **1. Prepare the New Version**
- Write a **forward migration** (add column):
  ```sql
  ALTER TABLE users ADD COLUMN new_column TEXT;
  ```
- Test locally with the new schema.

#### **2. Deploy Backend First**
- Deploy the new backend version **without** running migrations yet.
- Ensure the backend can handle both old and new schemas (e.g., default `NULL` for the new column).

#### **3. Deploy Database Migration**
- Run the migration **on a replica** first (if using read replicas):
  ```sh
  # Example with Prisma
  npx prisma migrate dev --name add_new_column
  ```
- Verify the migration works on the replica.

#### **4. Switch Traffic to New Backend**
- Update load balancer/DNS to point to the new backend.
- The backend will now see the new schema but can still read old data (column is nullable).

#### **5. Migrate Primary Database (Minimal Downtime)**
- **Option A: Blue-Green Migration** (if possible):
  - Spin up a new primary DB instance with the migration applied.
  - Switch applications to the new primary.
- **Option B: Online Migration** (for PostgreSQL):
  ```sql
  -- Step 1: Add column with default NULL
  ALTER TABLE users ADD COLUMN new_column TEXT;

  -- Step 2: Backfill data (if needed) in a separate process
  UPDATE users SET new_column = 'default' WHERE new_column IS NULL;

  -- Step 3: Drop default (if no longer needed)
  ALTER TABLE users ALTER COLUMN new_column DROP DEFAULT;
  ```
  - Do this during low-traffic periods.

#### **6. Verify and Rollback Plan**
- Monitor for errors (e.g., `pg_hba.conf` misconfigurations).
- Have a rollback script ready (e.g., revert migration or switch back to old backend).

#### **7. Clean Up Old Backend**
- Once confirmed stable, remove the old backend version.

**Key Tools:**
- **Zero-downtime migrations**: Use `gh-ost` (GitHub) or `pt-online-schema-change` (Percona) for large tables.
- **Feature flags**: Hide new features until the migration is complete.

## Question 9
**Q:** You are building a collaborative document editor (like Google Docs, simplified):
- Multiple users can edit the same document simultaneously
- Each user sees others' cursors in realtime
- Changes are broadcast to all connected clients
- If a user disconnects and reconnects, they get the latest document state

Implement the server side in Node.js using the ws library:
1. Connection handling — how do you associate a WebSocket connection with a user and document?
2. Message protocol — define the message types (JSON) for: join document, text change, cursor move, user left
3. Broadcasting — when user A makes a change, how do you send it to all other users in the same document but NOT back to user A?
4. Reconnection — how do you handle a user rejoining after disconnect?
5. What happens if two users edit the exact same position simultaneously? You don't need to implement OT/CRDT — just explain the problem and name the solution approaches.

Write the core server implementation (connection map, message routing, broadcast logic).

**A:**
Here's a Node.js implementation using the `ws` library for a collaborative document editor. This covers the core server-side functionality you requested:

```javascript
const WebSocket = require('ws');
const crypto = require('crypto');

// Server setup
const wss = new WebSocket.Server({ port: 8080 });

// Data structures to track connections and documents
const documents = new Map(); // docId -> { text: string, users: Map<userId, WebSocket> }
const userSessions = new Map(); // userId -> { docId: string, ws: WebSocket, cursor: {pos: number, userId: string} }

// Message types
const MESSAGE_TYPES = {
  JOIN_DOCUMENT: 'JOIN_DOCUMENT',
  TEXT_CHANGE: 'TEXT_CHANGE',
  CURSOR_MOVE: 'CURSOR_MOVE',
  USER_LEFT: 'USER_LEFT',
  DOCUMENT_STATE: 'DOCUMENT_STATE'
};

// Helper functions
function generateUserId() {
  return crypto.randomBytes(8).toString('hex');
}

function broadcastToDoc(docId, message, excludeWs = null) {
  const doc = documents.get(docId);
  if (!doc) return;

  doc.users.forEach((ws, userId) => {
    if (ws !== excludeWs && ws.readyState === WebSocket.OPEN) {
      ws.send(JSON.stringify(message));
    }
  });
}

function handleTextChange(docId, userId, change) {
  const doc = documents.get(docId);
  if (!doc) return;

  // Apply the change to the document
  const { start, end, text } = change;
  const newText = doc.text.substring(0, start) + text + doc.text.substring(end);

  // Update document state
  doc.text = newText;

  // Broadcast to all other users
  broadcastToDoc(docId, {
    type: MESSAGE_TYPES.TEXT_CHANGE,
    userId,
    change,
    text: newText
  });
}

function handleCursorMove(docId, userId, cursorPos) {
  const doc = documents.get(docId);
  if (!doc) return;

  // Update cursor position
  userSessions.set(userId, {
    ...userSessions.get(userId),
    cursor: { pos: cursorPos, userId }
  });

  // Broadcast to all other users
  broadcastToDoc(docId, {
    type: MESSAGE_TYPES.CURSOR_MOVE,
    userId,
    cursor: { pos: cursorPos, userId }
  });
}

// WebSocket connection handling
wss.on('connection', (ws) => {
  const userId = generateUserId();

  ws.on('message', (message) => {
    try {
      const data = JSON.parse(message);

      switch (data.type) {
        case MESSAGE_TYPES.JOIN_DOCUMENT:
          const { docId } = data;
          if (!documents.has(docId)) {
            documents.set(docId, {
              text: '',
              users: new Map()
            });
          }

          // Add user to document
          const doc = documents.get(docId);
          doc.users.set(userId, ws);

          // Store user session
          userSessions.set(userId, {
            docId,
            ws,
            cursor: { pos: 0, userId }
          });

          // Send current document state
          ws.send(JSON.stringify({
            type: MESSAGE_TYPES.DOCUMENT_STATE,
            text: doc.text,
            cursors: Array.from(doc.users.entries())
              .filter(([id]) => id !== userId)
              .map(([id, _]) => userSessions.get(id).cursor)
          }));

          // Notify other users
          broadcastToDoc(docId, {
            type: MESSAGE_TYPES.USER_LEFT,
            userId: null, // null means user joined
            userId: userId
          }, ws);

          break;

        case MESSAGE_TYPES.TEXT_CHANGE:
          handleTextChange(data.docId, userId, data.change);
          break;

        case MESSAGE_TYPES.CURSOR_MOVE:
          handleCursorMove(data.docId, userId, data.cursorPos);
          break;
      }
    } catch (err) {
      console.error('Error processing message:', err);
    }
  });

  ws.on('close', () => {
    // Find which document this user was in
    const session = Array.from(userSessions.entries()).find(
      ([_, sessionData]) => sessionData.ws === ws
    );

    if (session) {
      const [userId, sessionData] = session;
      const doc = documents.get(sessionData.docId);

      if (doc) {
        // Remove user from document
        doc.users.delete(userId);

        // Notify other users
        broadcastToDoc(sessionData.docId, {
          type: MESSAGE_TYPES.USER_LEFT,
          userId: userId
        });

        // Clean up session
        userSessions.delete(userId);
      }
    }
  });
});

// Reconnection handling
function handleReconnection(ws, userId) {
  const session = userSessions.get(userId);
  if (!session) return false;

  // Check if the document still exists
  if (!documents.has(session.docId)) {
    userSessions.delete(userId);
    return false;
  }

  // Reconnect the WebSocket
  ws.on('message', (message) => {
    try {
      const data = JSON.parse(message);
      // ... same message handling as above
    } catch (err) {
      console.error('Error processing reconnected message:', err);
    }
  });

  // Send current document state
  const doc = documents.get(session.docId);
  ws.send(JSON.stringify({
    type: MESSAGE_TYPES.DOCUMENT_STATE,
    text: doc.text,
    cursors: Array.from(doc.users.entries())
      .filter(([id]) => id !== userId)
      .map(([id, _]) => userSessions.get(id).cursor)
  }));

  return true;
}
```

### Key Components Explained:

1. **Connection Handling**:
   - Each WebSocket connection gets a unique `userId`
   - Users join documents by sending a `JOIN_DOCUMENT` message with a `docId`
   - We maintain two maps:
     - `documents`: Tracks document state and connected users per document
     - `userSessions`: Tracks each user's connection and cursor position

2. **Message Protocol**:
   ```json
   // Join document
   {
     "type": "JOIN_DOCUMENT",
     "docId": "string"
   }

   // Text change
   {
     "type": "TEXT_CHANGE",
     "docId": "string",
     "change": {
       "start": number,
       "end": number,
       "text": "string"
     }
   }

   // Cursor move
   {
     "type": "CURSOR_MOVE",
     "docId": "string",
     "cursorPos": number
   }

   // User left
   {
     "type": "USER_LEFT",
     "userId": "string"
   }

   // Document state (sent on join/reconnect)
   {
     "type": "DOCUMENT_STATE",
     "text": "string",
     "cursors": [
       {"pos": number, "userId": "string"}
     ]
   }
   ```

3. **Broadcasting**:
   - The `broadcastToDoc` function sends messages to all users in a document except the sender
   - Text changes and cursor moves are broadcast to all other users in the same document

4. **Reconnection**:
   - When a user reconnects, they send their `userId` (could be stored in localStorage)
   - The server checks if their session exists and sends the current document state
   - If the document doesn't exist, the session is cleaned up

5. **Conflict Resolution (Same Position Edits)**:
   **Problem**: When two users edit the same position simultaneously, the last change wins and the first is lost.

   **Solution Approaches**:
   - Operational Transformation (OT): Transform operations based on other operations to maintain convergence
   - Conflict-free Replicated Data Types (CRDTs): Data structures that guarantee eventual consistency
   - Last-write-wins: Simple but loses data (what we have in this basic implementation)
   - Position tracking: Require users to specify exact positions they're editing

This implementation provides the core functionality for a collaborative editor. For production use, you'd want to add:
- Authentication
- More robust error handling
- Document persistence
- Better conflict resolution
- Performance optimizations for large documents
- Typing indicators (when users are typing)

### 5. GraphQL vs REST Decision Guide

**Choose GraphQL when:**
1. **Client needs precise data**: Avoid over-fetching (REST often returns fixed schemas)
2. **Multiple client types**: Mobile, web, and IoT can request exactly what they need
3. **Real-time updates**: Works well with subscriptions (REST requires polling)
4. **Complex relationships**: Deeply nested data is easier to fetch in one request
5. **Evolving requirements**: Schema can evolve without breaking clients
6. **API versioning**: No need to version endpoints (clients get exactly what they ask for)

**Avoid GraphQL when:**
1. **Simple CRUD APIs**: REST may be simpler for basic operations
2. **Caching challenges**: GraphQL responses vary by client, making caching harder
3. **Performance-critical**: Initial load can be heavier (though this can be mitigated)
4. **Team unfamiliarity**: Steeper learning curve than REST
5. **Strict content negotiation**: REST's uniform interface is simpler for some use cases
6. **Microservices boundaries**: GraphQL works best when you control the entire stack
7. **Legacy systems**: May require complex bridges to existing REST APIs

**Hybrid Approach:**
Many modern APIs use both:
- GraphQL for complex queries
- REST for simple operations or when GraphQL overhead isn't justified

**Example where REST might be better:**
- Public API with many clients who need the same data
- Simple admin interfaces where all fields are always needed
- When you need to leverage existing REST infrastructure without modification

## Question 12
**Q:** You are asked to add real-time notifications to the existing project management tool. Users should be notified when: a task is assigned to them, a comment is added to their task, a deadline is approaching (within 24h).

Design the notification system:
1. What transport mechanism would you use and why? (WebSocket, SSE, polling, push notifications — compare and justify)
2. How do you store notifications? Design the database schema.
3. How do you handle a user who is offline — do they get missed notifications when they come back online?
4. The deadline check needs to run every hour for all tasks. How do you implement this? (cron job, queue, event-driven?) — show the implementation approach.
5. How do you scale this if there are 100,000 concurrent users?

**A:**
Here’s a comprehensive design for the real-time notification system in your project management tool, addressing scalability, reliability, and user experience:

---

### **1. Notification Transport Mechanism**
**Recommended:** **WebSocket (with fallback to Server-Sent Events (SSE))**
**Why?**
- **WebSocket**:
  - **Bidirectional, persistent connection**: Ideal for real-time updates (e.g., task assignments, comments).
  - **Low latency**: No need to re-establish connections for each update (unlike polling).
  - **Efficient**: Only sends data when events occur (unlike polling’s fixed intervals).
  - **Supports multiple event types**: Can handle task assignments, comments, and deadlines in the same connection.
- **Fallback to SSE**:
  - If WebSocket isn’t supported (e.g., older browsers), SSE provides a unidirectional, server-to-client stream.
- **Avoid Polling**:
  - Inefficient (high server load, unnecessary requests).
- **Push Notifications (Mobile/App)**:
  - Use **Firebase Cloud Messaging (FCM)** or **Apple Push Notification Service (APNS)** for mobile apps when the user’s app is in the background/closed.
  - WebSocket/SSE can still be used when the app is open.

**Architecture**:
- **Frontend**: WebSocket/SSE client library (e.g., `socket.io-client`, native WebSocket API).
- **Backend**: WebSocket server (e.g., Node.js with `ws`, Python with `websockets`, or a managed service like Pusher/Ably).
- **Mobile**: FCM/APNS for push + WebSocket for in-app real-time updates.

---

### **2. Database Schema for Notifications**
Store notifications in a **dedicated `notifications` table** with the following schema:
```sql
CREATE TABLE notifications (
  id BIGSERIAL PRIMARY KEY,
  user_id UUID REFERENCES users(id),       -- Who the notification is for
  task_id UUID REFERENCES tasks(id),        -- Related task (optional, for context)
  type ENUM('task_assigned', 'comment_added', 'deadline_approaching', 'mention'),  -- Notification type
  payload JSONB,                           -- Structured data (e.g., {comment: "text", author: "user_id"})
  is_read BOOLEAN DEFAULT FALSE,           -- Read status
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  metadata JSONB,                          -- Additional data (e.g., deadline timestamp)
  INDEX (user_id),                         -- Fast queries for user's notifications
  INDEX (type),                            -- Optional: Filter by type
  INDEX (is_read)                          -- Optimize for read/unread counts
);
```
**Additional Tables**:
- **`notification_preferences`** (user-specific settings, e.g., mute deadlines):
  ```sql
  CREATE TABLE notification_preferences (
    user_id UUID PRIMARY KEY REFERENCES users(id),
    task_assigned BOOLEAN DEFAULT TRUE,
    comment_added BOOLEAN DEFAULT TRUE,
    deadline_approaching BOOLEAN DEFAULT TRUE,
    mention BOOLEAN DEFAULT TRUE
  );
  ```

**Why This Design?**
- **Flexibility**: `payload` and `metadata` store unstructured data (e.g., comment text, deadline time).
- **Performance**: Indexes on `user_id` and `is_read` optimize queries.
- **Scalability**: Denormalized for read-heavy workloads (notifications are read frequently).

---

### **3. Handling Offline Users**
**Approach**:
1. **Client-Side Queue**:
   - If the WebSocket/SSE connection drops, the frontend buffers notifications locally (e.g., in `localStorage` or IndexedDB).
   - On reconnect, the client sends a request to fetch missed notifications (e.g., `GET /notifications?since=last_seen_at`).
2. **Server-Side Persistence**:
   - Notifications are stored in the database until the user fetches them (even if offline).
   - Use a **last_seen_at** timestamp in the `users` table to track when the user was last active.
3. **Sync on Reconnect**:
   - When the user comes back online, the server sends all unread notifications since `last_seen_at` via WebSocket/SSE or a bulk API call.
4. **Push Notifications**:
   - For mobile, use FCM/APNS to deliver notifications even if the app is closed. The app fetches details on launch.

**Example Flow**:
- User goes offline → Server continues to store notifications.
- User comes back online → Frontend syncs missed notifications via:
  ```http
  GET /notifications?since=2026-03-10T14:30:00Z
  ```
  → Server returns all unread notifications created after this time.

---

### **4. Deadline Check Implementation**
**Approach**: **Event-Driven with a Queue (Recommended)**
**Why Not Cron?**
- Cron runs on a fixed schedule but can’t handle dynamic workloads (e.g., 100K users with deadlines at the same time).
- Cron jobs are less resilient to failures.

**Implementation**:
1. **Database Trigger or Application Logic**:
   - When a task’s deadline is set/updated, immediately enqueue a **deadline check event** for 24 hours before the deadline.
   - Example (pseudo-code):
     ```python
     def on_task_deadline_set(deadline):
         check_time = deadline - timedelta(hours=24)
         enqueue_deadline_check(check_time, task_id)
     ```
2. **Queue System**:
   - Use a **distributed queue** (e.g., RabbitMQ, AWS SQS, or Kafka) to process deadline checks asynchronously.
   - Workers pick up events and send notifications when the deadline is approaching.
3. **Worker Process**:
   - Workers run continuously (not cron-based) and process events as they arrive.
   - Example worker logic:
     ```python
     def process_deadline_check(event):
         task = get_task(event.task_id)
         if is_within_24_hours(task.deadline):
             send_notification(task.assignee, "deadline_approaching", task)
     ```
4. **Scaling Workers**:
   - Add more workers horizontally to handle load spikes (e.g., many deadlines on the same day).

**Alternative: Scheduled Jobs with a Library**
- Use libraries like **ArangoDB’s scheduled jobs** or **Sidekiq (Ruby)** to run delayed jobs, but queues are more scalable.

**Database Optimization**:
- Add a `next_deadline_check` column to the `tasks` table to avoid scanning all tasks hourly:
  ```sql
  ALTER TABLE tasks ADD COLUMN next_deadline_check TIMESTAMP;
  ```
- Update this column when deadlines change.

### **5. Scaling to 100,000 Concurrent Users**
**Challenges**:
- WebSocket connections: ~100K concurrent connections can overwhelm a single server.
- Database load: High write throughput for notifications.
- Queue processing: Deadline checks must scale.

**Solutions**:
| Component          | Scaling Strategy                                                                 |
|--------------------|---------------------------------------------------------------------------------|
| **WebSocket Server** | Use a **load-balanced WebSocket cluster** (e.g., Nginx with WebSocket support, or Pusher/Ably). |
|                    | **Horizontal scaling**: Deploy multiple WebSocket servers behind a load balancer. |
|                    | **Connection pooling**: Reuse connections for related events (e.g., task updates). |
| **Database**       | **Read replicas**: Offload notification reads to replicas.                       |
|                    | **Sharding**: Split `notifications` table by `user_id` (e.g., consistent hashing). |
|                    | **Caching**: Cache unread counts (e.g., Redis) to reduce DB load.               |
| **Queue**          | **Distributed queue**: Use RabbitMQ/Kafka with multiple consumers.               |
|                    | **Partitioning**: Split deadlines by time ranges (e.g., shard by `deadline_date`). |
| **Push Notifications** | Use **FCM/APNS** with topic-based messaging to reduce server load.              |
| **Frontend**       | **Debounce API calls**: Throttle notification fetches (e.g., 1 call per second). |

**Architecture Diagram**:
```
User → [Load Balancer] → [WebSocket Cluster (Nginx + Workers)]
                     → [Database (Sharded)]
                     → [Queue (RabbitMQ/Kafka)]
                     → [Worker Pool (Deadline Checks)]
                     → [Push Notification Service (FCM/APNS)]
```

**Key Optimizations**:
1. **WebSocket Load Balancing**:
   - Use **sticky sessions** to ensure a user’s WebSocket stays on the same server (reduces connection overhead).
   - Example with Nginx:
     ```nginx
     upstream websocket_servers {
         ip_hash;  # Sticky sessions
         server ws1.example.com;
         server ws2.example.com;
     }
     ```
2. **Database Sharding**:
   - Shard the `notifications` table by `user_id % N` (where `N` is the number of shards).
   - Example: `user_id 1234 → shard 2` (if `N=3`).
3. **Queue Partitioning**:
   - Partition the queue by `deadline_date` (e.g., `deadline_2026-03-12` topic).
   - Workers subscribe to relevant partitions.
4. **Caching**:
   - Cache unread notification counts in Redis:
     ```bash
     SET user:123:unread_count 5 EX 3600  # Expire after 1 hour
     ```

**Fallback for Overload**:
- If the WebSocket cluster is overwhelmed, fall back to **polling** (e.g., "Check for updates every 30 seconds").
- Use **gradual feature rollout**: Enable WebSocket for new users first, then migrate others.

---

### **6. Example Workflow**
1. **Task Assigned**:
   - Backend publishes to WebSocket channel: `user:123:notifications`.
   - Frontend subscribes to this channel and displays the notification.
   - If offline, notification is stored in DB and synced on reconnect.
2. **Comment Added**:
   - Same as above, but with `type = "comment_added"` and `payload` containing the comment text.
3. **Deadline Approaching**:
   - Queue worker picks up the event 24h before deadline → sends WebSocket/SSE message.
   - If the user is offline, the notification waits in the DB until they reconnect.

- **Queue Metrics**: Track pending/dead-letter messages in RabbitMQ/Kafka.
- **Database**: Monitor query performance and shard balance.
- **Alerts**: Set up alerts for:
  - High WebSocket connection drops.
  - Queue backlog > 1000 messages.
  - Database replication lag.
