const router = Router();

interface TaskUpdateRequest extends Request {
  params: {

    taskId: string;
  };

router.patch(
  '/workspaces/:workspaceId/tasks/:taskId',
  async (req: TaskUpdateRequest, res: Response, next: NextFunction) => {

      const parsedBody = taskUpdateSchema.parse(req.body);

      const workspaceId = parseInt(req.params.workspaceId);
      const taskId = parseInt(req.params.taskId);

      const workspace = await getWorkspace(workspaceId);
      if (!workspace) {
        return res.sendStatus(404);
      }

      const task = await getTask(taskId);
      if (!task || task.workspace_id !== workspace.id) {
        return res.sendStatus(404);
      }

      if (parsedBody.assignee_id && !workspace.members.includes(parsedBody.assignee_id)) {
        return res.status(400).json({ error: 'Assignee is not a member of the workspace' });

        task.subtasks.some((subtask) => subtask.status !== 'done' && subtask.status !== 'cancelled')

        return res.sendStatus(409);
      }

      await updateTask(taskId, parsedBody);
      res.sendStatus(204);
    } catch (error) {
      if (error instanceof z.ZodError) {
        return res.status(400).json({ error: error.issues });
      }

**Q:** Continuing from the previous auth system:

1. POST /auth/refresh — use the refresh token from cookie to issue a new access token. Rotate the refresh token (issue new one, invalidate old one).
2. POST /auth/logout — invalidate the refresh token, clear the cookie
3. Implement refresh token rotation with a Redis-based token store (show the Redis data structure and TTL strategy)
4. How do you handle the case where a refresh token is used twice? (token reuse detection) — implement it
5. A user changes their password — how do you invalidate ALL their existing sessions? Implement it.
6. What is the difference between token blacklisting and token rotation? When would you use each?

**A:**
Sure, let's break down each of these requirements:

1. **POST /auth/refresh:**
   - Fetch the refresh token from the cookie.
   - Verify the refresh token with the Redis store (using its TTL for expiration check).
   - If valid, generate a new access token and rotate the refresh token by:
     - Invalidate the old refresh token in Redis.
     - Generate a new refresh token and save it in Redis with an appropriate TTL (e.g., 30 days).
     - Set the new refresh token in the cookie.
   - Return the new access token.

   **Redis data structure:**
   - `REFRESH_TOKENS:{user_id}:{refresh_token}`: Stores the refresh token with an expiration time (TTL).

   **TTL strategy:**
   - Refresh tokens have a TTL of 30 days. When a refresh token is used, it's rotated (new token is issued and old one is invalidated), and the new token's TTL is set to 30 days.

2. **POST /auth/logout:**
   - Fetch the refresh token from the cookie.
   - Invalidate the refresh token in the Redis store.
   - Clear the refresh token cookie.
   - Optionally, you can also invalidate the access token by removing it from a separate Redis store (`ACCESS_TOKENS:{user_id}:{access_token}`) if it's still valid.

3. **Handling refresh token reuse:**
   - When a refresh token is used, set a `used` flag in Redis along with its TTL.
   - Before using a refresh token, check if it has already been used. If so, return an error.
   - This prevents a refresh token from being used more than once, even if it hasn't expired yet.

4. **Invalidating all sessions when a user changes their password:**
   - When a user changes their password:
     - Invalidate all access tokens in the Redis store (`ACCESS_TOKENS:{user_id}:{access_token}`).
     - Invalidate the refresh token in the Redis store (`REFRESH_TOKENS:{user_id}:{refresh_token}`).
     - Optionally, you can also clear the refresh token cookie on the client side by sending a `Set-Cookie` header with the refresh token set to an empty value or an invalid token.

5. **Token blacklisting vs Token rotation:**
   - **Token blacklisting** involves maintaining a list of invalid tokens (e.g., compromised tokens or tokens from logout requests). When a token is presented, you check if it's in the blacklist before allowing access. This is useful when you want to quickly invalidate a token (e.g., logout), but it requires additional storage and lookup time for the blacklist.
   - **Token rotation** involves generating new tokens and invalidating old ones whenever a token is used or at regular intervals. This provides better security by minimizing the time a token is valid, but it requires additional processing to generate and manage new tokens.
   - You would use token blacklisting when you need to quickly invalidate tokens, such as during logout or password reset. Token rotation is useful for providing better security by minimizing token lifetime.

  const accessToken = jwt.sign({ userId }, 'access_secret', { expiresIn: '1h' });
  const refreshToken = jwt.sign({ userId }, 'refresh_secret', { expiresIn: '30d' });

  await client.setAsync(`REFRESH_TOKENS:${userId}:${newRefreshToken}`, 1, 'EX', 30 * 24 * 60 * 60);

  await client.delAsync(`REFRESH_TOKENS:${userId}`);
  await client.delAsync(`ACCESS_TOKENS:${userId}`);

async function verifyRefreshToken(userId, refreshToken) {
  const isUsed = await client.existsAsync(`REFRESH_TOKENS:${userId}:${refreshToken}:used`);
  if (isUsed) {
    throw new Error('Refresh token already used');

**A:**
To implement a collaborative document editor server in Node.js using the `ws` library, we'll follow these steps:

1. **Connection Handling**: We'll maintain a connection map that associates WebSocket connections with users and documents. Each user will be represented by a unique ID, and we'll use a `Map` to store the connections.

2. **Message Protocol**: We'll define the following message types in JSON format:

   - `joinDocument`: Sent when a user joins a document. It contains the user ID and the document ID.
     ```json
     {
       "type": "joinDocument",
       "userId": "user123",
       "documentId": "doc456"
     }
     ```
   - `textChange`: Sent when a user makes a text change. It contains the user ID, document ID, start index, and new text.
     ```json
     {
       "type": "textChange",
       "userId": "user123",
       "documentId": "doc456",
       "start": 5,
       "text": "new text"
     }
     ```
   - `cursorMove`: Sent when a user moves their cursor. It contains the user ID, document ID, and the new cursor position.
     ```json
     {
       "type": "cursorMove",
       "userId": "user123",
       "documentId": "doc456",
       "cursorPosition": 10
     }
     ```
   - `userLeft`: Sent when a user disconnects. It contains the user ID and document ID.
     ```json
     {
       "type": "userLeft",
       "userId": "user123",
       "documentId": "doc456"
     }
     ```

3. **Broadcasting**: When a user makes a change, we'll broadcast the update to all other connected users in the same document, excluding the sender. We'll use a `Set` to store all active connections for a given document and iterate through it, sending the message to each connection except the sender's.

4. **Reconnection**: To handle users rejoining after disconnecting, we'll maintain the current document state on the server. When a user reconnects, they'll send a `joinDocument` message with their user ID and the document ID they were previously editing. The server will then send the latest document state to the reconnected user.

5. **Handling Simultaneous Edits**: If two users edit the exact same position simultaneously, we'll follow the "last write wins" approach. This means that the second user's change will overwrite the first user's change. To minimize conflicts, we can implement some basic conflict resolution strategies like merging changes if possible or displaying a visual indication of conflicting changes to users.

Here's the core server implementation using the `ws` library:

```javascript
const WebSocket = require('ws');

const wss = new WebSocket.Server({ port: 8080 });
const connections = new Map();
const documentConnections = new Map();

wss.on('connection', (ws) => {
  ws.isAlive = true;
  ws.on('pong', () => {
    ws.isAlive = true;
  });

  ws.on('message', (message) => {
    const data = JSON.parse(message.toString());
    switch (data.type) {
      case 'joinDocument':
        handleJoinDocument(ws, data.userId, data.documentId);
        break;
      case 'textChange':
        handleTextChange(ws, data);
        break;
      case 'cursorMove':
        handleCursorMove(ws, data);
        break;
      case 'userLeft':
        handleUserLeft(ws, data.userId, data.documentId);
        break;
    }
  });

  ws.on('close', () => {
    ws.isAlive = false;
    for (const [documentId, connectionsSet] of documentConnections) {
      if (connectionsSet.has(ws)) {
        connectionsSet.delete(ws);
        if (connectionsSet.size === 0) {
          documentConnections.delete(documentId);
        }
      }
    }
  });
});

function heartbeat() {
  wss.clients.forEach((ws) => {
    if (!ws.isAlive) return ws.terminate();
    ws.isAlive = false;
    ws.ping(() => {});
  });
}

setInterval(heartbeat, 30000);

function handleJoinDocument(ws, userId, documentId) {
  if (!connections.has(userId)) {
    connections.set(userId, ws);
  }

  if (!documentConnections.has(documentId)) {
    documentConnections.set(documentId, new Set());
  }

  documentConnections.get(documentId).add(ws);

  // Send the latest document state to the rejoining user
  ws.send(JSON.stringify({ type: 'documentState', documentId, state: /* retrieve document state */ }));
}

function handleTextChange(ws, data) {
  broadcastToAllExcept(ws, data.documentId, 'textChange', data);
}

function handleCursorMove(ws, data) {
  broadcastToAllExcept(ws, data.documentId, 'cursorMove', data);
}

function handleUserLeft(ws, userId, documentId) {
  broadcastToAllExcept(ws, documentId, 'userLeft', { userId, documentId });
}

function broadcastToAllExcept(sender, documentId, messageType, data) {
  if (!documentConnections.has(documentId)) return;

  const connectionsSet = documentConnections.get(documentId);
  connectionsSet.forEach((ws) => {
    if (ws.readyState === WebSocket.OPEN && ws !== sender) {
      ws.send(JSON.stringify({ type: messageType, ...data }));
    }
  });
}
```
