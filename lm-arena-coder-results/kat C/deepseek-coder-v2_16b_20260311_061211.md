import React, { useState, useEffect, useMemo } from 'react';
import { Table, TableBody, TableCell, TableHead, TableRow, TableSortLabel, Checkbox, Pagination } from '@mui/material';

interface Column<T> {
  key: keyof T;
  header: string;
  width?: number;
  sortable?: boolean;
  renderCell?: (row: T) => ReactNode;
}

interface DataTableProps<T> {
  columns: Column<T>[];
  data: T[];
  pageSizeOptions?: number[];
  initialPageSize?: number;
  controlled?: boolean;
  defaultSelectedRows?: T[];
  onSelectRows?: (selectedRows: T[]) => void;
}

function DataTable<T>({ columns, data, pageSizeOptions = [10, 25, 50], initialPageSize = 10, controlled = false, defaultSelectedRows = [], onSelectRows }: DataTableProps<T>) {
  const [selectedRows, setSelectedRows] = useState<T[]>(defaultSelectedRows);
  const [page, setPage] = useState(0);
  const [pageSize, setPageSize] = useState(initialPageSize);
  const [sortConfig, setSortConfig] = useState<{ key: keyof T | null; direction: 'asc' | 'desc' | null }>({ key: null, direction: null });

  const handleSelectAll = (event: React.ChangeEvent<HTMLInputElement>) => {
    if (event.target.checked) {
      setSelectedRows(data.slice(page * pageSize, (page + 1) * pageSize));
    } else {
      setSelectedRows([]);
    }
  };

  const handleSelectRow = (row: T) => {
    if (selectedRows.includes(row)) {
      setSelectedRows(selectedRows.filter(selectedRow => selectedRow !== row));
    } else {
      setSelectedRows([...selectedRows, row]);
    }
  };

  useEffect(() => {
    if (controlled && onSelectRows) {
      onSelectRows(selectedRows);
    }
  }, [selectedRows, controlled, onSelectRows]);

  const sortedData = useMemo(() => {
    const sortableColumns = columns.filter(column => column.sortable);
    if (sortConfig.key && sortableColumns.some(column => column.key === sortConfig.key)) {
      return data.slice().sort((a, b) => {
        const isAsc = sortConfig.direction === 'asc';
        if (a[sortConfig.key] < b[sortConfig.key]) {
          return isAsc ? -1 : 1;
        }
        if (a[sortConfig.key] > b[sortConfig.key]) {
          return isAsc ? 1 : -1;
        }
        return 0;
      });
    }
    return data;
  }, [data, sortConfig]);

  const visibleData = useMemo(() => {
    return sortedData.slice(page * pageSize, (page + 1) * pageSize);
  }, [sortedData, page, pageSize]);

  return (
    <div>
      <Table>
        <TableHead>
          <TableRow>
            {columns.map(column => (
              <TableCell key={column.key as string} style={{ width: column.width }}>
                {column.sortable ? (
                  <TableSortLabel
                    active={sortConfig.key === column.key}
                    direction={sortConfig.key === column.key ? sortConfig.direction : 'asc'}
                    onClick={() => setSortConfig(config => ({ key: column.key, direction: config.key === column.key && config.direction === 'asc' ? 'desc' : 'asc' }))}
                  >
                    {column.header}
                  </TableSortLabel>
                ) : (
                  column.header
                )}
              </TableCell>
            ))}
          </TableRow>
        </TableHead>
        <TableBody>
          {visibleData.map((row, index) => (
            <TableRow key={index} hover onClick={() => handleSelectRow(row)} selected={selectedRows.includes(row)}>
              <TableCell>
                <Checkbox checked={selectedRows.includes(row)} onChange={() => handleSelectRow(row)} />
              </TableCell>
              {columns.map(column => (
                <TableCell key={column.key as string}>
                  {column.renderCell ? column.renderCell(row) : row[column.key]}
                </TableCell>
              ))}
            </TableRow>
          ))}
        </TableBody>
      </Table>
      <Pagination
        count={Math.ceil(data.length / pageSize)}
        page={page}
        onChange={(_, newPage) => setPage(newPage - 1)}
        rowsPerPage={pageSize}
        onRowsPerPageChange={(event) => setPageSize(Number(event.target.value))}
        rowsPerPageOptions={pageSizeOptions}
      />
    </div>
  );
}

Cancelling in-flight requests is crucial to prevent multiple overlapping requests from clashing.     const url = req.url;
    const params = new URLSearchParams(url.split('?')[1]);
    const userId = params.get('userId');
    const documentId = params.get('documentId');

        return;

        connections.set(documentId, new Map());
    }

    const docConnections = connections.get(documentId);
    docConnections.set(ws, { userId, documentId });

    ws.on('message', (message) => {
        handleMessage(ws, message, documentId);

    const { userId } = docConnections.get(ws);

            broadcastCursorMove(data, documentId);
            break;

    const docConnections = connections.get(documentId);

## Question 12
**Q:** You are asked to add real-time notifications to the existing project management tool. Users should be notified when: a task is assigned to them, a comment is added to their task, a deadline is approaching (within 24h).

Design the notification system:
1. What transport mechanism would you use and why? (WebSocket, SSE, polling, push notifications — compare and justify)
2. How do you store notifications? Design the database schema.
3. How do you handle a user who is offline — do they get missed notifications when they come back online?
4. The deadline check needs to run every hour for all tasks. How do you implement this? (cron job, queue, event-driven?) — show the implementation approach.
5. How do you scale this if there are 100,000 concurrent users?

**A:**
 To address your requirements, let's break down the design of the notification system step-by-step:

### 1. Transport Mechanism
**WebSocket**: This is a real-time communication protocol that provides a full-duplex communication channel over a single TCP connection. WebSockets are suitable for real-time notifications because they allow for instant communication between the server and the client without the need for polling.

**SSE (Server-Sent Events)**: SSE is a technology where a browser receives automatic updates from a server via HTTP connection. It's a simpler alternative to WebSockets and can be used for one-way communications from the server to the client.

**Polling**: This involves clients repeatedly sending requests to the server to check for updates. While effective, it can be inefficient and resource-intensive, especially for a large number of users.

**Push Notifications**: These are sent from a server to a client's browser via an API (like Firebase Cloud Messaging, APNs for iOS, etc.). Push notifications are effective for immediate alerts but require user interaction to be enabled and may be more intrusive.

**Comparison**: WebSockets and SSE are better for real-time notifications, avoiding the need to repeatedly poll the server. Polling can be inefficient and push notifications are more intrusive but are effective for immediate alerts.

**Recommendation**: For real-time notifications in a project management tool, **WebSocket** is a good choice due to its full-duplex communication and ability to push data from the server to the client without the client requesting it.

### 2. Database Schema for Notifications
For storing notifications, you can use a simple table structure:

```sql
CREATE TABLE notifications (
    id SERIAL PRIMARY KEY,
    user_id INT NOT NULL,
    message TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    read BOOLEAN DEFAULT FALSE,
    FOREIGN KEY (user_id) REFERENCES users(id)
);
```

This schema includes:
- `id`: Unique identifier for the notification.
- `user_id`: Identifier for the user who will receive the notification.
- `message`: Content of the notification.
- `created_at`: Timestamp when the notification was created.
- `read`: Boolean flag to indicate if the notification has been read by the user.

### 3. Handling Offline Users
For users who are offline, missed notifications can be stored temporarily and sent when they come back online. This can be implemented by saving notifications in a local storage (client-side) or by using a queuing system that sends notifications once the user is online.

### 4. Implementing the Deadline Check
**Cron Job**: A cron job can be set up to run a script every hour that checks all tasks for deadlines within the next 24 hours. This is a straightforward method but can be inefficient for a large number of users.

**Queue**: Using a queue system like RabbitMQ or Kafka can handle this in a more scalable manner. The system can dispatch a task every hour to check for approaching deadlines and enqueue notifications accordingly. This approach can be more efficient and scalable.

**Event-driven**: Implementing an event-driven system where the task status is periodically checked and any changes (like a new comment or assignment) trigger a notification. This can be integrated with an event scheduler like Celery to handle periodic checks.

**Recommendation**: **Queue** is a scalable and efficient approach, especially if integrated with a distributed system. This can be more maintainable and scalable compared to cron jobs or event-driven systems.

### 5. Scaling for 100,000 Concurrent Users
- **Optimize Database**: Use a scalable database like PostgreSQL with proper indexing and sharding if necessary.
- **Caching**: Use in-memory caching (like Redis) for frequently accessed data or to reduce database load.
- **Horizontal Scaling**: Deploy multiple instances of your application and use a load balancer to distribute the load.
- **Asynchronous Processing**: Use asynchronous processing for tasks like sending notifications (using Celery or similar) to handle requests efficiently.
- **Database Partitioning**: For large datasets, consider partitioning your database tables to improve performance and manageability.

By following these steps, you can design a robust notification system that meets the requirements and scales effectively for a large number of concurrent users.
