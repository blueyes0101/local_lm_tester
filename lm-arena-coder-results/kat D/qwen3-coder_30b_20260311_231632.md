## 5. HTTP Status Codes

### General Rules
- **200 OK** - Successful GET, PUT, PATCH requests
- **201 Created** - Successful POST requests that create resources
- **204 No Content** - Successful DELETE requests or PATCH with no content
- **400 Bad Request** - Invalid request data (validation errors)
- **401 Unauthorized** - Authentication required or invalid credentials
- **403 Forbidden** - Insufficient permissions to access resource
- **404 Not Found** - Resource doesn't exist
- **409 Conflict** - Resource conflict (e.g., duplicate workspace name)
- **422 Unprocessable Entity** - Validation errors that prevent processing
- **500 Internal Server Error** - Unexpected server errors

### Specific Endpoint Status Codes

#### Workspace Management
- POST /workspaces → 201 Created
- GET /workspaces/{workspaceId} → 200 OK, 404 Not Found
- PUT /workspaces/{workspaceId} → 200 OK, 404 Not Found
- DELETE /workspaces/{workspaceId} → 204 No Content, 404 Not Found

#### Project Management
- POST /projects → 201 Created
- GET /projects/{projectId} → 200 OK, 404 Not Found
- PUT /projects/{projectId} → 200 OK, 404 Not Found
- DELETE /projects/{projectId} → 204 No Content, 404 Not Found

#### Task Management
- POST /tasks → 201 Created
- GET /tasks/{taskId} → 200 OK, 404 Not Found
- PUT /tasks/{taskId} → 200 OK, 404 Not Found
- PATCH /tasks/{taskId}/status → 200 OK, 404 Not Found
- DELETE /tasks/{taskId} → 204 No Content, 404 Not Found

#### Comment Management
- POST /comments → 201 Created
- GET /comments/{commentId} → 200 OK, 404 Not Found
- PUT /comments/{commentId} → 200 OK, 404 Not Found
- DELETE /comments/{commentId} → 204 No Content, 404 Not Found

#### Authentication
- POST /auth/login → 200 OK (success), 401 Unauthorized (failure)
- POST /auth/logout → 200 OK

### Error Status Code Examples
- Validation errors → 400 Bad Request or 422 Unprocessable Entity
- Access denied → 403 Forbidden
- Resource not found → 404 Not Found
- Duplicate resource → 409 Conflict

**A:**

```typescript
// server.ts
import express, { Application, Request, Response, NextFunction } from 'express';
import bcrypt from 'bcrypt';
import jwt from 'jsonwebtoken';
import cookieParser from 'cookie-parser';
import { PrismaClient } from '@prisma/client';

const app: Application = express();
const prisma = new PrismaClient();
const PORT = process.env.PORT || 3000;

// Middleware
app.use(express.json());
app.use(cookieParser());

// Types
interface User {
  id: number;
  email: string;
  password: string;
  role: string;
}

interface AuthenticatedRequest extends Request {
  user?: {
    userId: number;
    email: string;
    role: string;
  };
}

// JWT Secret (should be in environment variables in production)
const JWT_SECRET = process.env.JWT_SECRET || 'your-super-secret-jwt-key';
const REFRESH_TOKEN_SECRET = process.env.REFRESH_TOKEN_SECRET || 'your-refresh-token-secret';

// Authentication Middleware
const authMiddleware = (req: AuthenticatedRequest, res: Response, next: NextFunction) => {
  try {
    const token = req.headers.authorization?.split(' ')[1];
    
    if (!token) {
      return res.status(401).json({
        error: 'Access token required',
        message: 'No access token provided'
      });
    }

    const decoded = jwt.verify(token, JWT_SECRET) as {
      userId: number;
      email: string;
      role: string;
      iat: number;
      exp: number;
    };

    req.user = {
      userId: decoded.userId,
      email: decoded.email,
      role: decoded.role
    };

    next();
  } catch (error) {
    if (error instanceof jwt.TokenExpiredError) {
      return res.status(401).json({
        error: 'Token expired',
        message: 'Access token has expired'
      });
    }
    
    if (error instanceof jwt.JsonWebTokenError) {
      return res.status(401).json({
        error: 'Invalid token',
        message: 'Invalid access token'
      });
    }

    return res.status(500).json({
      error: 'Authentication error',
      message: 'Failed to authenticate user'
    });
  }
};

// Registration endpoint
app.post('/auth/register', async (req: Request, res: Response) => {
  try {
    const { email, password } = req.body;

    // Validation
    if (!email || !password) {
      return res.status(400).json({
        error: 'Missing required fields',
        message: 'Email and password are required'
      });
    }

    if (password.length < 6) {
      return res.status(400).json({
        error: 'Password too short',
        message: 'Password must be at least 6 characters long'
      });
    }

    // Check if user already exists
    const existingUser = await prisma.user.findUnique({
      where: { email }
    });

    if (existingUser) {
      return res.status(409).json({
        error: 'User already exists',
        message: 'A user with this email already exists'
      });
    }

    // Hash password
    const hashedPassword = await bcrypt.hash(password, 12);

    // Create user
    const user = await prisma.user.create({
      data: {
        email,
        password: hashedPassword,
        role: 'user' // Default role
      },
      select: {
        id: true,
        email: true,
        role: true,
        createdAt: true
      }
    });

    res.status(201).json({
      message: 'User registered successfully',
      user
    });
  } catch (error) {
    console.error('Registration error:', error);
    res.status(500).json({
      error: 'Internal server error',
      message: 'Failed to register user'
    });
  }
});

// Login endpoint
app.post('/auth/login', async (req: Request, res: Response) => {
  try {
    const { email, password } = req.body;

    // Validation
    if (!email || !password) {
      return res.status(400).json({
        error: 'Missing credentials',
        message: 'Email and password are required'
      });
    }

    // Find user
    const user = await prisma.user.findUnique({
      where: { email }
    });

    if (!user) {
      return res.status(401).json({
        error: 'Invalid credentials',
        message: 'Invalid email or password'
      });
    }

    // Verify password
    const isPasswordValid = await bcrypt.compare(password, user.password);
    
    if (!isPasswordValid) {
      return res.status(401).json({
        error: 'Invalid credentials',
        message: 'Invalid email or password'
      });
    }

    // Generate access token (15 minutes)
    const accessToken = jwt.sign(
      {
        userId: user.id,
        email: user.email,
        role: user.role
      },
      JWT_SECRET,
      { expiresIn: '15m' }
    );

    // Generate refresh token (7 days)
    const refreshToken = jwt.sign(
      {
        userId: user.id,
        email: user.email
      },
      REFRESH_TOKEN_SECRET,
      { expiresIn: '7d' }
    );

    // Set refresh token in httpOnly cookie
    res.cookie('refreshToken', refreshToken, {
      httpOnly: true,
      secure: process.env.NODE_ENV === 'production', // Use secure cookies in production
      sameSite: 'strict',
      maxAge: 7 * 24 * 60 * 60 * 1000 // 7 days
    });

    res.json({
      message: 'Login successful',
      accessToken,
      user: {
        id: user.id,
        email: user.email,
        role: user.role
      }
    });
  } catch (error) {
    console.error('Login error:', error);
    res.status(500).json({
      error: 'Internal server error',
      message: 'Failed to login'
    });
  }
});

// Protected route - Get current user
app.get('/auth/me', authMiddleware, async (req: AuthenticatedRequest, res: Response) => {
  try {
    if (!req.user) {
      return res.status(401).json({
        error: 'Unauthorized',
        message: 'User not authenticated'
      });
    }

    const user = await prisma.user.findUnique({
      where: { id: req.user.userId },
      select: {
        id: true,
        email: true,
        role: true,
        createdAt: true
      }
    });

    if (!user) {
      return res.status(404).json({
        error: 'User not found',
        message: 'The authenticated user could not be found'
      });
    }

    res.json({
      user
    });
  } catch (error) {
    console.error('Get user error:', error);
    res.status(500).json({
      error: 'Internal server error',
      message: 'Failed to fetch user data'
    });
  }
});

// Error handling middleware
app.use((err: Error, req: Request, res: Response, next: NextFunction) => {
  console.error(err.stack);
  res.status(500).json({
    error: 'Internal server error',
    message: 'Something went wrong!'
  });
});

// 404 handler
app.use('*', (req: Request, res: Response) => {
  res.status(404).json({
    error: 'Not found',
    message: 'Route not found'
  });
});

app.listen(PORT, () => {
  console.log(`Server running on port ${PORT}`);
});

export default app;
```

## Security Reasoning for HttpOnly Cookies

**Why use httpOnly cookies instead of localStorage for refresh tokens?**

1. **XSS Protection**: 
   - localStorage is accessible via JavaScript, making it vulnerable to XSS attacks
   - httpOnly cookies are not accessible via JavaScript, preventing XSS attackers from stealing the token

2. **CSRF Protection**:
   - While both can be protected against CSRF, httpOnly cookies work better with SameSite cookies
   - They're automatically sent with requests but not readable by client-side scripts

3. **Token Isolation**:
   - httpOnly cookies are automatically included with HTTP requests to the same domain
   - This prevents accidental exposure in client-side code
   - They're not accessible via `document.cookie` or similar JavaScript APIs

4. **Security Best Practices**:
   - Modern security frameworks recommend using httpOnly cookies for sensitive tokens
   - The SameSite attribute provides additional protection against CSRF attacks
   - Secure flag ensures cookies are only sent over HTTPS in production

5. **Defense in Depth**:
   - Even if an XSS vulnerability exists, the refresh token won't be accessible to attackers
   - Access tokens can be short-lived (15 minutes) while refresh tokens last longer (7 days)
   - This layered approach minimizes damage from potential security breaches

This implementation provides a robust authentication system with proper error handling, security considerations, and follows RESTful API conventions.

```javascript
// Redis data structure for refresh tokens
const redisTokenStore = {
  // Key format: rt:<token_id>
  // Value: refresh_token_string
  // TTL: 7 days (604800 seconds)
  
  // Example Redis entries:
  // rt:123e4567-e89b-12d3-a456-426614174000 -> "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9..."
  // rt:uuid-v4-token -> "refresh_token_value"
  
  // Additional structures for enhanced security:
  // user:tokens:<user_id> -> Set of token IDs (for session management)
  // user:last_login:<user_id> -> Timestamp (for password change invalidation)
};

// Enhanced Redis storage with additional features
class TokenStore {
  constructor(redisClient) {
    this.client = redisClient;
  }
  
  async storeRefreshToken(userId, tokenId, refreshToken, ttlSeconds = 604800) {
    // Store the refresh token
    await this.client.setex(`rt:${tokenId}`, ttlSeconds, refreshToken);
    
    // Add to user's token set for session management
    await this.client.sadd(`user:tokens:${userId}`, tokenId);
    
    // Set TTL for user tokens set (optional cleanup)
    await this.client.expire(`user:tokens:${userId}`, ttlSeconds);
  }
  
  async validateRefreshToken(tokenId) {
    const token = await this.client.get(`rt:${tokenId}`);
    return token ? { valid: true, token } : { valid: false };
  }
  
  async invalidateRefreshToken(tokenId) {
    await this.client.del(`rt:${tokenId}`);
  }
  
  async invalidateUserTokens(userId) {
    // Get all tokens for user
    const tokenIds = await this.client.smembers(`user:tokens:${userId}`);
    
    // Invalidate all tokens
    if (tokenIds.length > 0) {
      await this.client.del(...tokenIds.map(id => `rt:${id}`));
      await this.client.del(`user:tokens:${userId}`);
    }
  }
  
  async getTokensForUser(userId) {
    return await this.client.smembers(`user:tokens:${userId}`);
  }
}

const tokenStore = new TokenStore(redisClient);
```

## 6. Token Blacklisting vs Token Rotation

```javascript
// Example of token blacklisting approach (less secure for refresh tokens)
class BlacklistTokenStore {
  constructor(redisClient) {
    this.client = redisClient;
  }
  
  // Add token to blacklist with TTL
  async addToBlacklist(tokenId, ttlSeconds = 3600) {
    await this.client.setex(`blacklisted:${tokenId}`, ttlSeconds, '1');
  }
  
  // Check if token is blacklisted
  async isBlacklisted(tokenId) {
    const result = await this.client.get(`blacklisted:${tokenId}`);
    return !!result;
  }
}

// Comparison function for different approaches
const compareTokenApproaches = () => {
  /*
  Token Rotation Approach:
  - Pros:
    * Better security (tokens are rotated)
    * No need for blacklists
    * Prevents replay attacks
    * Can invalidate all sessions with one command
    * Redis storage is more efficient
  - Cons:
    * Requires more Redis operations
    * Slightly more complex implementation
    * Token reuse detection needed
  
  Token Blacklisting Approach:
  - Pros:
    * Simpler to implement initially
    * Can immediately invalidate tokens
  - Cons:
    * Redis memory usage grows over time
    * Performance impact on token validation
    * Vulnerable to DoS attacks with many blacklisted tokens
    * Less secure against replay attacks
    * Requires cleanup of expired tokens
  
  When to use each:
  
  Use Token Rotation when:
  - Security is paramount
  - You want to invalidate sessions immediately (like password change)
  - You have a high volume of token usage
  - You can afford the extra Redis operations
  
  Use Token Blacklisting when:
  - Quick implementation is needed
  - You have low token volume
  - You're using access tokens for short-lived operations
  - You don't need immediate session invalidation
  
  Recommendation: Use token rotation for refresh tokens, as it's more secure and scalable.
  */
};

// Complete enhanced auth middleware with both approaches
const enhancedAuthMiddleware = {
  // Token rotation approach (recommended)
  rotateTokens: async (req, res, next) => {
    const refreshToken = req.cookies.refreshToken;
    
    if (!refreshToken) {
      return res.status(401).json({ error: 'No refresh token' });
    }
    
    try {
      const decoded = jwt.verify(refreshToken, process.env.REFRESH_TOKEN_SECRET);
      
      // Check if token exists in Redis (rotation)
      const storedToken = await redisClient.get(`rt:${decoded.jti}`);
      
      if (!storedToken || storedToken !== refreshToken) {
        return res.status(401).json({ error: 'Invalid refresh token' });
      }
      
      req.userId = decoded.userId;
      next();
    } catch (error) {
      return res.status(401).json({ error: 'Invalid token' });
    }
  },
  
  // Token blacklist approach (alternative)
  blacklistTokens: async (req, res, next) => {
    const refreshToken = req.cookies.refreshToken;
    
    if (!refreshToken) {
      return res.status(401).json({ error: 'No refresh token' });
    }
    
    try {
      const decoded = jwt.verify(refreshToken, process.env.REFRESH_TOKEN_SECRET);
      
      // Check blacklist
      const isBlacklisted = await redisClient.get(`blacklisted:${decoded.jti}`);
      
      if (isBlacklisted) {
        return res.status(401).json({ error: 'Token blacklisted' });
      }
      
      req.userId = decoded.userId;
      next();
    } catch (error) {
      return res.status(401).json({ error: 'Invalid token' });
    }
  }
};
```

```typescript
import React, { useState, useMemo, useCallback } from 'react';

interface Column<T> {
  key: keyof T;
  header: string;
  width?: number;
  sortable?: boolean;
  renderCell?: (row: T) => React.ReactNode;
}

interface DataTableProps<T> {
  columns: Column<T>[];
  data: T[];
  pageSizeOptions?: number[];
  defaultPageSize?: number;
  onRowSelect?: (selectedRows: T[]) => void;
  selectedRows?: T[];
  isControlled?: boolean;
  className?: string;
}

const DataTable = <T extends Record<string, any>>({
  columns,
  data,
  pageSizeOptions = [10, 25, 50],
  defaultPageSize = 10,
  onRowSelect,
  selectedRows: controlledSelectedRows = [],
  isControlled = false,
  className = '',
}: DataTableProps<T>) => {
  // State management
  const [currentPage, setCurrentPage] = useState(1);
  const [pageSize, setPageSize] = useState(defaultPageSize);
  const [sortConfig, setSortConfig] = useState<{ key: keyof T; direction: 'asc' | 'desc' } | null>(null);
  const [localSelectedRows, setLocalSelectedRows] = useState<T[]>([]);

  // Get selected rows based on controlled/uncontrolled mode
  const selectedRows = isControlled ? controlledSelectedRows : localSelectedRows;

  // Handle sorting
  const handleSort = useCallback((key: keyof T) => {
    let direction: 'asc' | 'desc' = 'asc';
    if (sortConfig && sortConfig.key === key) {
      if (sortConfig.direction === 'asc') direction = 'desc';
      else if (sortConfig.direction === 'desc') direction = null as any; // Reset to none
    }
    setSortConfig(direction ? { key, direction } : null);
  }, [sortConfig]);

  // Apply sorting and pagination
  const sortedAndPaginatedData = useMemo(() => {
    let result = [...data];

    // Apply sorting
    if (sortConfig) {
      result.sort((a, b) => {
        const aValue = a[sortConfig.key];
        const bValue = b[sortConfig.key];

        if (aValue < bValue) return sortConfig.direction === 'asc' ? -1 : 1;
        if (aValue > bValue) return sortConfig.direction === 'asc' ? 1 : -1;
        return 0;
      });
    }

    // Apply pagination
    const startIndex = (currentPage - 1) * pageSize;
    return result.slice(startIndex, startIndex + pageSize);
  }, [data, sortConfig, currentPage, pageSize]);

  // Handle row selection
  const handleRowSelect = useCallback((row: T) => {
    if (isControlled && onRowSelect) {
      const isSelected = selectedRows.some(r => r === row);
      const newSelectedRows = isSelected
        ? selectedRows.filter(r => r !== row)
        : [...selectedRows, row];
      onRowSelect(newSelectedRows);
    } else {
      setLocalSelectedRows(prev => {
        const isSelected = prev.some(r => r === row);
        return isSelected
          ? prev.filter(r => r !== row)
          : [...prev, row];
      });
    }
  }, [selectedRows, isControlled, onRowSelect]);

  // Handle select all
  const handleSelectAll = useCallback(() => {
    if (isControlled && onRowSelect) {
      if (selectedRows.length === data.length) {
        onRowSelect([]);
      } else {
        onRowSelect([...data]);
      }
    } else {
      setLocalSelectedRows(prev => {
        if (prev.length === data.length) return [];
        return [...data];
      });
    }
  }, [data, selectedRows, isControlled, onRowSelect]);

  // Page size change
  const handlePageSizeChange = useCallback((newPageSize: number) => {
    setPageSize(newPageSize);
    setCurrentPage(1); // Reset to first page when changing page size
  }, []);

  // Calculate pagination info
  const totalPages = Math.ceil(data.length / pageSize);
  const isSelectAllChecked = data.length > 0 && selectedRows.length === data.length;

  return (
    <div className={`data-table ${className}`}>
      {/* Table */}
      <table>
        <thead>
          <tr>
            <th>
              <input
                type="checkbox"
                checked={isSelectAllChecked}
                onChange={handleSelectAll}
              />
            </th>
            {columns.map((column) => (
              <th
                key={String(column.key)}
                style={{ width: column.width ? `${column.width}px` : 'auto' }}
                onClick={() => column.sortable && handleSort(column.key)}
              >
                {column.header}
                {column.sortable && sortConfig?.key === column.key && (
                  <span>
                    {sortConfig.direction === 'asc' ? ' ↑' : ' ↓'}
                  </span>
                )}
              </th>
            ))}
          </tr>
        </thead>
        <tbody>
          {sortedAndPaginatedData.map((row, index) => (
            <tr key={index}>
              <td>
                <input
                  type="checkbox"
                  checked={selectedRows.some(r => r === row)}
                  onChange={() => handleRowSelect(row)}
                />
              </td>
              {columns.map((column) => (
                <td key={String(column.key)}>
                  {column.renderCell ? column.renderCell(row) : row[column.key]}
                </td>
              ))}
            </tr>
          ))}
        </tbody>
      </table>

      {/* Pagination */}
      <div className="pagination">
        <span>Rows per page: </span>
        <select value={pageSize} onChange={(e) => handlePageSizeChange(Number(e.target.value))}>
          {pageSizeOptions.map((size) => (
            <option key={size} value={size}>
              {size}
            </option>
          ))}
        </select>
        <span>
          {Math.min((currentPage - 1) * pageSize + 1, data.length)} -{' '}
          {Math.min(currentPage * pageSize, data.length)} of {data.length}
        </span>
        <button
          onClick={() => setCurrentPage(prev => Math.max(1, prev - 1))}
          disabled={currentPage === 1}
        >
          Previous
        </button>
        <button
          onClick={() => setCurrentPage(prev => Math.min(totalPages, prev + 1))}
          disabled={currentPage === totalPages}
        >
          Next
        </button>
      </div>
    </div>
  );
};

export default DataTable;
```

```jsx
const useSearch = (initialQuery = '') => {
  const [query, setQuery] = useState(initialQuery);
  const [results, setResults] = useState([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);
  const [cachedResults, setCachedResults] = useState(new Map());
  
  // Refs for tracking in-flight requests and timeouts
  const timeoutRef = useRef(null);
  const abortControllerRef = useRef(null);
  const cacheTimeoutRef = useRef(null);

  // Cancel in-flight request
  const cancelRequest = useCallback(() => {
    if (abortControllerRef.current) {
      abortControllerRef.current.abort();
    }
  }, []);

  // Clear cache after 5 minutes
  const clearCache = useCallback(() => {
    if (cacheTimeoutRef.current) {
      clearTimeout(cacheTimeoutRef.current);
    }
    cacheTimeoutRef.current = setTimeout(() => {
      setCachedResults(prev => {
        const newMap = new Map(prev);
        newMap.forEach((value, key) => {
          if (Date.now() - value.timestamp > 5 * 60 * 1000) {
            newMap.delete(key);
          }
        });
        return newMap;
      });
    }, 5 * 60 * 1000);
  }, []);

  // Fetch search results
  const fetchResults = useCallback(async (searchQuery) => {
    // Cancel previous request
    cancelRequest();
    
    // Create new abort controller for this request
    const controller = new AbortController();
    abortControllerRef.current = controller;
    
    // Check cache first
    if (cachedResults.has(searchQuery)) {
      const cached = cachedResults.get(searchQuery);
      if (Date.now() - cached.timestamp < 5 * 60 * 1000) {
        setResults(cached.data);
        setError(null);
        setLoading(false);
        return;
      }
    }

    // Set loading state
    setLoading(true);
    setError(null);

    try {
      const response = await fetch(`/api/search?q=${encodeURIComponent(searchQuery)}`, {
        signal: controller.signal
      });

      if (!response.ok) {
        throw new Error(`HTTP error! status: ${response.status}`);
      }

      const data = await response.json();
      
      // Cache results
      setCachedResults(prev => {
        const newMap = new Map(prev);
        newMap.set(searchQuery, {
          data,
          timestamp: Date.now()
        });
        return newMap;
      });
      
      setResults(data);
    } catch (err) {
      if (err.name !== 'AbortError') {
        setError(err.message);
        setResults([]);
      }
    } finally {
      setLoading(false);
    }
  }, [cachedResults, cancelRequest]);

  // Handle query changes with debounce
  useEffect(() => {
    // Clear previous timeout
    if (timeoutRef.current) {
      clearTimeout(timeoutRef.current);
    }

    // Set new timeout
    timeoutRef.current = setTimeout(() => {
      if (query.trim() !== '') {
        fetchResults(query);
      } else {
        setResults([]);
        setError(null);
        setLoading(false);
      }
    }, 300);

    return () => {
      if (timeoutRef.current) {
        clearTimeout(timeoutRef.current);
      }
    };
  }, [query, fetchResults]);

  // Clear cache on unmount
  useEffect(() => {
    clearCache();
    return () => {
      if (cacheTimeoutRef.current) {
        clearTimeout(cacheTimeoutRef.current);
      }
    };
  }, [clearCache]);

  return {
    query,
    setQuery,
    results,
    loading,
    error,
    retry: () => fetchResults(query)
  };
};

// SearchPage component
const SearchPage = () => {

  // Sync URL with search query
  useEffect(() => {
    const url = new URL(window.location);
    if (query) {
      url.searchParams.set('q', query);
    } else {
      url.searchParams.delete('q');
    }
    window.history.replaceState({}, '', url);
  }, [query]);

  // Initialize from URL
  useEffect(() => {
    const urlParams = new URLSearchParams(window.location.search);
    const initialQuery = urlParams.get('q') || '';
    if (initialQuery) {
      setQuery(initialQuery);
    }
  }, [setQuery]);

  return (
    <div className="search-page">
      <div className="search-container">
        <input
          type="text"
          value={query}
          onChange={(e) => setQuery(e.target.value)}
          placeholder="Search..."
          className="search-input"
        />
        
        {loading && (
          <div className="spinner">Loading...</div>
        )}
        
        {error && (
          <div className="error">
            <p>{error}</p>
            <button onClick={retry}>Retry</button>
          </div>
        )}
        
        {!loading && !error && results.length > 0 && (
          <ul className="results-list">
            {results.map((result, index) => (
              <li key={index}>{result}</li>
            ))}
          </ul>
        )}
        
        {!loading && !error && results.length === 0 && query && (
          <p className="no-results">No results found</p>
        )}
      </div>
    </div>
  );
};
```

## Why is cancelling in-flight requests important?

Cancelling in-flight requests prevents several critical bugs:

1. **Race Condition Bug**: Without cancellation, if a user types "app" then "apple", two requests might be sent:
   - First request for "app" 
   - Second request for "apple"
   
   If the "app" request completes after "apple", the results for "app" could overwrite the correct "apple" results, showing outdated data.

2. **Memory Leaks**: Accumulated abandoned requests can cause memory leaks and unnecessary network usage.

3. **UI Inconsistency**: Users might see results from different queries mixed together, creating a confusing experience.

4. **Resource Waste**: Unnecessary API calls consume bandwidth and server resources.

5. **User Experience Degradation**: The interface might flicker between different search results as requests complete out of order.

By cancelling previous requests, we ensure that only the most recent search query's results are displayed, maintaining a consistent and predictable user experience.

```javascript
const WebSocket = require('ws');
const http = require('http');
const express = require('express');

// Server state
const connections = new Map(); // connectionId -> { ws, userId, documentId }
const documents = new Map();   // documentId -> { users: Set<connectionId>, content: string }

// Message types
const MESSAGE_TYPES = {
  JOIN_DOCUMENT: 'JOIN_DOCUMENT',
  TEXT_CHANGE: 'TEXT_CHANGE',
  CURSOR_MOVE: 'CURSOR_MOVE',
  USER_LEFT: 'USER_LEFT'
};

// Server setup
const app = express();
const server = http.createServer(app);
const wss = new WebSocket.Server({ server });

// Generate unique IDs
function generateId() {
  return Math.random().toString(36).substr(2, 9);
}

// Handle new WebSocket connections
wss.on('connection', (ws) => {
  const connectionId = generateId();
  console.log(`New connection: ${connectionId}`);
  
  // Store connection info
  connections.set(connectionId, { ws, userId: null, documentId: null });
  
  // Handle incoming messages
  ws.on('message', (message) => {
    try {
      const data = JSON.parse(message);
      handleIncomingMessage(connectionId, data);
    } catch (error) {
      console.error('Error parsing message:', error);
      ws.send(JSON.stringify({ type: 'ERROR', message: 'Invalid message format' }));
    }
  });
  
  // Handle connection close
  ws.on('close', () => {
    console.log(`Connection closed: ${connectionId}`);
    const connection = connections.get(connectionId);
    if (connection && connection.documentId) {
      leaveDocument(connectionId, connection.documentId);
    }
    connections.delete(connectionId);
  });
});

// Handle incoming messages
function handleIncomingMessage(connectionId, data) {
  const connection = connections.get(connectionId);
  if (!connection) return;
  
  switch (data.type) {
    case MESSAGE_TYPES.JOIN_DOCUMENT:
      joinDocument(connectionId, data);
      break;
    case MESSAGE_TYPES.TEXT_CHANGE:
      handleTextChange(connectionId, data);
      break;
    case MESSAGE_TYPES.CURSOR_MOVE:
      handleCursorMove(connectionId, data);
      break;
    default:
      connection.ws.send(JSON.stringify({ 
        type: 'ERROR', 
        message: `Unknown message type: ${data.type}` 
      }));
  }
}

// Join a document
function joinDocument(connectionId, data) {
  const { userId, documentId, content = '' } = data;
  const connection = connections.get(connectionId);
  
  if (!connection) return;
  
  // Update connection info
  connection.userId = userId;
  connection.documentId = documentId;
  
  // Initialize document if it doesn't exist
  if (!documents.has(documentId)) {
    documents.set(documentId, { users: new Set(), content });
  }
  
  // Add user to document
  const doc = documents.get(documentId);
  doc.users.add(connectionId);
  
  // Send current document state to joining user
  connection.ws.send(JSON.stringify({
    type: MESSAGE_TYPES.JOIN_DOCUMENT,
    documentId,
    content: doc.content,
    users: Array.from(doc.users).map(id => {
      const user = connections.get(id);
      return user ? { userId: user.userId, cursor: user.cursor } : null;
    }).filter(Boolean)
  }));
  
  // Broadcast to other users in the document
  broadcastToDocument(documentId, connectionId, {
    type: MESSAGE_TYPES.USER_LEFT,
    userId,
    timestamp: Date.now()
  });
}

// Handle text changes
function handleTextChange(connectionId, data) {
  const { documentId, content, position, length } = data;
  const connection = connections.get(connectionId);
  
  if (!connection || !documentId) return;
  
  // Update document content
  const doc = documents.get(documentId);
  if (doc) {
    // Simple text replacement (in a real system, you'd use OT/CRDT)
    const newContent = doc.content.substring(0, position) + 
                      content + 
                      doc.content.substring(position + length);
    doc.content = newContent;
    
    // Broadcast change to all other users
    broadcastToDocument(documentId, connectionId, {
      type: MESSAGE_TYPES.TEXT_CHANGE,
      userId: connection.userId,
      content,
      position,
      length,
      timestamp: Date.now()
    });
  }
}

// Handle cursor movement
function handleCursorMove(connectionId, data) {
  const { documentId, cursor } = data;
  const connection = connections.get(connectionId);
  
  if (!connection || !documentId) return;
  
  // Store cursor position in connection
  connection.cursor = cursor;
  
  // Broadcast to other users
  broadcastToDocument(documentId, connectionId, {
    type: MESSAGE_TYPES.CURSOR_MOVE,
    userId: connection.userId,
    cursor,
    timestamp: Date.now()
  });
}

// Leave document
function leaveDocument(connectionId, documentId) {
  const doc = documents.get(documentId);
  if (doc) {
    doc.users.delete(connectionId);
    
    // If no users left, remove document
    if (doc.users.size === 0) {
      documents.delete(documentId);
    } else {
      // Notify remaining users
      broadcastToDocument(documentId, connectionId, {
        type: MESSAGE_TYPES.USER_LEFT,
        userId: connections.get(connectionId)?.userId,
        timestamp: Date.now()
      });
    }
  }
}

// Broadcast to all users in a document except sender
function broadcastToDocument(documentId, senderConnectionId, message) {
  const doc = documents.get(documentId);
  if (!doc) return;
  
  const messageString = JSON.stringify(message);
  
  doc.users.forEach(connectionId => {
    if (connectionId !== senderConnectionId) {
      const connection = connections.get(connectionId);
      if (connection && connection.ws.readyState === WebSocket.OPEN) {
        connection.ws.send(messageString);
      }
    }
  });
}

// Start server
const PORT = process.env.PORT || 3000;
server.listen(PORT, () => {
  console.log(`Server running on port ${PORT}`);
});

module.exports = { wss, connections, documents };
```

### Explanation of Key Components

1. **Connection Handling**
   - Each WebSocket connection is tracked with a unique ID
   - Connections are associated with user IDs and document IDs
   - Connection metadata stored in `connections` Map

2. **Message Protocol**
   - JOIN_DOCUMENT: User joins a document with userId, documentId, and optional content
   - TEXT_CHANGE: Content changes with position/length info for precise editing
   - CURSOR_MOVE: Real-time cursor position updates
   - USER_LEFT: Notification when users disconnect

3. **Broadcasting**
   - `broadcastToDocument()` sends messages to all users in a document except the sender
   - Uses Set of connection IDs to efficiently track users in each document
   - Filters out sender connection ID before broadcasting

4. **Reconnection**
   - When user reconnects, they rejoin with same userId and documentId
   - Server maintains document state and sends current content
   - No special reconnection logic needed beyond normal join process

5. **Concurrency Issues**
   - **Problem**: Two users editing same position simultaneously causes conflicts
   - **Solution approaches**:
     - **Operational Transformation (OT)**: Transform operations based on order
     - **Conflict-free Replicated Data Types (CRDTs)**: Use mathematical properties to resolve conflicts
     - **Last Write Wins (LWW)**: Accept latest change, but lose intermediate edits
     - **Version vectors**: Track version numbers and merge histories

This implementation provides a solid foundation that can be extended with proper OT/CRDT libraries for robust collaborative editing.

```javascript
const DataLoader = require('dataloader');

// Create a DataLoader for users
const createUserLoader = (db) => {
  return new DataLoader(async (userIds) => {
    const result = await db.query(
      'SELECT * FROM users WHERE id = ANY($1)',
      [userIds]

    // Map results back to the original order
    const userMap = {};
```

## 5. When to Choose GraphQL vs REST

### Choose GraphQL when:

1. **Complex data requirements**: Clients need different combinations of fields
2. **Mobile apps**: Need to minimize bandwidth and optimize for sparse data
3. **Multiple clients**: Different UIs (web, mobile, IoT) need different data shapes
4. **Frequent API changes**: You want to evolve your API without breaking existing clients
5. **Nested relationships**: Complex object graphs with many-to-many relationships
6. **Real-time updates**: Need subscriptions for live data

### Don't choose GraphQL when:

1. **Simple APIs**: CRUD operations on straightforward resources
2. **Performance-critical**: When you need maximum control over HTTP caching and optimization
3. **Existing REST ecosystem**: You're already invested in REST patterns and tooling
4. **Small teams**: The complexity overhead isn't justified for simple use cases
5. **Static data requirements**: Clients always need the same set of fields
6. **Caching requirements**: You need fine-grained HTTP caching control
7. **Security concerns**: Need specific HTTP headers, authentication methods, or request/response formats that GraphQL doesn't easily support

### Example where REST would be better:
```javascript
// REST for simple CRUD operations
GET /api/users/123/orders?status=pending
GET /api/products?category=electronics&sort=price
```

This is simpler and more cacheable than equivalent GraphQL queries, especially when you don't need the flexibility of field selection.

## 1. Transport Mechanism Selection

### WebSocket vs SSE vs Polling vs Push Notifications

| Mechanism | Pros | Cons | Recommendation |
|-----------|------|------|----------------|
| **WebSocket** | Bidirectional, low latency, real-time | Complex implementation, connection management | **Best for primary transport** |
| **Server-Sent Events (SSE)** | Simple, one-way, persistent connections | Unidirectional, browser support limitations | **Good fallback** |
| **Polling** | Simple to implement, widely supported | High latency, resource intensive | **Not recommended** |
| **Push Notifications** | Works offline, mobile-first | Platform-specific, requires registration | **Complementary for mobile apps** |

**Recommendation: WebSocket as primary transport with SSE fallback**

```javascript
// Client-side implementation example
class NotificationService {
    constructor() {
        this.ws = null;
        this.reconnectAttempts = 0;
        this.maxReconnectAttempts = 5;

    connect() {
        // Try WebSocket first
        if ('WebSocket' in window) {
            this.ws = new WebSocket('wss://api.example.com/notifications');
            this.setupWebSocketHandlers();

            // Fallback to SSE
            this.setupSSE();

            this.handleNotification(notification);
```

## 3. Offline Notification Handling

### Implementation Approach: Notification Queue with Persistence

```javascript
// Server-side notification handler
class NotificationManager {
    constructor() {
        this.activeUsers = new Map(); // userId -> WebSocket connection
        this.offlineNotifications = new Map(); // userId -> [notifications]
    }
    
    async handleUserOnline(userId, ws) {
        // Store active connection
        this.activeUsers.set(userId, ws);
        
        // Deliver queued notifications
        if (this.offlineNotifications.has(userId)) {
            const queued = this.offlineNotifications.get(userId);
            for (const notification of queued) {
                await this.deliverToUser(userId, notification, ws);
            }
            this.offlineNotifications.delete(userId);
        }
    }
    
    async handleUserOffline(userId) {
        // Remove from active users
        this.activeUsers.delete(userId);
        
        // Optionally store recent notifications for quick retrieval
        const recentNotifications = await this.getRecentUnreadNotifications(userId, 10);
        if (recentNotifications.length > 0) {
            this.offlineNotifications.set(userId, recentNotifications);
        }
    }
    
    async deliverToUser(userId, notification, ws = null) {
        // Check if user is online
        const activeWs = this.activeUsers.get(userId);
        
        if (activeWs) {
            // Send directly via WebSocket
            try {
                await this.sendWebSocketMessage(activeWs, notification);
                await this.markAsDelivered(notification.id);
            } catch (error) {
                console.error('Failed to deliver notification:', error);
                // Queue for later delivery or store in offline queue
                this.queueForLaterDelivery(userId, notification);
            }
        } else {
            // Store for when user comes back online
            this.queueForLaterDelivery(userId, notification);

        }
    }
```

```javascript
    concurrency: 10 // Process 10 jobs concurrently
});

// Schedule deadline checks (cron-like approach)
class DeadlineScheduler {
    constructor() {
        this.schedulePeriodicChecks();
    }
    
    schedulePeriodicChecks() {
        // Run every hour
        setInterval(async () => {
            await this.checkAllTasks();
        }, 60 * 60 * 1000);
    }
    
    async checkAllTasks() {
        const tasks = await Task.find({

                userId: task.assignedTo
            }, {

            });
        }
    }
```
