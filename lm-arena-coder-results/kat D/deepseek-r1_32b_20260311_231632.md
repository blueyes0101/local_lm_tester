          setResults(data);
          setError(null);
          setLoading(false);
          // Store result in cache
          SEARCH_CACHE.set(q, { results: data, timestamp: Date.now(), controller });
        }
      })
      .catch(err => {
        if (err.name === 'AbortError') return;
        setError('Failed to fetch results. Please try again.');
        setLoading(false);
      });
  }, [cacheKey]);

  useEffect(() => {
    const timeout = setTimeout(() => {
      debounceSearch(query);
    }, 300);

    return () => clearTimeout(timeout);
  }, [query, debounceSearch]);

    debounceSearch(query);
  };

  return {
    results,
    loading,
    error,
    retry

Cancelling in-flight requests is important to prevent several issues:
