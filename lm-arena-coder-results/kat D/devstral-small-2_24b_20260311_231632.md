import { z } from 'zod';

// Database mock
const workspaces: Record<number, { id: number; name: string; members: number[] }> = {
  1: { id: 1, name: 'Workspace 1', members: [1, 2, 3] },
  2: { id: 2, name: 'Workspace 2', members: [4, 5] },
};

const tasks: Record<number, { id: number; workspaceId: number; title: string; status: string; assigneeId: number | null; dueDate: string | null }> = {
  1: { id: 1, workspaceId: 1, title: 'Task 1', status: 'todo', assigneeId: 1, dueDate: '2023-12-31' },
  2: { id: 2, workspaceId: 1, title: 'Task 2', status: 'in_progress', assigneeId: 2, dueDate: null },
  3: { id: 3, workspaceId: 2, title: 'Task 3', status: 'todo', assigneeId: 4, dueDate: '2024-01-15' },
};

const subtasks: Record<number, { id: number; taskId: number; status: string }[]> = {
  1: [{ id: 101, taskId: 1, status: 'todo' }], // Task 1 has an open subtask
  2: [], // Task 2 has no subtasks
  3: [{ id: 102, taskId: 3, status: 'done' }], // Task 3 has a done subtask
};

// Validation schemas
const taskUpdateSchema = z.object({
  title: z.string().min(1).max(200).optional(),
  status: z.union([
    z.literal('todo'),
    z.literal('in_progress'),
    z.literal('done'),
    z.literal('cancelled')
  ]).optional(),
  assignee_id: z.union([z.number().int().positive(), z.null()]).optional(),
  due_date: z.union([z.string().datetime(), z.null()]).optional(),
}).partial();

const paramsSchema = z.object({
  workspaceId: z.coerce.number().int().positive(),
  taskId: z.coerce.number().int().positive(),
});

const app = express();
app.use(express.json());

// Error handling middleware
app.use((err: Error, req: Request, res: Response, next: NextFunction) => {
  if (err instanceof z.ZodError) {
    return res.status(400).json({
      error: 'Validation failed',
      details: err.errors.map(e => ({
        field: e.path.join('.'),
        message: e.message,
      })),
    });
  }
  next();
});

// PATCH endpoint
app.patch(
  '/workspaces/:workspaceId/tasks/:taskId',
  async (req: Request, res: Response) => {
    try {
      // Validate params
      const params = paramsSchema.parse(req.params);

      // Validate body
      const updateData = taskUpdateSchema.parse(req.body);

      // Check if workspace exists
      const workspace = workspaces[params.workspaceId];
      if (!workspace) {
        return res.status(404).json({ error: 'Workspace not found' });
      }

      // Check if task exists and belongs to workspace
      const task = tasks[params.taskId];
      if (!task || task.workspaceId !== params.workspaceId) {
        return res.status(404).json({ error: 'Task not found' });
      }

      // Validate assignee_id if provided
      if (updateData.assignee_id !== undefined) {
        const assigneeId = updateData.assignee_id;
        if (assigneeId !== null && !workspace.members.includes(assigneeId)) {
          return res.status(400).json({
            error: 'Validation failed',
            details: [{ field: 'assignee_id', message: 'Assignee is not a member of the workspace' }],
          });
        }
      }

      // Check for cancelled status with open subtasks
      if (updateData.status === 'cancelled') {
        const openSubtasks = subtasks[params.taskId]?.filter(st => st.status !== 'done');
        if (openSubtasks && openSubtasks.length > 0) {
          return res.status(409).json({
            error: 'Cannot cancel task with open subtasks',
            details: [{ field: 'status', message: 'Task has open subtasks' }],
          });
        }
      }

      // Update task
      const updatedTask = {
        ...task,
        ...updateData,
        // Convert snake_case to camelCase for consistency
        assigneeId: updateData.assignee_id,
        dueDate: updateData.due_date,
      };

      tasks[params.taskId] = updatedTask;

      res.json(updatedTask);
    } catch (error) {
      if (error instanceof z.ZodError) {
        return res.status(400).json({
          error: 'Validation failed',
          details: error.errors.map(e => ({
            field: e.path.join('.'),
            message: e.message,
          })),
        });
      }
      throw error;
    }
  }
);

const PORT = 3000;
app.listen(PORT, () => {
  console.log(`Server running on port ${PORT}`);
});

    if not refresh_token:

### 3. Redis Data Structure and TTL Strategy
```
Key format: refresh:{user_id}:{token_hash}
Value: "1" (just a marker that the token exists)
TTL: Same as refresh token expiry (e.g., 7 days)

Additional keys:
- used:{token_hash}: "1" (when token is used for refresh)
  TTL: 1 hour (to prevent replay attacks)
- user_sessions:{user_id}: Set of all active refresh tokens for the user
  TTL: Not set (managed manually when tokens are added/removed)
```

### 4. Token Reuse Detection Implementation
The implementation is already shown in the refresh endpoint:
1. When a refresh token is used, we store it in Redis with key `used:{token_hash}`
2. Before processing any refresh request, we check if the token has been marked as used
3. If it has, we reject the request with a 401 error

### 6. Token Blacklisting vs Token Rotation

**Token Blacklisting:**
- Approach: Maintain a list of invalidated tokens
- Implementation: Store revoked tokens in Redis with infinite TTL
- Use cases:
  - When you need to immediately invalidate tokens (e.g., password change)
  - When you have a small number of active tokens
  - When you need to support token revocation for specific tokens

**Token Rotation:**
- Approach: Issue new tokens and invalidate old ones during refresh
- Implementation: As shown in the refresh endpoint above
- Use cases:
  - When you want to limit token lifetime without maintaining a blacklist
  - When you have many active tokens and want to avoid memory issues
  - When you want to automatically expire tokens after some time

**Key Differences:**
1. Blacklisting requires maintaining a list of all invalid tokens
2. Rotation automatically expires old tokens when new ones are issued
3. Blacklisting can be immediate, rotation requires token expiry
4. Rotation is more scalable for large numbers of tokens

**When to Use Each:**
- Use blacklisting when you need immediate revocation (like password changes)
- Use rotation when you want automatic token expiry without maintaining a blacklist
- In practice, many systems use both: rotation for normal token expiry and blacklisting for immediate revocation cases

The implementation above uses rotation as the primary mechanism but includes blacklisting functionality for password changes.

### 4. Why Cancel In-Flight Requests?

**Importance of Canceling In-Flight Requests:**

1. **Prevents Race Conditions:**
   - Without cancellation, if a user types quickly, multiple requests might be made
   - The last request might complete after earlier ones, showing stale results
   - Example: User types "app" → "apple" → "applesauce" - without cancellation, the "app" request might complete last

2. **Avoids Unnecessary Network Traffic:**
   - Each request consumes bandwidth and server resources
   - Canceling irrelevant requests reduces unnecessary load

3. **Prevents State Corruption:**
   - If a request completes after the component has unmounted or the query changed
   - The response might update state with irrelevant data

4. **Improves Performance:**
   - The browser doesn't need to process responses for abandoned requests
   - Reduces memory usage from pending requests

**Bug It Prevents:**
The main bug prevented is showing stale results. For example:
1. User searches "react"
2. Results start loading
3. User searches "react hooks" before the first request completes
4. The "react" request finishes after "react hooks" request
5. Without cancellation, the "react" results would overwrite the "react hooks" results

With cancellation, the "react" request is aborted when the new search starts, ensuring only the most recent results are shown.

**Note:** In a production implementation, you would need to track the AbortController instances to properly cancel requests. The example above shows the concept but would need enhancement for full functionality.

# Collaborative Document Editor Server Implementation

Here's a complete Node.js server implementation using the `ws` library for a simplified collaborative document editor:

```javascript
const WebSocket = require('ws');
const crypto = require('crypto');

// Data structures to track connections and documents
const documentConnections = new Map(); // documentId -> Set<WebSocket>
const userSessions = new Map(); // sessionId -> { documentId, socket, username, cursorPosition }
const documentStates = new Map(); // documentId -> documentContent

// Message types
const MESSAGE_TYPES = {
  JOIN_DOCUMENT: 'JOIN_DOCUMENT',
  TEXT_CHANGE: 'TEXT_CHANGE',
  CURSOR_MOVE: 'CURSOR_MOVE',
  USER_LEFT: 'USER_LEFT',
  DOCUMENT_STATE: 'DOCUMENT_STATE',
  CURSOR_UPDATE: 'CURSOR_UPDATE'
};

// Create WebSocket server
const wss = new WebSocket.Server({ port: 8080 });

wss.on('connection', (ws) => {
  console.log('New client connected');

  // Generate a unique session ID for this connection
  const sessionId = crypto.randomBytes(8).toString('hex');

  ws.on('message', (message) => {
    try {
      const data = JSON.parse(message);

      switch (data.type) {
        case MESSAGE_TYPES.JOIN_DOCUMENT:
          handleJoinDocument(ws, sessionId, data);
          break;
        case MESSAGE_TYPES.TEXT_CHANGE:
          handleTextChange(ws, sessionId, data);
          break;
        case MESSAGE_TYPES.CURSOR_MOVE:
          handleCursorMove(ws, sessionId, data);
          break;
        default:
          console.log('Unknown message type:', data.type);
      }
    } catch (err) {
      console.error('Error processing message:', err);
    }
  });

  ws.on('close', () => {
    handleDisconnect(sessionId);
  });
});

function handleJoinDocument(ws, sessionId, data) {
  const { documentId, username } = data;

  // Remove any existing session for this socket
  userSessions.forEach((session, id) => {
    if (session.socket === ws) {
      userSessions.delete(id);
    }
  });

  // Add to document connections
  if (!documentConnections.has(documentId)) {
    documentConnections.set(documentId, new Set());
  }
  documentConnections.get(documentId).add(ws);

  // Store user session
  userSessions.set(sessionId, {
    documentId,
    socket: ws,
    username,
    cursorPosition: 0
  });

  // Send current document state to new user
  const currentState = documentStates.get(documentId) || '';
  ws.send(JSON.stringify({
    type: MESSAGE_TYPES.DOCUMENT_STATE,
    documentId,
    content: currentState,
    users: Array.from(documentConnections.get(documentId) || [])
      .map(socket => {
        const session = Array.from(userSessions.values()).find(s => s.socket === socket);
        return session ? { username: session.username, cursorPosition: session.cursorPosition } : null;
      })
      .filter(Boolean)
  }));

  // Broadcast new user to others
  broadcastToDocument(documentId, {
    type: MESSAGE_TYPES.CURSOR_UPDATE,
    username,
    cursorPosition: 0,
    sessionId
  }, ws);
}

function handleTextChange(ws, sessionId, data) {
  const { documentId, content } = data;
  const session = userSessions.get(sessionId);

  if (!session || session.documentId !== documentId) {
    return; // Ignore if not in this document
  }

  // Update document state
  documentStates.set(documentId, content);

  // Broadcast change to all other users in document
  broadcastToDocument(documentId, {
    type: MESSAGE_TYPES.TEXT_CHANGE,
    documentId,
    content
  }, ws);
}

function handleCursorMove(ws, sessionId, data) {
  const { documentId, cursorPosition } = data;
  const session = userSessions.get(sessionId);

  if (!session || session.documentId !== documentId) {
    return; // Ignore if not in this document
  }

  // Update cursor position
  session.cursorPosition = cursorPosition;

  // Broadcast cursor move to all other users in document
  broadcastToDocument(documentId, {
    type: MESSAGE_TYPES.CURSOR_UPDATE,
    username: session.username,
    cursorPosition,
    sessionId
  }, ws);
}

function handleDisconnect(sessionId) {
  const session = userSessions.get(sessionId);
  if (!session) return;

  // Remove from document connections
  const documentId = session.documentId;
  const connections = documentConnections.get(documentId);
  if (connections) {
    connections.delete(session.socket);
    if (connections.size === 0) {
      documentConnections.delete(documentId);
    }
  }

  // Broadcast user left
  broadcastToDocument(documentId, {
    type: MESSAGE_TYPES.USER_LEFT,
    username: session.username,
    sessionId
  }, session.socket);

  // Clean up session
  userSessions.delete(sessionId);
}

function broadcastToDocument(documentId, message, excludeSocket) {
  const connections = documentConnections.get(documentId);
  if (!connections) return;

  connections.forEach(socket => {
    if (socket !== excludeSocket && socket.readyState === WebSocket.OPEN) {
      socket.send(JSON.stringify(message));
    }
  });
}

// Handle reconnection by allowing the same session to reconnect
// The client would need to send the same sessionId when rejoining
// The server would then update the socket reference in userSessions
```

## Key Components Explained:

### 1. Connection Handling
- Each WebSocket connection gets a unique session ID
- We maintain three main data structures:
  - `documentConnections`: Maps document IDs to sets of connected WebSockets
  - `userSessions`: Maps session IDs to user information (document, socket, username, cursor)
  - `documentStates`: Stores the current content of each document

### 2. Message Protocol
The defined message types are:
- `JOIN_DOCUMENT`: `{ type, documentId, username }`
- `TEXT_CHANGE`: `{ type, documentId, content }`
- `CURSOR_MOVE`: `{ type, documentId, cursorPosition }`
- `USER_LEFT`: `{ type, username, sessionId }`
- `DOCUMENT_STATE`: `{ type, documentId, content, users }` (server to client)
- `CURSOR_UPDATE`: `{ type, username, cursorPosition, sessionId }` (server to client)

### 3. Broadcasting
The `broadcastToDocument` function:
- Takes a document ID, message, and optional socket to exclude
- Gets all connections for that document
- Sends the message to each connection except the excluded one
- Checks socket ready state before sending

### 4. Reconnection
- The client would need to maintain its session ID
- When reconnecting, it would send the same session ID in a JOIN_DOCUMENT message
- The server would update the socket reference in `userSessions` for that session ID
- The user would receive the current document state and see other users' cursors

### 5. Simultaneous Edits Problem
**Problem**: When two users edit the same position, the last change received by the server would overwrite the other, causing data loss.

**Solution Approaches**:
1. **Operational Transform (OT)**: Transform operations to account for concurrent changes
2. **Conflict-free Replicated Data Types (CRDTs)**: Data structures that guarantee convergence
3. **Last Write Wins**: Simple but loses data (not recommended)
4. **Client-side Merge**: Have clients merge changes before sending to server
5. **Document Locking**: Only allow one user to edit at a time (poor UX)

For a production system, I'd recommend implementing OT or CRDTs, but for this simplified version, we're just showing the basic infrastructure.

This implementation provides the core functionality while leaving room for more sophisticated conflict resolution to be added later.

## 5. When to Choose GraphQL vs REST

### Choose GraphQL when:

1. **Multiple frontends with different data needs**: GraphQL allows each client to request exactly what it needs
2. **Frequent API changes**: GraphQL schema can evolve without versioning
3. **Complex data requirements**: When you need to fetch related data in a single request
4. **Real-time updates**: With subscriptions, GraphQL can push updates to clients
5. **Avoiding over-fetching/under-fetching**: Clients specify exactly what they need

### Choose REST when:

1. **Simple APIs**: For straightforward CRUD operations
2. **Caching needs**: REST has better built-in caching support (HTTP caching)
3. **Performance-critical applications**: REST can be more efficient for simple requests
4. **Team familiarity**: If your team is more experienced with REST
5. **Public APIs**: When you want to clearly define endpoints and versions

### Key considerations:
- GraphQL adds complexity to the server (schema design, resolver management)
- REST is simpler to implement but can lead to over-fetching/under-fetching
- GraphQL requires more client-side code for queries/mutations
- REST has better tooling support for documentation and testing
