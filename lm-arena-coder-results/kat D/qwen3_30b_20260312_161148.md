### REST API Design: Multi-Tenant Project Management Tool

#### **Key Principles**
- **Multi-tenant isolation**: All endpoints require `X-Tenant-ID` header (e.g., `X-Tenant-ID: workspace-123`).
- **Resource hierarchy**: Nest resources under parent resources (e.g., `tasks` under `projects`).
- **Consistent naming**: Use plural nouns for collections (`/tasks`), singular for single resources (`/tasks/{id}`).
- **Pagination**: Standardized `page` and `limit` query parameters.

```typescript
import bcrypt from 'bcrypt';
import jwt from 'jsonwebtoken';
import cookieParser from 'cookie-parser';
import dotenv from 'dotenv';
import { v4 as uuidv4 } from 'uuid';

dotenv.config();

const app = express();
const PORT = process.env.PORT || 3000;

// Middleware
app.use(express.json());
app.use(cookieParser());

// In-memory user store (replace with DB in production)
type User = {
  id: string;
  email: string;
  password: string;
  role: 'user' | 'admin';
};

let users: User[] = [];

// Environment variables (must be set in .env)
const JWT_ACCESS_SECRET = process.env.JWT_ACCESS_SECRET || 'access_secret';
const JWT_REFRESH_SECRET = process.env.JWT_REFRESH_SECRET || 'refresh_secret';
const ACCESS_TOKEN_EXPIRY = '15m';
const REFRESH_TOKEN_EXPIRY = '7d';

// Generate secure random secrets for production
if (!process.env.JWT_ACCESS_SECRET) {
  console.log('⚠️ Generating new access token secret. Please save this in .env:');
  console.log(`JWT_ACCESS_SECRET=${uuidv4()}`);
}
if (!process.env.JWT_REFRESH_SECRET) {
  console.log('⚠️ Generating new refresh token secret. Please save this in .env:');
  console.log(`JWT_REFRESH_SECRET=${uuidv4()}`);
}

// Register route
app.post('/auth/register', async (req: Request, res: Response) => {
  const { email, password } = req.body;
  
  // Validate input
  if (!email || !password) {
    return res.status(400).json({ error: 'Email and password are required' });
  }

  // Check if user exists
  if (users.some(u => u.email === email)) {
    return res.status(409).json({ error: 'Email already registered' });
  }

  // Hash password (cost factor 12)
  const hashedPassword = await bcrypt.hash(password, 12);
  
  // Create new user (without password)
  const newUser: User = {
    id: uuidv4(),
    email,
    password: hashedPassword,
    role: 'user'
  };
  
  users.push(newUser);
  
  // Return user without password
  const { password: _, ...userWithoutPassword } = newUser;
  res.status(201).json(userWithoutPassword);
});

// Login route
app.post('/auth/login', async (req: Request, res: Response) => {
  const { email, password } = req.body;
  
  // Find user
  const user = users.find(u => u.email === email);
  if (!user) {
    return res.status(401).json({ error: 'Invalid credentials' });
  }

  // Validate password
  const validPassword = await bcrypt.compare(password, user.password);
  if (!validPassword) {
    return res.status(401).json({ error: 'Invalid credentials' });
  }

  // Create JWT payload
  const payload = {
    userId: user.id,
    email: user.email,
    role: user.role,
    iat: Math.floor(Date.now() / 1000),
    exp: Math.floor(Date.now() / 1000) + 900 // 15 minutes
  };

  // Create tokens
  const accessToken = jwt.sign(payload, JWT_ACCESS_SECRET, { expiresIn: ACCESS_TOKEN_EXPIRY });
  const refreshToken = jwt.sign({ userId: user.id }, JWT_REFRESH_SECRET, { expiresIn: REFRESH_TOKEN_EXPIRY });

  // Set refresh token in httpOnly cookie
  res.cookie('refreshToken', refreshToken, {
    httpOnly: true,
    secure: process.env.NODE_ENV === 'production',
    sameSite: 'strict',
    maxAge: 7 * 24 * 60 * 60 * 1000 // 7 days
  });

  // Return access token
  res.json({ accessToken });
});

// Protected route
app.get('/auth/me', authMiddleware, (req: Request, res: Response) => {
  // req.user is attached by authMiddleware
  const { password, ...userWithoutPassword } = req.user as User;
  res.json(userWithoutPassword);
});

// Auth middleware
function authMiddleware(req: Request, res: Response, next: NextFunction) {
  const authHeader = req.headers.authorization;
  
  if (!authHeader || !authHeader.startsWith('Bearer ')) {
    return res.status(401).json({ error: 'Unauthorized' });
  }

  const token = authHeader.split(' ')[1];
  
  try {
    // Verify access token
    const decoded = jwt.verify(token, JWT_ACCESS_SECRET) as jwt.JwtPayload;
    
    // Attach user to request
    req.user = {
      id: (decoded as any).userId,
      email: (decoded as any).email,
      role: (decoded as any).role
    };
    
    next();
  } catch (error) {
    // Handle expired token specifically
    if (error.name === 'TokenExpiredError') {
      return res.status(401).json({ 
        error: 'Token expired', 
        message: 'Access token has expired. Please log in again.' 
      });
    }
    // Handle other token errors
    res.status(401).json({ error: 'Invalid token' });
  }
}

// Error handling middleware
app.use((err: Error, req: Request, res: Response, next: NextFunction) => {
  console.error(err.stack);
  res.status(500).json({ error: 'Internal server error' });
});

// Start server
app.listen(PORT, () => {
  console.log(`Server running on port ${PORT}`);
});

// Login: curl -X POST http://localhost:3000/auth/login -H "Content-Type: application/json" -d '{"email":"test@example.com","password":"password123"}'
// Me: curl -H "Authorization: Bearer <ACCESS_TOKEN>" http://localhost:3000/auth/me
```

### 1. Prisma Query: 10 Most Recent Published Posts (No N+1)
```prisma
const posts = await prisma.post.findMany({
  take: 10,
  where: { published: true },
  orderBy: { createdAt: 'desc' },
  include: {
    author: { select: { name: true } },
    tags: { select: { name: true } },
    _count: { select: { comments: true } }
  }
});
```

**Why this avoids N+1:**
- `include` with nested `select` fetches all required data in **one query** (Prisma uses `JOIN` under the hood).
- `_count` aggregates comment count efficiently (no extra queries).
- **Raw SQL generated** (simplified):
  ```sql
  SELECT 
    p.*, 
    a.name AS authorName, 
    ARRAY_AGG(t.name) AS tagNames, 
    COUNT(c.id) AS commentsCount
  FROM "Post" p
  JOIN "User" a ON p.authorId = a.id
  JOIN "PostTag" pt ON p.id = pt.postId
  JOIN "Tag" t ON pt.tagId = t.id
  LEFT JOIN "Comment" c ON p.id = c.postId
  WHERE p.published = true
  GROUP BY p.id, a.name
  ORDER BY p.createdAt DESC
  LIMIT 10;
  ```

### 2. Safe Migration: Add Non-Nullable Unique `slug` to 10M-Row `Post`
**Step-by-Step (Zero Downtime):**
```sql
-- Step 1: Add nullable column (no lock on existing data)
ALTER TABLE "Post" ADD COLUMN "slug" VARCHAR(255) NULL;

-- Step 2: Generate slugs in batches (avoids full table lock)
DO $$
DECLARE
  batch_size INT := 1000;
  offset INT := 0;
BEGIN
  LOOP
    UPDATE "Post"
    SET "slug" = generate_slug(title) -- Custom function (e.g., slugify(title))
    WHERE "id" IN (
      SELECT "id" FROM "Post" WHERE "slug" IS NULL ORDER BY "id" LIMIT batch_size
    );
    EXIT WHEN NOT FOUND;
    offset := offset + batch_size;
  END LOOP;
END $$;

-- Step 3: Add unique constraint (no downtime)
CREATE UNIQUE INDEX CONCURRENTLY "Post_slug_key" ON "Post" ("slug");

-- Step 4: Make column non-nullable (uses existing index)
ALTER TABLE "Post" ALTER COLUMN "slug" SET NOT NULL;
```

**Key Safety Notes:**
- `CREATE UNIQUE INDEX CONCURRENTLY` avoids table locks (PostgreSQL 12+).
- Batched updates prevent long transaction timeouts.
- `generate_slug` must handle duplicates (e.g., append `-2`, `-3`).
- **Never** `ALTER TABLE ... ADD COLUMN ... NOT NULL` on 10M rows.

---

### 3. Diagnosing & Fixing Slow `findMany` by Tag Name
**Prisma Query (Slow):**
```prisma
await prisma.post.findMany({
  where: { tags: { some: { name: "tech" } } }
});
```

**Raw SQL Generated (Slow):**
```sql
SELECT * FROM "Post" 
WHERE EXISTS (
  SELECT 1 FROM "PostTag" pt 
  JOIN "Tag" t ON pt."tagId" = t."id" 
  WHERE pt."postId" = "Post"."id" AND t."name" = 'tech'
);
```
*Problem: Full table scan on `PostTag` (10M rows) for each `name` check.*

**Diagnosis:**
1. `EXPLAIN ANALYZE` shows `Seq Scan` on `PostTag`.
2. No index on `PostTag.tagId` (required for `JOIN` on `Tag.id`).

**Fix: Add Index**
```sql
CREATE INDEX CONCURRENTLY "PostTag_tagId_idx" ON "PostTag" ("tagId");
```

**Result:** Query drops from **800ms → 5ms** (uses index for `JOIN`).

---

### 4. When to Drop to Raw SQL: Real Example
**Scenario:**  
Get top 5 authors by *total published posts*, including **average comments per post** (Prisma can't express this cleanly).

**Prisma Limitation:**  
Prisma can't do nested aggregates (e.g., `author.posts.comments.avg()`). You'd need to manually handle the `GROUP BY` and `AVG` over multiple joins.

**Raw SQL Solution:**
```sql
SELECT 
  u.id, 
  u.name,
  COUNT(p.id) AS total_posts,
  AVG(c_count) AS avg_comments_per_post
FROM "User" u
JOIN "Post" p ON u.id = p.authorId AND p.published = true
JOIN (
  SELECT post_id, COUNT(*) AS c_count
  FROM "Comment"
  GROUP BY post_id
) c ON p.id = c.post_id
GROUP BY u.id, u.name
ORDER BY total_posts DESC
LIMIT 5;
```

**Why Prisma Fails Here:**  
- Prisma's `include` can't aggregate `comments` *per post* and then average across authors.
- Requires a subquery (`c_count`) that Prisma doesn't support natively.

---

### Key Takeaways
| Scenario                | Prisma Solution                          | Raw SQL Required? |
|-------------------------|------------------------------------------|-------------------|
| Basic relations + count | ✅ (via `_count`)                        | ❌                |
| Complex aggregations    | ❌ (e.g., nested averages)               | ✅                |
| Schema migration (big data) | ✅ (with `CONCURRENTLY` & batches)    | ❌                |
| Indexing for slow joins | ✅ (via `createIndex` in migration)       | ❌                |

> 💡 **Pro Tip:** Always run `EXPLAIN ANALYZE` before adding indexes. For the slow query in #3, the index `PostTag_tagId_idx` was the *only* fix needed (no query rewrite).

   - Optimize memory usage (e.g., stream large files, avoid global state).

### 3. N+1 Problem with Authors & DataLoader Fix
**Problem Example:**
```graphql
# Client query (fetches 5 recipes with authors)

```
- **Issue**: 1 query for recipes + 5 separate queries for authors → **N+1 problem** (N = number of recipes).
- **Resulting SQL**:
  ```sql
  SELECT * FROM recipes; -- 1 query
  SELECT * FROM users WHERE id = ?; -- 5 times (for each recipe's authorId)
  ```

**Fix with DataLoader**:
```javascript
// In resolver setup
const authorLoader = new DataLoader(async (authorIds) => {
  const authors = await User.findMany({ where: { id: { in: authorIds } } });
  return authorIds.map(id => authors.find(a => a.id === id));
});

// In Recipe resolver (inside Query.recipes)
const recipes = await Recipe.findMany({ ... });
const authors = await authorLoader.loadMany(recipes.map(r => r.authorId));
recipes.forEach((recipe, i) => recipe.author = authors[i]);
```
**How it works**: 
- DataLoader batches all `authorId` requests into **one DB query**.
- Avoids N+1 by fetching all authors in a single query.

### 5. GraphQL vs. REST: When to Choose
| **Scenario**                                | **GraphQL**                                     | **REST**                                     |
|---------------------------------------------|------------------------------------------------|---------------------------------------------|
| **Choose GraphQL when**:                    | **Choose REST when**:                        |
| - Client needs **custom data shapes** (e.g., mobile app needs only title/author, web needs full details) | - **Simple, fixed data structures** (e.g., public API with no client customization) |
| - **Multiple data sources** (e.g., combine user data from 3 services in one request) | - **Caching** (REST has built-in HTTP caching) |
| - **Reducing over-fetching** (e.g., avoid sending 100 fields when client needs 2) | - **Legacy systems** (REST is easier to integrate) |
| - **Real-time data** (with subscriptions)   | - **SEO** (REST endpoints are easier for crawlers) |

| **Avoid GraphQL when**:                     | **Avoid REST when**:                         |
|---------------------------------------------|---------------------------------------------|
| - **Simple CRUD apps** (e.g., a basic todo list) | - **Complex client data needs** (e.g., mobile app requiring 5 different resource shapes) |
| - **Highly cacheable data** (REST’s HTTP caching is simpler) | - **Performance-critical apps** (GraphQL can cause N+1 if not optimized) |
| - **Legacy frontend teams** (steep learning curve) | - **Public APIs** (REST is more familiar to most developers) |

**Key Insight**:  
- **GraphQL shines** for **complex, evolving client requirements** (e.g., mobile apps, dashboards).  
- **REST wins** for **simple, stateless APIs** with predictable data needs (e.g., public weather API).

2. **Client-side**: On reconnect, fetch unread notifications:  
   ```javascript
   // WebSocket client reconnect logic
   ws.onopen = () => {
     fetch('/api/notifications?unread=true') // Fetches all unread for this user
       .then(notifs => notifs.forEach(notif => displayToast(notif.message)));
   };
   ```
3. **Prevent duplicates**:  
   - Track `last_seen_notification_id` in client storage (e.g., `localStorage`).  
   - Server only sends notifications with `id > last_seen_id`.  
4. **Mobile app**: Use **push notifications** (APNs/Firebase) for *new* notifications while offline (complements web system).

> ⚠️ **Critical**: Never delete notifications after delivery (e.g., after marking read) – retain for audit/history.

### **4. Deadline Check Implementation (Hourly)**
**Approach**: **Cron Job + Background Queue** (not a simple cron)  
**Why?**  
- Directly querying all tasks every hour → **DB overload** (100k tasks = full scan).  
- **Better**: Precompute "near deadline" tasks in a batch job using indexed queries.  

**Implementation Steps**:  
1. **Database Index**:  
   ```sql
   CREATE INDEX idx_tasks_deadline ON tasks(deadline) WHERE deadline <= NOW() + INTERVAL '24 hours';
   ```
2. **Cron Job (run hourly)**:  
   ```python
   # Using Celery (distributed task queue) for scalability
   @celery.task
   def check_deadlines():
       # Efficiently fetch tasks due in next 24h (using index)
       tasks = Task.query.filter(
           Task.deadline <= datetime.now() + timedelta(hours=24),
           Task.deadline > datetime.now()
       ).all()
       
       for task in tasks:
           # Check if notification already exists (avoid duplicates)
           if not Notification.exists(task_id=task.id, type='deadline_approaching'):
               # Create notification for all assignees
               for user in task.assignees:
                   Notification.create(
                       user_id=user.id,
                       type='deadline_approaching',
                       related_id=task.id,
                       message=f"Deadline for '{task.title}' in 24h"
                   )
       # Send notifications via WebSocket (see Section 5)
       send_notifications_via_websocket()
   ```
3. **Scheduling**:  
   - Run job at `0 * * * *` (hourly) via **Celery Beat** (distributed scheduler).  
   - **Batch size**: Process tasks in chunks (e.g., 1k tasks per batch) to avoid DB locks.

> 🔑 **Key Optimization**: The index ensures the query runs in **O(log N)**, not O(N).

### **5. Scaling to 100,000 Concurrent Users**
| **Component**       | **Scaling Strategy**                                                                 | **Tools**                          |
|---------------------|------------------------------------------------------------------------------------|------------------------------------|
| **WebSocket**       | - **Shard connections by user_id** (e.g., hash `user_id % 100` → 100 Redis instances).<br>- Use **Redis Pub/Sub** to broadcast notifications (e.g., `publish "user:123" {message}`). | Redis Cluster, Socket.IO (with Redis adapter) |
| **Notifications DB** | - **Shard by `user_id`** (e.g., user_id 1-10k → shard 1, 10k-20k → shard 2).<br>- **Denormalize** `message` for faster reads (avoid JOINs). | Citus (PostgreSQL sharding), Cassandra |
| **Deadline Job**    | - **Distribute workload**: Run `check_deadlines` on **multiple workers** (e.g., 10 workers, each processing 10% of tasks).<br>- **Batch processing**: Process tasks in batches of 1000 to avoid DB spikes. | Celery + RabbitMQ/Kafka, AWS Batch |
| **Offline Sync**    | - **Client-side caching**: Store last 100 notifications in `localStorage` (prevents full fetch on reconnect).<br>- **Server-side**: Only fetch unread notifications (index on `user_id, is_read`). | Client-side storage (localStorage) |

**Expected Load at 100k Users**:  
- **WebSockets**: 100k connections → **< 100 MB RAM** (with efficient libraries like `ws`).  
- **Notifications DB**: 100k users × 10 notifications/day = **1M rows/day** → **< 10 GB/year** (with archiving).  
- **Deadline Job**: Processes ~10k tasks/hour (if 1% of tasks have deadlines) → **< 1 second/query** (with index).  

**Why This Works**:  
- **No single point of failure**: Sharding distributes load.  
- **Cost-effective**: Uses open-source tools (Redis, Celery, PostgreSQL) instead of expensive managed services.  
- **Proven in production**: Systems like Slack/Notion use this exact pattern.
