  role: 'user' | 'admin';

app.post('/auth/register', async (req, res) => {
  const { email, password } = req.body;

  if (!email || !password) {
    return res.status(400).json({ error: 'Email and password are required' });
  }

  // Check if user exists
  const existingUser = users.find(u => u.email === email);
  if (existingUser) {
    return res.status(400).json({ error: 'User already exists' });
  }

  // Hash password
  const salt = await bcrypt.genSalt(12);
  const passwordHash = await bcrypt.hash(password, salt);

  // Create user
  const newUser: User = {
    id: Date.now().toString(),
    email,
    passwordHash,
    role: 'user'
  };

  users.push(newUser);

  res.status(201).json({
    id: newUser.id,
    email: newUser.email,
    role: newUser.role
  });
});

// 2. Login
app.post('/auth/login', async (req, res) => {
  const { email, password } = req.body;

  if (!email || !password) {
    return res.status(400).json({ error: 'Email and password are required' });
  }

  // Find user
  const user = users.find(u => u.email === email);
  if (!user) {
    return res.status(401).json({ error: 'Invalid credentials' });
  }

  // Verify password
  const validPassword = await bcrypt.compare(password, user.passwordHash);
  if (!validPassword) {
    return res.status(401).json({ error: 'Invalid credentials' });
  }

  // Generate tokens
  const tokens = generateTokens(user.id, user.email, user.role);

  // Set refresh token cookie
  res.cookie('refreshToken', tokens.refreshToken, {
    httpOnly: true,
    secure: process.env.NODE_ENV === 'production',
    maxAge: 7 * 24 * 60 * 60 * 1000,
    sameSite: 'strict'
  });

  res.json({
    accessToken: tokens.accessToken,
    user: {
      id: user.id,
      email: user.email,
      role: user.role
    }
  });
});

app.get('/auth/me', AuthMiddleware, (req, res) => {
  res.json(req.user);
});

// Auth middleware
export const AuthMiddleware = (req: express.Request, res: express.Response, next: express.NextFunction) => {
  const accessToken = req.headers.authorization?.split(' ')[1];

  if (!accessToken) {
    return res.status(401).json({ error: 'Unauthorized' });
  }

  try {
    const decoded = jwt.verify(accessToken, process.env.JWT_SECRET!);
    req.user = decoded as { userId: string; email: string; role: string; iat: number; exp: number };
    next();
  } catch (error) {
    return res.status(401).json({ error: 'Invalid token' });
  }
};

  const accessToken = jwt.sign(

    { expiresIn: '15m' }
  );

    { userId, email, role },
    process.env.JWT_REFRESH_SECRET!,
    { expiresIn: '7d' }

};

Why use httpOnly cookies for refresh tokens:

1. **Protection against XSS (Cross-Site Scripting)**:
   - httpOnly cookies are not accessible via JavaScript
   - Prevents malicious scripts from stealing the refresh token
   - Even if an attacker compromises your frontend, they can't access the refresh token

2. **Secure Transmission**:
   - Cookies can be marked as `secure` (only sent over HTTPS)
   - `SameSite=strict` prevents CSRF attacks
   - Better security compared to localStorage which is accessible to JavaScript

3. **Separation of Concerns**:
   - Access tokens (in localStorage) are for frontend API calls
   - Refresh tokens (in httpOnly cookies) are for backend refresh logic
   - Maintains proper security boundaries

4. **Token Management**:
   - Easier to manage token lifecycle
   - Can automatically remove expired refresh tokens
   - Better control over token revocation

6. Token Blacklisting vs. Rotation:

**Token Blacklisting:**
- Store tokens in a database as "blacklisted" when they need to be invalidated
- Check each token against the blacklist before allowing access
- Pros: Simple to implement
- Cons: 
  - Database load increases
  - Hard to scale
  - Not real-time (depends on cache)
  - Storage requirements grow

**Token Rotation:**
- Issue a new token and invalidate the old one
- Store only current valid tokens
- Pros:
  - More secure
  - Lower storage requirements
  - Better performance
- Cons:
  - More complex implementation
  - Requires coordination between services

**When to use each:**
- Use blacklisting for:
  - Simple applications
  - Short-lived tokens
  - When real-time invalidation isn't critical

- Use rotation for:
  - Security-critical applications
  - Long-lived tokens
  - High-security environments
  - When immediate invalidation is required

This implementation provides a secure token management system with:

- Efficient session invalidation
- Redis-based storage
- Protection against token reuse
- Password change session invalidation

  columns: Array<{
    key: keyof T;
    header: string;
    width?: string;
    sortable?: boolean;
    renderCell?: (rowData: T) => React.ReactNode;
  }>;
  data: T[];
  pageSize?: number;
  controlled?: boolean;
  selectedRows?: T[];
  onSelectRows?: (rows: T[]) => void;
  pagination?: boolean;
  initialSort?: { key: keyof T; order: 'asc' | 'desc' | 'none' };
}

const DataTable = <T extends object>({
  columns,
  data,
  pageSize = 10,
  controlled = false,
  selectedRows = [],
  onSelectRows,
  pagination = true,
  initialSort = { key: 'id' as keyof T, order: 'none' },
}: DataTableProps<T>) => {
  const [sort, setSort] = useState({
    key: initialSort.key,
    order: initialSort.order,
  });
  const [page, setPage] = useState(1);

  const sortedData = useMemo(() => {
    if (sort.order === 'none') return data;
    return [...data].sort((a, b) => {
      const aVal = a[sort.key];
      const bVal = b[sort.key];
      if (typeof aVal === 'string') {
        return sort.order === 'asc'
          ? aVal.localeCompare(bVal)
          : bVal.localeCompare(aVal);
      }
      return sort.order === 'asc' ? aVal - bVal : bVal - aVal;
    });
  }, [data, sort]);

  const paginatedData = useMemo(() => {
    const start = (page - 1) * pageSize;
    return sortedData.slice(start, start + pageSize);
  }, [sortedData, page, pageSize]);

  const selectedRowIds = new Set(
    selectedRows.map((row) => (row as any).id)
  );

  const handleSort = (key: keyof T) => {
    let newOrder: 'asc' | 'desc' | 'none' = 'asc';
    if (sort.key === key) {
      newOrder = sort.order === 'asc' ? 'desc' : 'none';
    }
    setSort({ key, order: newOrder });
  };

  const handleRowSelect = (row: T) => {
    const isSelected = selectedRowIds.has((row as any).id);
    const newSelectedRows = isSelected
      ? selectedRows.filter((r) => r.id !== row.id)
      : [...selectedRows, row];
    if (controlled) {
      onSelectRows?.(newSelectedRows);
    } else {
      selectedRowIds.clear();
      newSelectedRows.forEach((r) => selectedRowIds.add(r.id));
    }
  };

  const handleSelectAll = () => {
    const allSelected = selectedRowIds.size === data.length;
    const newSelectedRows = allSelected
      ? []
      : data.map((row) => row);
    if (controlled) {
      onSelectRows?.(newSelectedRows);
    } else {
      selectedRowIds.clear();
      newSelectedRows.forEach((r) => selectedRowIds.add(r.id));
    }
  };

  return (
    <div className="table-container">
      <table>
        <thead>
          <tr>
            <th>
              <input
                type="checkbox"
                checked={selectedRowIds.size === data.length}
                onChange={handleSelectAll}
              />
            </th>
            {columns.map((column) => (
              <th
                key={column.key}
                onClick={() => column.sortable && handleSort(column.key)}
                style={{ width: column.width }}
              >
                {column.header}
                {column.sortable && (
                  <span>
                    {sort.key === column.key && sort.order !== 'none' && (
                      <span>{sort.order === 'asc' ? ' ↑' : ' ↓'}</span>
                    )}
                  </span>
                )}
              </th>
            ))}
          </tr>
        </thead>
        <tbody>
          {paginatedData.map((row) => (
            <tr key={(row as any).id}>
              <td>
                <input
                  type="checkbox"
                  checked={selectedRowIds.has((row as any).id)}
                  onChange={() => handleRowSelect(row)}
                />
              </td>
              {columns.map((column) => (
                <td key={column.key}>
                  {column.renderCell
                    ? column.renderCell(row)
                    : (row as any)[column.key]}
                </td>
              ))}
            </tr>
          ))}
        </tbody>
      </table>
      {pagination && (

            disabled={page === 1}
          >

const CACHE_DURATION = 5 * 60 * 1000; // 5 minutes

export function useSearch(query: string) {
  const [loading, setLoading] = useState(false);
  const [data, setData] = useState<SearchResponse | null>(null);
  const [error, setError] = useState<string | null>(null);

  // Cache storage
  const cacheKey = `search_${query}`;
  const cache = {
    get: () => {
      const cached = localStorage.getItem(cacheKey);
      if (cached) {
        const { timestamp, result } = JSON.parse(cached);
        if (Date.now() - timestamp < CACHE_DURATION) {
          return result;
        }
      }
      return null;
    },
    set: (result: SearchResponse) => {
      localStorage.setItem(cacheKey, JSON.stringify({
        timestamp: Date.now(),
        result
      }));
    }
  };

  // Debounced search function
  const debouncedSearch = debounce((searchQuery: string) => {
    if (!searchQuery) {
      setData(null);
      return;
    }

    const cachedResult = cache.get();
    if (cachedResult) {
      setData(cachedResult);
      return;
    }

    const controller = new AbortController();
    setLoading(true);
    setError(null);

    fetch(`/api/search?q=${searchQuery}`, { signal: controller.signal })
      .then(res => res.json())
      .then((result: SearchResponse) => {
        if (controller.signal.aborted) return;
        setData(result);
        cache.set(result);
      })
      .catch(err => {
        if (controller.signal.aborted) return;
        setError(err.message);
      })
      .finally(() => {
        if (!controller.signal.aborted) {
          setLoading(false);
        }
      });

    // Cleanup function to cancel in-flight request
    return () => controller.abort();
  }, 300);

  // Effect to trigger search on query change
  useEffect(() => {
    const cleanup = debouncedSearch(query);
    return cleanup;
  }, [query]);

  return { loading, data, error };

  constructor(initialText = '') {
    this.text = initialText;
    this.version = 0;
    this.users = new Map(); // userId -> cursorPosition
  }
}

// Server setup
const wss = new WebSocket.Server({ port: 8080 });
const documents = new Map(); // documentId -> Document
const connections = new Map(); // ws -> Connection

class Connection {
  constructor(ws) {
    this.ws = ws;
    this.userId = uuid.v4();
    this.documentId = null;
    this.cursorPosition = 0;
  }
}

wss.on('connection', (ws) => {
  const connection = new Connection(ws);
  connections.set(ws, connection);

  ws.on('message', (message) => {
    const data = JSON.parse(message);
    handleMessage(connection, data);
  });

  ws.on('close', () => {
    handleDisconnect(connection);
  });
});

// Message protocol:
// 1. JOIN_DOC
// 2. TEXT_CHANGE
// 3. CURSOR_MOVE
// 4. USER_LEFT

function handleMessage(connection, data) {
  const { type, payload } = data;

  switch (type) {
    case 'JOIN_DOC': {
      const { documentId } = payload;
      connection.documentId = documentId;
      
      // Create document if it doesn't exist
      if (!documents.has(documentId)) {
        documents.set(documentId, new Document());
      }
      
      const document = documents.get(documentId);
      document.users.set(connection.userId, connection.cursorPosition);

      // Send latest state to new user
      broadcastToUser(connection, {
        type: 'DOC_STATE',
        payload: {
          text: document.text,
          version: document.version,
          users: Array.from(document.users.entries())
        }
      });

      // Notify others of new user
      broadcastToDocumentExcept(connection, {
        type: 'USER_JOINED',
        payload: {
          userId: connection.userId,
          cursorPosition: connection.cursorPosition
        }
      });
      break;
    }

    case 'TEXT_CHANGE': {
      const { start, end, newText } = payload;
      const document = documents.get(connection.documentId);
      
      // Apply change
      document.text = document.text.slice(0, start) + newText + document.text.slice(end);
      document.version++;
      
      broadcastToDocumentExcept(connection, {
        type: 'TEXT_CHANGE',
        payload: {
          start,
          end,
          newText,
          version: document.version
        }
      });
      break;
    }

    case 'CURSOR_MOVE': {
      const { position } = payload;
      connection.cursorPosition = position;
      
      const document = documents.get(connection.documentId);
      document.users.set(connection.userId, position);

      broadcastToDocumentExcept(connection, {
        type: 'CURSOR_MOVE',
        payload: {
          userId: connection.userId,
          position
        }
      });
      break;
    }
  }
}

function handleDisconnect(connection) {
  if (!connection.documentId) return;

  const document = documents.get(connection.documentId);
  if (document) {
    document.users.delete(connection.userId);
    broadcastToDocumentExcept(connection, {
      type: 'USER_LEFT',
      payload: {
        userId: connection.userId
      }
    });
  }

  connections.delete(connection.ws);
}

function broadcastToDocumentExcept(connection, message) {
  const document = documents.get(connection.documentId);
  if (!document) return;

  for (const [userId, userConnection] of connections.entries()) {
    if (userConnection.documentId === connection.documentId && userConnection.userId !== connection.userId) {
      broadcastToUser(userConnection, message);
    }
  }
}

function broadcastToUser(connection, message) {
  connection.ws.send(JSON.stringify(message));
}

**A:**
I'll break down each part of the notification system design:

1. Transport Mechanism:
I recommend using WebSocket with a fallback to SSE (Server-Sent Events) for browsers that don't support WebSocket.

Why WebSocket:
- Real-time, bi-directional communication
- Lower latency
- Better performance with less overhead compared to polling
- Can handle multiple connections efficiently

Fallback to SSE:
- Ensures compatibility with older browsers
- Simpler to implement than WebSocket
- Server-to-client only, but sufficient for notifications

Implementation approach:
```python
# Using FastAPI with WebSocket support
from fastapi import WebSocket

class NotificationManager:
    def __init__(self):
        self.active_connections = {}

    async def connect(self, websocket: WebSocket, user_id: str):
        await websocket.accept()
        if user_id not in self.active_connections:
            self.active_connections[user_id] = []
        self.active_connections[user_id].append(websocket)

    async def disconnect(self, websocket: WebSocket, user_id: str):
        if user_id in self.active_connections:
            self.active_connections[user_id].remove(websocket)
            if not self.active_connections[user_id]:
                del self.active_connections[user_id]

    async def send_personal_notification(self, user_id: str, message: str):
        if user_id in self.active_connections:
            for connection in self.active_connections[user_id]:
                await connection.send_text(message)

    # SSE fallback
    @app.get("/notifications/{user_id}")
    async def get_notifications(user_id: str):
        while True:
            if await has_notifications(user_id):
                yield {"data": get_notification(user_id)}
            await asyncio.sleep(5)
```
