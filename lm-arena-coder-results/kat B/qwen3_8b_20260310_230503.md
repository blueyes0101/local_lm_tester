```typescript
import express, { Request, Response, NextFunction } from 'express';
import cors from 'cors';
import bcrypt from 'bcrypt';
import jwt from 'jsonwebtoken';
import { v4 as uuidv4 } from 'uuid';
import { User } from './models/user';

const app = express();
app.use(express.json());
app.use(cors());

// In-memory user storage
const users: User[] = [];

// Register route
app.post('/auth/register', async (req: Request, res: Response) => {
  const { email, password } = req.body;
  const existingUser = users.find(user => user.email === email);
  if (existingUser) {
    return res.status(400).send('Email already exists');
  }
  const hashedPassword = await bcrypt.hash(password, 12);
  const newUser: User = {
    id: uuidv4(),
    email,
    password: hashedPassword,
    role: 'user'
  };
  users.push(newUser);
  res.status(201).send({ id: newUser.id, email: newUser.email, role: newUser.role });
});

// Login route
app.post('/auth/login', async (req: Request, res: Response) => {
  const { email, password } = req.body;
  const user = users.find(u => u.email === email);
  if (!user || !(await bcrypt.compare(password, user.password))) {
    return res.status(401).send('Invalid credentials');
  }
  const accessToken = generateAccessToken(user);
  const refreshToken = uuidv4();
  res.cookie('refreshToken', refreshToken, {
    httpOnly: true,
    secure: process.env.NODE_ENV === 'production',
    maxAge: 7 * 24 * 60 * 60 * 1000 // 7 days
  });
  res.json({ accessToken });
});

// Auth middleware
function authMiddleware(req: Request, res: Response, next: NextFunction) {
  const accessToken = req.headers.authorization?.split(' ')[1];
  if (!accessToken) {
    return res.status(401).send('No token provided');
  }
  try {
    const decoded = jwt.verify(accessToken, process.env.JWT_SECRET as string);
    req.user = decoded;
    next();
  } catch (err) {
    if (err.name === 'TokenExpiredError') {
      return res.status(401).send({ error: 'Token expired' });
    }
    res.status(401).send('Invalid token');
  }
}

// Protected route
app.get('/auth/me', authMiddleware, (req: Request, res: Response) => {
  res.json(req.user);
});

// Generate access token
function generateAccessToken(user: User): string {
  const payload = {
    userId: user.id,
    email: user.email,
    role: user.role,
    iat: Math.floor(Date.now() / 1000),
    exp: Math.floor(Date.now() / 1000) + 15 * 60 // 15 minutes
  };
  return jwt.sign(payload, process.env.JWT_SECRET as string, { expiresIn: '15m' });
}

// Start server
const PORT = 3000;
app.listen(PORT, () => {
  console.log(`Server running on http://localhost:${PORT}`);
});
```

### 🔐 Security Considerations: `httpOnly` Cookie

**Implementation:**  
- **Store Refresh Tokens per User:**  
  Use a Redis set or hash to track all active refresh tokens for a user.  
  ```bash
  SADD refresh_tokens:user123 "token1" "token2" "token3"
  ```
- **On Password Change:**  
  - Delete the entire set for the user:  
    ```bash
    DEL refresh_tokens:user123
    ```

- **Result:** All active sessions (refresh tokens) for the user are invalidated.  

---

### 6. **Token Blacklisting vs. Rotation**  
**Blacklisting:**  
- **Definition:** Tokens are added to a blacklist (e.g., Redis set) and are invalidated immediately.  
- **Use Case:** When tokens cannot be rotated (e.g., JWTs with long expiration).  
- **Example:** Logout invalidates the token by adding it to a blacklist.  

**Rotation:**  
- **Definition:** Issue new tokens and invalidate the old ones.  
- **Use Case:** When tokens can be periodically renewed (e.g., refresh tokens).  
- **Example:** Refresh token rotation invalidates the old token and issues a new one.  

**When to Use Each:**  
- **Blacklisting:** Required for systems where tokens cannot be rotated (e.g., JWTs).  
- **Rotation:** Preferred for systems with refresh tokens, as it avoids the need for a blacklist and allows for periodic renewal.  

**Key Difference:**  
- **Blacklisting** is a reactive approach (invalidating tokens after use).  
- **Rotation** is a proactive approach (replacing tokens before they expire).

interface RowData {
  id: string;
  // Add more fields as needed
}

interface ColumnDefinition {
  key: string;
  header: string;
  width?: string;
  sortable?: boolean;
  renderCell?: (row: RowData) => React.ReactNode;
}

interface DataTableProps {
  columns: ColumnDefinition[];
  data: RowData[];
  onRowSelect?: (selectedRows: string[]) => void;
  onSort?: (columnKey: string, direction: 'asc' | 'desc' | 'none') => void;
  onPageChange?: (page: number) => void;
  onPageSizeChange?: (size: number) => void;
}

const DataTable: React.FC<DataTableProps> = ({
  columns,
  data,
  onRowSelect,
  onSort,
  onPageChange,
  onPageSizeChange,
}) => {
  // State for row selection, pagination, sorting
  const [selectedRows, setSelectedRows] = useState<string[]>([]);
  const [currentPage, setCurrentPage] = useState(1);
  const [pageSize, setPageSize] = useState(10);
  const [sortColumn, setSortColumn] = useState<string | null>(null);
  const [sortDirection, setSortDirection] = useState<'asc' | 'desc' | 'none'>('none');

  // Memoized sorted data
  const sortedData = useMemo(() => {
    if (!sortColumn) return data;

    const sorted = [...data].sort((a, b) => {
      const column = columns.find(col => col.key === sortColumn);
      if (!column) return 0;

      const valA = column.renderCell ? column.renderCell(a) : a[column.key];
      const valB = column.renderCell ? column.renderCell(b) : b[column.key];

      if (typeof valA === 'string' && typeof valB === 'string') {
        return sortDirection === 'asc' ? valA.localeCompare(valB) : valB.localeCompare(valA);
      } else if (typeof valA === 'number' && typeof valB === 'number') {
        return sortDirection === 'asc' ? valA - valB : valB - valA;
      }
      return 0;
    });

    return sorted;
  }, [data, sortColumn, sortDirection]);

  // Memoized paginated data
  const paginatedData = useMemo(() => {
    const start = (currentPage - 1) * pageSize;
    const end = start + pageSize;
    return sortedData.slice(start, end);
  }, [sortedData, currentPage, pageSize]);

  // Handle row selection
  const handleRowSelect = (rowId: string) => {
    setSelectedRows(prev => {
      if (prev.includes(rowId)) {
        return prev.filter(id => id !== rowId);
      } else {
        return [...prev, rowId];
      }
    });
  };

  // Handle select all rows
  const handleSelectAll = () => {
    const allRowIds = sortedData.map(row => row.id);
    setSelectedRows(prev => {
      if (prev.length === allRowIds.length) {
        return [];
      } else {
        return allRowIds;
      }
    });
  };

  // Handle column sorting
  const handleSort = (columnKey: string) => {
    if (!columns.find(col => col.key === columnKey)) return;

    if (sortColumn === columnKey) {
      setSortDirection(prev => {
        if (prev === 'asc') return 'desc';
        if (prev === 'desc') return 'none';
        return 'asc';
      });
    } else {
      setSortColumn(columnKey);
      setSortDirection('asc');
    }

```jsx
  const [debouncedQuery, setDebouncedQuery] = useState(initialQuery);
  const [results, setResults] = useState(null);
  const [loading, setLoading] = useState(false);

    if (!debouncedQuery) {
      setResults(null);
      setLoading(false);
      setError(null);
      return;
    }

    // Abort any previous request
    if (abortControllerRef.current) {
      abortControllerRef.current.abort();
    }
    abortControllerRef.current = new AbortController();

    setLoading(true);
    setError(null);

    fetch(`/api/search?q=${encodeURIComponent(debouncedQuery)}`, {
      signal: abortControllerRef.current.signal,

        if (!res.ok) throw new Error("Network response was not ok");
        return res.json();

        const cached = cache.current.get(debouncedQuery);

        } else {

          setResults(data);

      })
      .catch((err) => {
        if (err.name === "AbortError") return;
        setError("Failed to fetch results. Please try again.");
        setLoading(false);
      });
  }, [debouncedQuery]);

  };
```

```javascript
const wss = new WebSocket.Server({ port: 8080 });

let userIdCounter = 0;
let connectionToUser = new Map();  // WebSocket -> userId
let userToConnection = new Map();  // userId -> WebSocket
let documents = new Map();         // documentId -> text
let userDocuments = new Map();     // documentId -> Set of userIds

function generateUserId() {
    return `user_${userIdCounter++}`;
}

function applyTextChange(text, position, content) {
    return text.slice(0, position) + content + text.slice(position);
}

wss.on('connection', (ws) => {
    const userId = generateUserId();
    connectionToUser.set(ws, userId);
    userToConnection.set(userId, ws);

    ws.on('message', (message) => {
        const data = JSON.parse(message);
        handleMessage(ws, data);
    });

    ws.on('close', () => {
        const userId = connectionToUser.get(ws);
        if (userId) {
            userToConnection.delete(userId);
            connectionToUser.delete(ws);

            for (const [docId, users] of userDocuments.entries()) {
                users.delete(userId);
                if (users.size === 0) {
                    userDocuments.delete(docId);
                }
            }
        }
    });

function handleMessage(ws, data) {
    const userId = connectionToUser.get(ws);
    switch (data.type) {
        case 'join':
            handleJoin(ws, data, userId);
            break;
        case 'textChange':
            handleTextChange(ws, data, userId);
            break;
        case 'cursorMove':
            handleCursorMove(ws, data, userId);
            break;
        case 'userLeft':
            handleUserLeft(ws, data, userId);
            break;

function handleJoin(ws, data, userId) {
    const docId = data.documentId;

    if (!documents.has(docId)) {
        documents.set(docId, '');
    }

    if (!userDocuments.has(docId)) {
        userDocuments.set(docId, new Set());
    }

    userDocuments.get(docId).add(userId);

    const text = documents.get(docId);
    ws.send(JSON.stringify({ type: 'documentText', text }));

    const text = documents.get(docId) || '';
```

### 5. **GraphQL vs. REST: When to Use Each**

| **Use GraphQL When**                                 | **Use REST When**                                 |
|------------------------------------------------------|----------------------------------------------------|
| - Flexible data fetching (client specifies what data to retrieve) | - Simple CRUD operations (e.g., GET /recipes) |
| - Reduce the number of requests (one GraphQL request instead of multiple REST calls) | - Need strict data formats (e.g., JSON APIs with fixed schemas) |
| - Complex data models with nested relationships       | - Caching is critical (REST supports HTTP caching better) |
| - Real-time updates (via subscriptions)              | - API is simple and predictable (e.g., mobile apps with minimal data needs) |

**Avoid GraphQL When:**
- The API is overly simple or has minimal data requirements.
- You need to enforce strict data validation or authentication at the API level.
- Performance is critical and requires fine-grained control over request/response size.

## Question 12
**Q:** You are asked to add real-time notifications to the existing project management tool. Users should be notified when: a task is assigned to them, a comment is added to their task, a deadline is approaching (within 24h).

Design the notification system:
1. What transport mechanism would you use and why? (WebSocket, SSE, polling, push notifications — compare and justify)
2. How do you store notifications? Design the database schema.
3. How do you handle a user who is offline — do they get missed notifications when they come back online?
4. The deadline check needs to run every hour for all tasks. How do you implement this? (cron job, queue, event-driven?) — show the implementation approach.
5. How do you scale this if there are 100,000 concurrent users?

**A:**
### 1. **Transport Mechanism: WebSocket**  
**Why WebSocket?**  
WebSocket provides **bidirectional, real-time communication** between the client and server, making it ideal for instant notifications. Unlike polling (which is inefficient) or SSE (unidirectional), WebSocket allows the server to push updates immediately when events occur (e.g., task assignment, comment addition). For deadline alerts, WebSocket ensures notifications are delivered **instantly** when the deadline is within 24 hours.  
**Why Not Other Options?**  
- **Polling**: Inefficient due to frequent client-server round trips.  
- **SSE**: Only supports server-to-client unidirectional communication, limiting use cases like user-initiated actions.  
- **Push Notifications**: Platform-dependent (e.g., mobile OS APIs) and less reliable for web apps.  

**Implementation**:  
- Use WebSocket libraries (e.g., `ws` for Node.js, `Socket.IO` for real-time features).  
- Maintain a **connection pool** to track active users and their WebSocket connections.  

---

### 2. **Database Schema for Notifications**  
**Schema Design**:  
```sql
CREATE TABLE notifications (
    id UUID PRIMARY KEY,
    user_id UUID NOT NULL,
    task_id UUID NOT NULL,
    notification_type ENUM('ASSIGNED', 'COMMENTED', 'DEADLINE') NOT NULL,
    message TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    read_status BOOLEAN DEFAULT FALSE,
    delivery_status ENUM('PENDING', 'SENT', 'FAILED') DEFAULT 'PENDING'
);
```
**Key Fields**:  
- `user_id`: Identifier for the user receiving the notification.  
- `task_id`: Link to the task triggering the notification.  
- `notification_type`: Type of event (assignment, comment, deadline).  
- `message`: Customizable notification content.  
- `read_status`: Tracks if the user has read the notification.  
- `delivery_status`: Tracks delivery attempts (e.g., if the user is offline).  

**Additional Tables**:  
- **tasks**: Store task details (e.g., `deadline`, `assignee_id`).  
- **users**: Track user preferences (e.g., notification preferences).  

---

### 3. **Handling Offline Users**  
**Strategy**:  
- **Queue Notifications**: When a user is offline (no active WebSocket connection), store the notification in the `notifications` table with `delivery_status = 'PENDING'`.  
- **Reconnect Detection**: Use a **heartbeat mechanism** (e.g., client sends periodic pings) to track online/offline status.  
- **Delivery on Reconnect**: When the user reconnects, the server checks for pending notifications and sends them via WebSocket.  
- **Fallback for Missed Notifications**: For critical alerts (e.g., deadlines), use **email/SMS** as a fallback if the user remains offline for extended periods.  

**Implementation**:  
- Client sends a `keepalive` message every 30 seconds.  
- Server marks users as offline if no heartbeat is received for 5 minutes.  
- On reconnect, the server queries the database for `PENDING` notifications and sends them.  

---

### 4. **Deadline Check Every Hour**  
**Implementation Approach**:  
- **Scheduled Task (Cron Job)**:  
  Use a cron job (or a scheduler like `Celery`/`Airflow`) to run a script every hour.  
  - Query tasks where `deadline <= NOW() + 24h` and `assignee_id` is valid.  
  - For each task, generate a `DEADLINE` notification and store it in the database.  
- **Optimization**:  
  - Index `tasks.deadline` for fast filtering.  
  - Use a **background worker** (e.g., `Celery` with Redis) to process the task check asynchronously.  

**Example Code (Pseudocode)**:  
```python
def check_deadlines():
    tasks = db.query("SELECT * FROM tasks WHERE deadline <= NOW() + INTERVAL '24 hours'")
    for task in tasks:
        if task.assignee_id:
            notify_user(task.assignee_id, "DEADLINE", f"Task {task.id} is due in 24 hours")
```

### 5. **Scaling for 100,000 Concurrent Users**  
**Key Strategies**:  
1. **WebSocket Scaling**:  
   - Use **load balancers** (e.g., NGINX) to distribute WebSocket connections across multiple servers.  
   - Implement **WebSocket clustering** (e.g., `Socket.IO` with Redis for room management).  

2. **Database Scaling**:  
   - Use **sharded databases** (e.g., partition `notifications` by `user_id` or `task_id`).  
   - Add **caching** (e.g., Redis) for frequently accessed data (e.g., user connections, task deadlines).  

3. **Message Queue for Notifications**:  
   - Use **Kafka/RabbitMQ** to decouple notification generation from delivery.  
   - Workers process messages from the queue and send notifications via WebSocket.  

4. **Distributed Scheduling**:  
   - Use **Celery with Redis** or **Airflow** to manage the hourly deadline check.  
   - Ensure the scheduler runs on multiple nodes for redundancy.  

5. **Offline Notification Handling**:  
   - Store notifications in a **write-ahead log** (e.g., `notifications` table) and use a **background job** to deliver them when the user is online.
