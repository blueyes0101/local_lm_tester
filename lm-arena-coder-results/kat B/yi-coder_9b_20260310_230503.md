**Q:** Implement the following endpoint in Node.js + Express + TypeScript:

PATCH /workspaces/:workspaceId/tasks/:taskId

Requirements:
- Partial update: only fields provided in the body should change
- Allowed fields: title (string, 1-200 chars), status (todo|in_progress|done|cancelled), assignee_id (integer or null), due_date (ISO8601 or null)
- Validate that workspaceId and taskId are valid integers
- Validate that the task belongs to the workspace (return 404 if not)
- Validate that assignee_id (if provided) is a member of the workspace
- Return 400 with field-level validation errors if validation fails
- Return 409 if trying to set status to 'cancelled' when task has open subtasks
- Use Zod for validation
- Write the complete handler + validation schema + error middleware
- Write 4 unit tests covering: success, validation error, 404, 409

**A:**
Here's a basic implementation of the required endpoint. This assumes a couple of things:

- You have a workspace model `Workspace` with methods `isMember` and `findTask`
- You have a task model `Task` with methods `update` and `subtasksCount`
- You're using an ORM like Sequelize that provides these methods.

```typescript
import express from 'express';
import { z } from 'zod';

const router = express.Router();

const taskUpdateSchema = z.object({
  title: z.string().max(200).optional(),
  status: z.enum(['todo', 'in_progress', 'done', 'cancelled']).optional(),
  assignee_id: z.number().int().nullable().optional(),
  due_date: z.date().nullable().optional(),
});

async function validateRequest(req, res, next) {
  const { workspaceId, taskId } = req.params;
  const workspace = await Workspace.findByPk(workspaceId);
  if (!workspace) return res.status(404).send('Workspace not found');
  const task = await workspace.findTask(taskId);
  if (!task) return res.status(404).send('Task not found');
  const assignee_id = req.body.assignee_id;
  if (assignee_id !== undefined && !await workspace.isMember(assignee_id))
    return res.status(400).send('Invalid assignee_id');
  if (task.status === 'cancelled' && task.subtasksCount() > 0)
    return res.status(409).send('Cannot cancel task with open subtasks');
  req.workspace = workspace;
  req.task = task;
  next();
}

router.patch('/workspaces/:workspaceId/tasks/:taskId', validateRequest, async (req, res) => {
  const { workspace, task } = req;
  const data = taskUpdateSchema.parse(req.body);
  await task.update({ ...data, workspaceId: workspace.id });
  res.send(task);
});

export default router;
```

5. Write authMiddleware that validates the access token and attaches user to req.user
6. What happens if someone sends an expired token? Show the exact error response.
7. Why httpOnly cookie for refresh token instead of localStorage? Explain the security reasoning.

**A:**
Here's a step by step guide on how to create a complete authentication system in Node.js + Express + TypeScript:

Firstly, install the necessary packages:

```bash
$ npm install express bcryptjs jsonwebtoken cookie-parser dotenv
```

Now, let's define an `AuthController` that implements the required endpoints:

```typescript
import express from 'express';
import bcrypt from 'bcryptjs';
import jwt from 'jsonwebtoken';
import { User } from './models/User';

const router = express.Router();

router.post('/register', async (req, res) => {
  const { email, password } = req.body;

  const hashedPassword = await bcrypt.hash(password, 12);
  const user = new User({ email, password: hashedPassword });
  
  try {
    await user.save();
    res.status(201).send({ user: user.toJSON() });
  } catch (err) {
    res.status(500).send({ error: err.toString() });
  }
});

router.post('/login', async (req, res) => {
  const { email, password } = req.body;

  const user = await User.findOne({ email });
  if (!user) {
    return res.status(400).send({ error: 'User not found' });
  }

  const passwordMatches = await bcrypt.compare(password, user.password);
  if (!passwordMatches) {
    return res.status(400).send({ error: 'Wrong password' });
  }

  const accessToken = jwt.sign({ userId: user.id, email: user.email, role: user.role }, 'SECRET', { expiresIn: '15m' });
  const refreshToken = jwt.sign({ userId: user.id }, 'SECRET', { expiresIn: '7d' });

  res.cookie('refresh_token', refreshToken, { httpOnly: true });
  res.send({ accessToken });
});

router.get('/me', requireAuth, (req, res) => {
  res.send(req.user);
});

function requireAuth(req, res, next) {
  const { authorization } = req.headers;
  if (!authorization) {
    return res.status(403).send({ error: 'Not authenticated' });
  }
  
  const token = authorization.replace('Bearer ', '');
  jwt.verify(token, 'SECRET', (err, payload) => {
    if (err) {
      return res.status(403).send({ error: 'Invalid token' });
    }
    
    req.user = payload;
    next();
  });
}

export default router;
```

In this code, the `register` function hashes the password using bcrypt before saving a new user to the database. The `login` function validates the email and password, generates JWT tokens and sends the access token. The `me` route is protected by the `requireAuth` middleware, which validates the access token and attaches the user to the request.

If an expired token is sent, the `requireAuth` middleware will return a 403 error with the message "Invalid token".

The httpOnly cookie for the refresh token is used to prevent it from being accessed by client-side JavaScript, improving the security of the refresh token. If it were stored in localStorage, an attacker could manipulate the refresh token if the user has enabled third-party cookies.

3. Redis data structure and TTL strategy:
Refresh tokens can be stored in a Redis set with the user_id as the member and the refresh token as the set key. The TTL (Time to Live) of each set can be set to a reasonable expiration time (e.g., 30 days).

4. Token Reuse Detection:
A Set in Redis can be used to keep track of issued refresh tokens. Each time a refresh token is used, it can be added to the set. If a refresh token is attempted to be used again, it can be checked if it's in the set. If it is, the request can be rejected.

5. Invalidate all existing sessions:
When a user changes their password, all their existing refresh tokens can be invalidated. This can be done by deleting all refresh tokens associated with the user in the Redis store.

6. Token Blacklisting vs Token Rotation:
Token blacklisting is a strategy where tokens are invalidated after they are used. If a token is used, it's added to a blacklist and all future requests with that token are rejected.

Token rotation is a strategy where tokens are rotated after they are used. A new token is issued and the old token is invalidated. This strategy is often used in case a token is compromised and it's better to invalidate it as soon as possible. It's less strict but more flexible than blacklisting.

For token rotation, we have a more reactive approach to handle token misuse. But, it may lead to more frequent token changes which could potentially impact the user experience, while blacklisting might lead to a user being blocked even before they have a chance to react to the potential danger. The choice between blacklisting and rotation depends on the specific requirements and constraints of the system.

## Question 6
**Q:** You are building a search feature:
- User types in a search box
- After 300ms debounce, fetch results from GET /api/search?q=...
- Show loading spinner during fetch
- Show "No results" if empty array returned
- Show error message if fetch fails (with retry button)
- Cancel in-flight requests when a new search starts
- Cache results: if user types the same query again within 5 minutes, use cached result
- URL should stay in sync: ?q=searchterm (browser back/forward should work)

Implement this in React using hooks only (no external state library). Show:
1. Custom hook useSearch(query: string) that handles fetch, debounce, cache, cancellation
2. The SearchPage component using this hook
3. URL sync implementation
4. Why is cancelling in-flight requests important here? What bug does it prevent?

**A:**
```jsx
import React, { useState, useEffect, useCallback } from 'react';
import { useLocation, useHistory } from 'react-router-dom';

let controller = null;
const CACHE_TTL = 5 * 60 * 1000; // 5 minutes in ms
const cache = {};

function useSearch(query) {
  const [state, setState] = useState({ status: 'idle' });
  const doSearch = useCallback(async () => {
    if (controller) {
      controller.abort();
    }
    controller = new AbortController();
    try {
      setState({ status: 'loading' });
      const res = await fetch(`/api/search?q=${query}`, { signal: controller.signal });
      const data = await res.json();
      setState({ status: 'success', data });
    } catch (error) {
      if (error.name === 'AbortError') {
        setState({ status: 'idle' });
      } else {
        setState({ status: 'error', error });
      }
    }
  }, [query]);

  useEffect(() => {
    const cached = cache[query];
    if (cached && Date.now() - cached.timestamp < CACHE_TTL) {
      setState({ status: 'success', data: cached.data });
    } else {
      doSearch();
    }
  }, [doSearch, query]);

  return state;
}

function SearchPage() {
  const location = useLocation();
  const history = useHistory();
  const [search, setSearch] = useState(location.search);
  const { status, data, error } = useSearch(search);

  useEffect(() => {
    const handlePopState = () => {
      setSearch(location.search);
    };
    window.addEventListener('popstate', handlePopState);
    return () => {
      window.removeEventListener('popstate', handlePopState);
    };
  }, [location.search]);

  const doSearch = (value) => {
    setSearch(value);
    history.push({ search: value });
  };

  return (
    <div>
      <input onChange={e => doSearch(e.target.value)} value={search} />
      {status === 'loading' && <div>Loading...</div>}
      {status === 'error' && <div>Error: {error.message} <button onClick={doSearch}>Retry</button></div>}
      {status === 'success' && data.length === 0 && <div>No results</div>}
      {status === 'success' && data.length > 0 && <div>{data.map(d => <div key={d.id}>{d.name}</div>)}</div>}
    </div>
  );
}
```

Cancelling in-flight requests is important here to prevent race conditions. If the user starts a new search while an old one is in progress, we want to cancel the old search and start a new one. Without cancellation, we might end up displaying results from the old search instead of the new one.

This can cause a bug where, for example, a user searches for "cat", then quickly searches for "dog". Before "dog" is finished loading, the "cat" results are displayed. This is especially common with fast-typing users or very fast devices with a fast internet connection.

By cancelling the old request, we ensure that the results displayed are always from the most recent search.

3. Broadcasting — when user A makes a change, how do you send it to all other users in the same document but NOT back to user A?
4. Reconnection — how do you handle a user rejoining after disconnect?
5. What happens if two users edit the exact same position simultaneously? You don't need to implement OT/CRDT — just explain the problem and name the solution approaches.

Write the core server implementation (connection map, message routing, broadcast logic).

**A:**
```javascript
const WebSocket = require('ws');
const assert = require('assert');

const wss = new WebSocket.Server({ port: 8080 });

// Store all connections. Key is documentId
const documents = new Map();

// Store each user's last known state. Key is WebSocket connection
const users = new Map();

wss.on('connection', function(ws) {
    const userId = Math.random().toString(36).substr(2, 9); // Generate random userId
    users.set(ws, {userId, documentId: null});

    ws.on('message', function(message) {
        const msg = JSON.parse(message);
        assert(msg.type, 'No message type specified');

        switch (msg.type) {
            case 'joinDocument':
                assert(msg.documentId, 'No documentId specified');
                users.get(ws).documentId = msg.documentId;

                if (!documents.has(msg.documentId)) {
                    documents.set(msg.documentId, new Set());
                }
                documents.get(msg.documentId).add(ws);

                // Send cursor position to new user
                documents.get(msg.documentId).forEach(ws => {
                    if (ws.readyState === ws.OPEN) {
                        const user = users.get(ws);
                        ws.send(JSON.stringify({
                            type: 'cursorPosition',
                            userId: user.userId,
                            position: user.cursorPosition
                        }));
                    }
                });
                break;
            case 'textChange':
                assert(msg.position && msg.text, 'Invalid text change message');
                // Broadcast text change message
                broadcast({
                    type: 'textChange',
                    position: msg.position,
                    text: msg.text
                }, ws);
                break;
            case 'cursorMove':
                assert(msg.position, 'Invalid cursor move message');
                users.get(ws).cursorPosition = msg.position;

                // Broadcast cursor move message
                broadcast({
                    type: 'cursorPosition',
                    userId,
                    position: msg.position
                }, ws);
                break;
            case 'userLeft':
                // Clean up user and document state
                documents.get(users.get(ws).documentId).delete(ws);
                users.delete(ws);
                break;
            default:
                console.log(`Unknown message type: ${msg.type}`);
        }
    });

    ws.on('close', function() {
        ws.send(JSON.stringify({type: 'userLeft'}));
    });
});

function broadcast(msg, exclude) {
    const documentId = users.get(exclude).documentId;
    documents.get(documentId).forEach(ws => {
        if (ws !== exclude && ws.readyState === ws.OPEN) {
            ws.send(JSON.stringify(msg));
        }
    });
}
```

1. Each WebSocket connection is associated with a user and document through the `users` map, which stores user metadata indexed by WebSocket connection.
2. The message protocol is defined in JSON format. For example, a text change message looks like `{type: 'textChange', position: {row, col}, text: '...'}`.
3. When user A makes a change, the server broadcasts the message to all other users in the same document, excluding user A.
4. If a user reconnects, they will receive the latest document state by having the server re-send their user's text changes and cursor positions.
5. If two users edit the exact same position simultaneously, there's a race condition. The server will receive the two changes, and both will be applied. If this causes a conflict, it's up to the client to handle this, such as by undoing the last change. The users will see the other's changes, and if they disagree, they can manually resolve the conflict.
